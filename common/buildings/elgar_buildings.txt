#Opera House

building_elgar_opera = {

	building_group = bg_opera

	city_type = city

	levels_per_mesh = 5

	expandable = yes

	unlocking_technologies = {
		elgar_classicism_tech
	}

	production_method_groups = {
		pmg_elgar_orchestra
		pmg_elgar_ballet_school
	}

	can_build_government = {
		custom_tooltip = {
			text = elgar_opera_can_construct_tt
			OR = {
				any_scope_building = {
					is_building_type = building_elgar_opera
				}
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 5
				}
			}
		}
	}

	can_build_private = {
		custom_tooltip = {
			text = elgar_opera_can_construct_tt
			OR = {
				any_scope_building = {
					is_building_type = building_elgar_opera
				}
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 5
				}
			}
		}
	}

	texture = "gfx/interface/icons/building_icons/building_elgar_opera.dds"

	required_construction = construction_cost_medium

	ownership_type = self

	background = "gfx/interface/icons/building_icons/backgrounds/building_panel_bg_monuments.dds"
}
