	type = country

		trigger = {
			has_variable = gaudi_engineer_var

	}

		localization_key = gaudi_engineer_no_engineer_loc
	}
}

GUI_gaudi_civil_engineer_experience_type = {
	type = country

	text = {
		trigger = {
			NOT = { has_variable = gaudi_engineer_var }
		}
		localization_key = GUI_gaudi_civil_engineer_experience_no_engineer_loc
	}

	text = {
		trigger = {
			var:gaudi_engineer ?= {
				var:gaudi_civil_engineer_experience >= 300
			}
		}
		localization_key = GUI_gaudi_civil_engineer_experience_finished_loc
	}

	text = {
		trigger = {
			var:gaudi_engineer ?= {
				var:gaudi_civil_engineer_experience >= 100
				var:gaudi_civil_engineer_experience < 300
			}
		}
		localization_key = GUI_gaudi_civil_engineer_experience_trait_loc
	}

	text = {
		localization_key = GUI_gaudi_civil_engineer_experience_ongoing_loc
	}
}
