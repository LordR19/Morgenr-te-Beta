#Scientific Society Name: Will pick  either a player set name (not finished yet) or if the player didn't set one the generic name.
#The generic name will have historical names as well for some countries.
academics_scientific_society_name_custom_loc = {
	type = country
	random_valid = no #From top to bottom decides which name list is appropriate.

	text = {
		localization_key = academics_scientific_society_generic_name
	}

}

#Engineer: Informas whether there is an Engineer or not.
academics_scientific_society_has_engineer_custom_loc = {
	type = country
	random_valid = no #From top to bottom decides which name list is appropriate.
	
	text = {
		trigger = {
			has_variable = gaudi_engineer_var
		}
		localization_key = academics_scientific_society_has_engineer
	}
	
	text = {
		trigger = {
			NOT = {
				has_variable = gaudi_engineer_var
			}
		}
		localization_key = academics_scientific_society_no_engineer
	}

}

#Replacement Tooltip for the Engineer --> Used to show whether you will replace an existing Engineer
academics_replacement_engineer_custom_loc = {
	type = country
	random_valid = no #From top to bottom decides which name list is appropriate.
	
	text = {
		trigger = {
			has_variable = gaudi_engineer_var
		}
		localization_key = academics_would_replace_engineer_loc
	}
	
	text = {
		trigger = {
			NOT = {
				has_variable = gaudi_engineer_var
			}
		}
		localization_key = academics_will_not_replace_engineer_loc
	}
