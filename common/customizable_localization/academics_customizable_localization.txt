#Engineer: Informas whether there is an Engineer or not.
academics_scientific_society_has_engineer_custom_loc = {
	type = country
	random_valid = no #From top to bottom decides which name list is appropriate.
	
	text = {
		trigger = {
			has_variable = gaudi_engineer_var
		}
		localization_key = academics_scientific_society_has_engineer
	}
	
	text = {
		trigger = {
			NOT = {
				has_variable = gaudi_engineer_var
			}
		}
		localization_key = academics_scientific_society_no_engineer
	}

}
