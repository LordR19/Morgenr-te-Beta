#Replacement Tooltip for the Engineer --> Used to show whether you will replace an existing Engineer
academics_replacement_engineer_custom_loc = {
	type = country
	random_valid = no #From top to bottom decides which name list is appropriate.
	
	text = {
		trigger = {
			has_variable = gaudi_engineer_var
		}
		localization_key = academics_would_replace_engineer_loc
	}
	
	text = {
		trigger = {
			NOT = {
				has_variable = gaudi_engineer_var
			}
		}
		localization_key = academics_will_not_replace_engineer_loc
	}

}
