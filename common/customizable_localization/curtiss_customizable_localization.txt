#Balloon names for different cultures/countries
curtiss_balloon_type = {
	type = country
	
	text = {
		trigger = {
			OR = {
				country_has_primary_culture = cu:swiss
				AND = {
					exists = c:SWI
					c:SWI = root
				}
			}
		}
		localization_key = curtiss_custom_swiss_balloon_name
	}

	text = {
		trigger = {
			country_has_primary_culture = cu:french
			AND = {
				exists = c:FRA
				c:FRA = root
			}
		}
		localization_key = curtiss_custom_french_balloon_name
	}

	text = {
		trigger = {
			OR = {
				country_has_primary_culture = cu:north_italian
				country_has_primary_culture = cu:south_italian
				AND = {
					exists = c:ITA
					c:ITA = root
				}
			}
		}
		localization_key = curtiss_custom_italian_balloon_name
	}

	text = {
		trigger = {
			OR = {
				country_has_primary_culture = cu:english
				AND = {
					exists = c:GBR
					c:GBR = root
				}
			}
		}
		localization_key = curtiss_custom_english_balloon_name
	}

	text = {
		trigger = {
			OR = {
				country_has_primary_culture = cu:north_german
				country_has_primary_culture = cu:south_german
				AND = {
					exists = c:GER
					c:GER = root
				}
			}
		}
		localization_key = curtiss_custom_german_balloon_name
	}

	text = {
		localization_key = curtiss_custom_generic_balloon_name
	}

}

		}
		localization_key = curtiss_custom_death_sea
	}

	text = {
		localization_key = curtiss_custom_death_hydrogen_ignition
	}

	text = {
		localization_key = curtiss_custom_death_firework
	}

	text = {
		localization_key = curtiss_custom_death_redneck
	}

	text = {
		localization_key = curtiss_custom_death_comet
	}

		localization_key = curtiss_custom_death_stork

	}

	text = {
		trigger = {
			is_at_war = yes
		}
		localization_key = curtiss_custom_death_stray_bullet
	}

	text = {
		trigger = {
			mg:electricity = {
				market_goods_production > 200

		}
		localization_key = curtiss_custom_death_power_line
	}

	text = {
		trigger = {
			any_scope_state = {

	text = {

	text = {

	}

	text = {
		localization_key = curtiss_custom_death_updraft
	}

	text = {
		trigger = {

	text = {
		localization_key = curtiss_custom_death_failed_fixing

	text = {
		localization_key = curtiss_custom_death_heart_attack
	}

	text = {
		localization_key = curtiss_custom_death_balloon_crash
	}

	text = {
		trigger = {
			any_country = {
				has_strategic_land_adjacency = root

			}

	}
