#National Writer is a great writer
manzoni_yearly_literary_tradition_national_writer_great_tt_type = {
	type = country
	
	text = {
		trigger = {
			var:manzoni_writer_var ?= {
				OR = {
					has_trait = manzoni_great_writer
					var:manzoni_writer_experience >= 500
				}
			}
		}
		localization_key = manzoni_yearly_literary_tradition_national_writer_great_tt_loc
	}

	text = {
		localization_key = manzoni_yearly_literary_tradition_national_writer_great_inactive_tt_loc
	}
}
