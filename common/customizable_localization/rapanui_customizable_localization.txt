#Expedition Ship names for different cultures/countries
rapanui_expedition_ship_type = { #From top to bottom decides which name list is appropriate. E.g. is a country with French as primary culture --> French names
	type = country

	text = {
		trigger = {
			OR = {
				country_has_primary_culture = cu:north_italian
				country_has_primary_culture = cu:south_italian
				country_has_primary_culture = cu:maltese
				AND = {
					exists = c:ITA
					c:ITA = root
				}
			}
		}
		localization_key = rapanui_custom_italian_expedition_ship_name
	}

	text = {
		localization_key = rapanui_custom_generic_expedition_ship_name
	}
}
