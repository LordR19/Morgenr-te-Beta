je_curtiss_found_schneider_trophy = {
	icon = "gfx/interface/icons/event_icons/event_aviation.dds"

	group = je_group_curtiss_aviation

	immediate = {
		set_variable = {
			name = je_curtiss_found_schneider_trophy_var
			days = 1
		}
	}

	complete = {
		scope:journal_entry = { 
			is_goal_complete = yes 
		}
		any_scope_state = {
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_port
			}
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 15
			}
		}
		hidden_trigger = {
			NOT = { has_variable = je_curtiss_found_schneider_trophy_var }
		}
	}

	on_complete = {
		trigger_event = { 
			id = curtiss.100
			popup = yes
		}
	}

	fail = {
	}

	on_fail = {
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:curtiss_seaplane_nations #value or variable you use to meassure progress
		}
	}

	goal_add_value = {
		value = 5
	}

	progressbar = yes

	weight = 25
}

	icon = "gfx/interface/icons/event_icons/event_aviation.dds"

	group = je_group_curtiss_aviation

	is_shown_when_inactive = {
		has_technology_researched = military_aviation
		NOR = {
			has_law = law_type:law_agrarianism 
			has_law = law_type:law_traditionalism 
			has_law = law_type:law_cooperative_ownership
			has_law = law_type:law_command_economy 	
		}
		NOR = {
			has_variable = curtiss_private_airline
			has_variable = curtiss_state_airline
		}
    }

    possible = {
        has_technology_researched = curtiss_modern_aviation_tech
		OR = { 
			has_law = law_type:law_laissez_faire
			has_law = law_type:law_interventionism
		}
		exists = ig:ig_industrialists
    }

	complete = {
		any_scope_state = {
			any_scope_building = {
				is_building_type = building_airport
				level >= 2
			}
		}
		ig:ig_industrialists = {
			is_marginal = no
		}
	}

	on_complete = {
		set_variable = {
            name = curtiss_flight_route
            value = 1
		}
		trigger_event = { 
			id = curtiss.20
			popup = yes
		}
	}

	invalid = {
		OR = {
			has_law = law_type:law_agrarianism 
			has_law = law_type:law_traditionalism 
			has_law = law_type:law_cooperative_ownership

}
