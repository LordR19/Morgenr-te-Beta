je_animal_breeding = {	

	group = je_group_dubois_nature

	on_yearly_pulse = {
		random_events = {
			5 = dubois.1311 #Dubois 1311: Hidden Event to calculate change for the animal population
		}
	}
	
	immediate = {
		set_variable = {
			name = dubois_breeding_animal_amount_var
			value = 20
		}
	}
	
    fail = {
		custom_tooltip = {
			text = je_breeding_programm_failed_tt
			var:dubois_breeding_animal_amount_var <= 1
		}
    }
	
	on_fail = {
		trigger_event = {
			id = dubois.1322
			popup = yes
		}
	}
	
	complete = {
		custom_tooltip = {
			text = je_breeding_programm_succesfull_tt
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
	}
	
	on_complete = {
		trigger_event = {
			id = dubois.1321
			popup = yes
		}
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:dubois_breeding_animal_amount_var #value or variable you use to meassure progress
		}
	}

	goal_add_value = {
		value = 4000
	}

	progressbar = yes
	
    weight = 200
    should_be_pinned_by_default = yes
}

###PALEONTOLOGICAL EXPEDITION###

je_dubois_paleontological_expedition = {
	icon = "gfx/interface/icons/event_icons/event_fossil.dds"

	group = je_group_dubois_nature

	on_yearly_pulse = {
		random_events = {
			chance_to_happen = 100
			chance_of_no_event = {
				value = 80
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							var:dubois_paleontologist_experience >= 100
						}
					}
					subtract = 2
				}
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							var:dubois_paleontologist_experience >= 200
						}
					}
					subtract = 3
				}
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							var:dubois_paleontologist_experience >= 300
						}
					}
					subtract = 5
				}
			}
			
			#Random Formation Events
			30 = dubois.202 #Dubois 202: Shark Tooth
			20 = dubois.203 #Dubois 203: Dinosaur Footprints
			20 = dubois.204 #Dubois 204: Palaeosinopa
			20 = dubois.205 #Dubois 205: Bones in the mineshaft
			15 = dubois.206 #Dubois 206: Composita
			15 = dubois.207 #Dubois 207: Trilobite
			15 = dubois.208 #Dubois 208: Ammonites
			30 = dubois.209 #Dubois 209: Farns
			5 = dubois.210 #Dubois 210: Dinosauria Eggs

			#Special Fossils

			10 = dubois.221 #Dubois 221: Eusthenopteron
			10 = dubois.222 #Dubois 222: Archaeopteryx
			10 = dubois.223 #Dubois 223: Megatherium
			10 = dubois.224 #Dubois 224: Ichthyosaurus
			10 = dubois.225 #Dubois 225: Smilodon
			10 = dubois.226 #Dubois 226: Pterodactylus

			#Human Fossils

			10 = dubois.251 #Dubois 251: Unknown Human Bone (Neanderthalensis)
			5 = dubois.252 #Dubois 252: Unknown Human Bone (Erectus)
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOR = { #The Progress should only be granted if the paleontologist is not currently travelling or excavating
						has_modifier = dubois_paleontologist_travell_to_location_modifier
						has_journal_entry = je_dubois_dinosaur_excavation
					}
				}

				#Triggers the event if the excavation progress is big enough
				if = {
					limit = {
						dubois_dinosaur_excavation_finding_progress >= dubois_dinosaur_excavation_finding_progress_goal_value
						dubois_has_potential_dinosaur_bone = yes
					}
					change_variable = {
						name = dubois_dinosaur_excavation_finding_progress
						subtract = dubois_dinosaur_excavation_finding_progress_goal_value
					}
					trigger_event = dubois.211
				}

				#The monthly amount gets added
				change_variable = {
					name = dubois_dinosaur_excavation_finding_progress
					add = 5 #TODO change this to a script_value
				}

				#Triggers the event if the excavation progress is big enough
				if = {
					limit = {
						dubois_dinosaur_excavation_finding_progress >= dubois_dinosaur_excavation_finding_progress_goal_value
						dubois_has_potential_dinosaur_bone = yes
					}
					change_variable = {
						name = dubois_dinosaur_excavation_finding_progress
						subtract = dubois_dinosaur_excavation_finding_progress_goal_value
					}
					trigger_event = dubois.211
				}
			}
		}
	}
	
	immediate = {
		var:dubois_paleontologist_var = {
			save_scope_as = dubois_paleontologist_scope
		}
		if = {
			limit = {
				NOT = { has_variable = dubois_dinosaur_excavation_finding_progress }
			}
			set_variable = {
				name = dubois_dinosaur_excavation_finding_progress
				value = 0
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_dubois_paleontological_expedition_goal
			always = no
		}
	}

	on_complete = {
	}

	invalid = {
		OR = {
			dubois_has_paleontologist = no
			NOT = {
				has_modifier = dubois_paleontological_research_modifier
			}
		}
	}

	current_value = {
		value = var:dubois_dinosaur_excavation_finding_progress
	}

	goal_add_value = {
		value = dubois_dinosaur_excavation_finding_progress_goal_value
	}

	progressbar = yes

	should_be_pinned_by_default = yes

	weight = 25
}
