###PALEONTOLOGICAL EXPEDITION###

je_dubois_paleontological_expedition = {
	
	##################################################################################################################################
	# !!! Warning: The AI uses another system. Check ai_paleontology_yearly_effect and ai_paleontology_monthly_effect for the AI !!! #
	##################################################################################################################################

	icon = "gfx/interface/icons/event_icons/event_fossil.dds"

	group = je_group_dubois_nature

	on_yearly_pulse = {
		effect = {
			#Random Special Fossil Events (Makes sure they get executed randomly during the year
			trigger_event = {
				on_action  = dubois_on_yearly_pulse_special_fossil_events
				months = { 1 11 }
			}

			#Random Progress Events (Makes sure they get executed randomly during the year
			if = {
				limit = { dubois_has_potential_dinosaur_bone = yes }
				trigger_event = {
					on_action  = dubois_on_yearly_pulse_paleontological_expedition
					months = { 1 11 }
				}
			}

			#Paleontological Report: adds regular fossils and notifies the player about the total amount of yearly fossils.
			dubois_add_x_fossils_effect = { AMOUNT = dubois_yearly_regular_fossils_value }
			#Sets the variable for the notification
			if = {
				limit = {
					has_variable = dubois_yearly_fossils_var
				}
				set_variable = {
					name = dubois_fossils_notification_var
					value = var:dubois_yearly_fossils_var
					months = 2
				}
				set_variable = {
					name = dubois_yearly_fossils_var
					value = 0
				}
			}
			#Actual Notification
			post_notification = dubois_paleontological_report_notification
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOR = { #The Progress should only be granted if the paleontologist is not currently travelling or excavating
						has_modifier = dubois_paleontologist_travell_to_location_modifier
						has_journal_entry = je_dubois_dinosaur_excavation
					}
				}

				#Triggers the event if the excavation progress is big enough
				if = {
					limit = {
						var:dubois_dinosaur_excavation_finding_progress >= dubois_dinosaur_excavation_finding_progress_goal_value
						dubois_has_potential_dinosaur_bone = yes
					}
					change_variable = {
						name = dubois_dinosaur_excavation_finding_progress
						subtract = dubois_dinosaur_excavation_finding_progress_goal_value
					}
					trigger_event = dubois.211
				}

				#The monthly amount gets added
				set_variable = {
					name = dubois_dinosaur_excavation_finding_estimated_progress
					value = dubois_dinosaur_excavation_find_monthly_progress_value
				}
				if = {
					limit = { dubois_has_potential_dinosaur_bone = yes }
					change_variable = {
						name = dubois_dinosaur_excavation_finding_progress
						add = dubois_dinosaur_excavation_find_monthly_progress_value
					}
				}

				#Triggers the event if the excavation progress is big enough
				if = {
					limit = {
						var:dubois_dinosaur_excavation_finding_progress >= dubois_dinosaur_excavation_finding_progress_goal_value
						dubois_has_potential_dinosaur_bone = yes
					}
					change_variable = {
						name = dubois_dinosaur_excavation_finding_progress
						subtract = dubois_dinosaur_excavation_finding_progress_goal_value
					}
					trigger_event = dubois.211
				}
			}

			#estimates the amount of fossils that will be found this year
			set_variable = {
				name = dubois_estimated_regular_fossils_year_var
				value = dubois_yearly_regular_fossils_value
				days = 40
			}
		}
	}
	
	immediate = {
		var:dubois_paleontologist_var = {
			save_scope_as = dubois_paleontologist_scope
		}
		set_variable = {
			name = dubois_dinosaur_excavation_finding_estimated_progress
			value = dubois_dinosaur_excavation_find_monthly_progress_value
		}
		if = {
			limit = {
				NOT = { has_variable = dubois_dinosaur_excavation_finding_progress }
			}
			set_variable = {
				name = dubois_dinosaur_excavation_finding_progress
				value = 0
			}
		}
	}

	complete = {
		trigger_if = {
			limit = { dubois_has_potential_dinosaur_bone = yes }
			custom_tooltip = {
				text = je_dubois_paleontological_expedition_goal_tt
				always = no
			}
		}
		custom_tooltip = {
			text = je_dubois_paleontological_expedition_invalid_loc
			always = no
		}
	}

	on_complete = {
	}

	invalid = {
		OR = {
			dubois_has_paleontologist = no
			NOT = {
				has_modifier = dubois_paleontological_research_modifier
			}
		}
	}

	on_invalid = {
		remove_variable = dubois_dinosaur_excavation_finding_estimated_progress
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:dubois_dinosaur_excavation_finding_progress #value or variable you use to meassure progress
		}
	}

	goal_add_value = {
		value = dubois_dinosaur_excavation_finding_progress_goal_value
	}

	progressbar = yes

	should_be_pinned_by_default = yes

	weight = 25
}

###Improve Forestry###
je_dubois_improve_forestry_project = {
	icon = "gfx/interface/icons/event_icons/event_botany.dds"

	scripted_button = dubois_stop_naturalist_project_button

	group = je_group_dubois_nature

	on_monthly_pulse = {
		effect = {
			#Calculates the Progress added (see dubois_scripted_effects for calculation)
			dubois_naturalist_improvement_progress = yes
			#Adds the monthly progress to the variable
			change_variable = {
				name = dubois_naturalist_improvement_progress_var
				add = var:dubois_monthly_naturalist_improvement_progress_var
			}
			if = {
				limit = { var:dubois_naturalist_improvement_progress_var > dubois_improvement_progress_cost }
				if = {
					limit = {
						any_scope_building = {
							has_modifier = dubois_naturalist_focus_building_modifier
							NOT = { has_modifier = dubois_building_throughput_add_modifier }
						}
					}
					random_scope_building = {
						limit = {
							has_modifier = dubois_naturalist_focus_building_modifier
							NOT = { has_modifier = dubois_building_throughput_add_modifier }
						}
						add_modifier = {
							name = dubois_building_throughput_add_modifier
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = dubois_improvement_progress_cost
					}
				}
				else_if = {
					limit = {
						any_scope_building = {
							has_modifier = dubois_naturalist_focus_building_modifier
							NOT = { has_modifier = dubois_building_production_mult_modifier }
						}
					}
					random_scope_building = {
						limit = {
							has_modifier = dubois_naturalist_focus_building_modifier
							NOT = { has_modifier = dubois_building_production_mult_modifier }
						}
						add_modifier = {
							name = dubois_building_production_mult_modifier
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = dubois_improvement_progress_cost
					}
				}
				else = {
					random_scope_building = {
						limit = {
							has_modifier = dubois_naturalist_focus_building_modifier
							NOT = { has_modifier = dubois_building_goods_input_mult_modifier }
						}
						add_modifier = {
							name = dubois_building_goods_input_mult_modifier
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = dubois_improvement_progress_cost
					}
				}
			}
		}
	}

	immediate = {
		set_variable = dubois_can_stop_active_improvement_project_var #This is used to check if the country has a project already
		ordered_scope_building = {
			limit = {
				level > 0
				is_building_type = building_logging_camp
				NOR = {
					has_modifier = dubois_building_throughput_add_modifier
					has_modifier = dubois_building_production_mult_modifier
					has_modifier = dubois_building_goods_input_mult_modifier
				}
			}
			order_by = earnings
			add_modifier = {
				name = dubois_naturalist_focus_building_modifier
			}
			root = {
				set_variable = {
					name = dubois_building_naturalist_improvement_var
					value = prev
				}
			}
		}
		#Sets the Naturalist Improvement Progress at the beginning
		set_variable = {
			name = dubois_naturalist_improvement_progress_var
			value = 0
		}
		dubois_naturalist_improvement_progress = yes #Determines how much Naturalist Improvement Progress will be added next month
	}

	complete = {
		custom_tooltip = {
			text = dubois_building_has_throughput_modifier_tt
			any_scope_building = {
				has_modifier = dubois_naturalist_focus_building_modifier
				has_modifier = dubois_building_throughput_add_modifier
			}
		}
		custom_tooltip = {
			text = dubois_building_has_production_modifier_tt
			any_scope_building = {
				has_modifier = dubois_naturalist_focus_building_modifier
				has_modifier = dubois_building_production_mult_modifier
			}
		}
		custom_tooltip = {
			text = dubois_building_has_input_modifier_tt
			any_scope_building = {
				has_modifier = dubois_naturalist_focus_building_modifier
				has_modifier = dubois_building_goods_input_mult_modifier
			}
		}
	}

	on_complete = {
		remove_variable = dubois_can_stop_active_improvement_project_var #Needs to be removed, when the JE is removed
		dubois_remove_je_cost_modifiers_effect = yes
		
		#Those should be removed in every Naturalist je
		remove_variable = dubois_naturalist_improvement_progress_var
		remove_variable = dubois_monthly_naturalist_improvement_progress_var
		remove_variable = naturalist_experience_add_var

		every_scope_building = {
			limit = {
				has_modifier = dubois_naturalist_focus_building_modifier
			}
			remove_modifier = dubois_naturalist_focus_building_modifier
		}
		var:dubois_naturalist_var ?= {
			set_character_busy_and_immortal = no
		}
		trigger_event = {
			id = dubois.52
			popup = yes
		}
	}

	current_value = {
		#Keeps track of the Naturalist Progress. If the Progress reaches the progress_cost, a modifier will be added.
		value = var:dubois_naturalist_improvement_progress_var
	}

	goal_add_value = {
		#You can change this in dubois_script_values
		value = dubois_improvement_progress_cost
	}

	invalid = {
		custom_tooltip = {
			text = dubois_stop_naturalist_improving_project_tt
			has_variable = dubois_stop_naturalist_improving_project_var
		}
	}

	on_invalid = {
		remove_variable = dubois_can_stop_active_improvement_project_var #Needs to be removed, when the JE is removed
		dubois_remove_je_cost_modifiers_effect = yes
		
		#Those should be removed in every naturalist je
		remove_variable = dubois_naturalist_improvement_progress_var
		remove_variable = dubois_monthly_naturalist_improvement_progress_var
		remove_variable = naturalist_experience_add_var

		every_scope_building = {
			limit = {
				has_modifier = dubois_naturalist_focus_building_modifier
			}
			remove_modifier = dubois_naturalist_focus_building_modifier
		}
		remove_variable = dubois_stop_naturalist_improving_project_var
		var:dubois_naturalist_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	progressbar = yes

	weight = 30

	should_be_pinned_by_default = yes
}

						}
						add_modifier = {
							name = dubois_building_production_mult_modifier
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = dubois_improvement_progress_cost
					}
				}
				else = {
					random_scope_building = {
						limit = {
							has_modifier = dubois_naturalist_focus_building_modifier
							NOT = { has_modifier = dubois_building_goods_input_mult_modifier }
						}
						add_modifier = {
							name = dubois_building_goods_input_mult_modifier
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = dubois_improvement_progress_cost
					}
				}
			}
		}
	}

	immediate = {
		set_variable = dubois_can_stop_active_improvement_project_var #This is used to check if the country has a project already
		ordered_scope_building = {
			limit = {
				level > 0
				is_building_type = building_rubber_plantation
				NOR = {
					has_modifier = dubois_building_throughput_add_modifier
					has_modifier = dubois_building_production_mult_modifier
					has_modifier = dubois_building_goods_input_mult_modifier
				}
			}
			order_by = earnings
			add_modifier = {
				name = dubois_naturalist_focus_building_modifier
			}
			root = {
				set_variable = {
					name = dubois_building_naturalist_improvement_var
					value = prev
				}
			}
		}
		#Sets the Naturalist Improvement Progress at the beginning
		set_variable = {
			name = dubois_naturalist_improvement_progress_var
			value = 0
		}
		dubois_naturalist_improvement_progress = yes #Determines how much Naturalist Improvement Progress will be added next month
	}

	complete = {

			text = dubois_building_has_throughput_modifier_tt

				has_modifier = dubois_naturalist_focus_building_modifier
