#Impressionism

je_elgar_impressionism = {
	icon = "gfx/interface/icons/event_icons/event_impressionism.dds"

	group = je_group_elgar_culture

	is_shown_when_inactive = {
		NOT = {
			has_technology_researched = elgar_impressionism_tech
			has_technology_researched = elgar_expressionism_tech
			has_technology_researched = elgar_art_nouveau_tech
		}
	}

	possible = {
		has_technology_researched = elgar_classicism_tech
	}

	immediate = {
		set_variable = {
			name = je_elgar_impressionism_var
			days = 1
		}
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:elgar_music_tradition #value or variable you use to meassure progress
		}
	}

	goal_add_value = {
		value = 850
	}

	complete = {
		any_scope_building = {
			is_building_type = building_elgar_opera
		}
		has_technology_researched = elgar_irrationalism_tech
		has_technology_researched = realism
		ig:ig_landowners = {
			is_powerful = yes
		}
		NOT = {
			ig:ig_industrialists = {
				is_powerful = yes
			}
		}
		ig:ig_intelligentsia = {
			ig_approval >= 5
	    }
		custom_tooltip = {
			text = je_elgar_impressionism_success
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
		hidden_trigger = {
			NOT = { has_variable = je_elgar_impressionism_var }
		}
	}

	on_complete = {
		add_technology_researched = elgar_impressionism_tech
	}

	fail = {
		OR = {
			has_technology_researched = elgar_expressionism_tech
			has_technology_researched = elgar_art_nouveau_tech
		}
	}

	invalid = {
		has_technology_researched = elgar_impressionism_tech
	}

	progressbar = yes

	weight = 50
}

#Expressionism

je_elgar_expressionism = {
	icon = "gfx/interface/icons/event_icons/event_expressionism.dds"

	group = je_group_elgar_culture

	is_shown_when_inactive = {
		NOT = {
			has_technology_researched = elgar_impressionism_tech
			has_technology_researched = elgar_expressionism_tech
			has_technology_researched = elgar_art_nouveau_tech
		}
	}

	possible = {
		has_technology_researched = elgar_classicism_tech
	}

	immediate = {
		set_variable = {
			name = je_elgar_expressionism_var
			days = 1
		}
	}

	current_value = {
		value = var:elgar_music_tradition
	}

	goal_add_value = {
		value = 850 #value you want the variable to reach
		subtract = var:elgar_music_tradition #current value of the variable
		if = { #prevents the value of the goal_add_value to be below one
			limit = {
				var:elgar_music_tradition >= 850
			}
			value = 1
		}
	}

	complete = {
		any_scope_building = {
			is_building_type = building_elgar_opera
		}
		has_technology_researched = elgar_irrationalism_tech
		has_technology_researched = realism
		OR = {
			ig:ig_intelligentsia = {
				is_powerful = yes
			}
			ig:ig_trade_unions = {
				is_powerful = yes
			}
		}
		NOT = {
			ig:ig_landowners = {
				is_powerful = yes
			}
		}
		OR = {
			ig:ig_intelligentsia = {
				ig_approval >= 5
			}
			ig:ig_trade_unions = {
				ig_approval >= 5
			}
		}

		}

			NOT = { has_variable = je_elgar_expressionism_var }

	fail = {

			has_technology_researched = elgar_art_nouveau_tech
		}
	}

	invalid = {
		has_technology_researched = elgar_expressionism_tech
	}

	progressbar = yes

	weight = 50
}

		random_events = {
			340 = elgar.421 #Elgar 421: Composer progress

			10 = elgar.412 #Elgar 412: Mediocre Concerto
			10 = elgar.413 #Elgar 413: Poor Concerto
			10 = elgar.414 #Elgar 414: Ambassador in audience
			10 = elgar.415 #Elgar 415: Ballet
			10 = elgar.416 #Elgar 416: Film music
		}
	}

	current_value = {
		value = var:elgar_composer_progress
	}

	goal_add_value = {
		add = 30
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}

	}

je_elgar_composer_composing = {
	icon = "gfx/interface/icons/event_icons/event_music.dds"

	group = je_group_elgar_music

	on_monthly_pulse = {
		random_events = {
			320 = elgar.421 #Elgar 421: Composer progress
			
			10 = elgar.422 #Elgar 422: No inspiration
			10 = elgar.423 #Elgar 423: A walk
			10 = elgar.424 #Elgar 424: Disturbances
			10 = elgar.425 #Elgar 425: Not enough comfort
			10 = elgar.426 #Elgar 426: Sea Shanty
			10 = elgar.427 #Elgar 427: March
			10 = elgar.428 #Elgar 428: Kunstlied
			10 = elgar.429 #Elgar 429: Sonata for Piano
		}
	}

	current_value = {
		value = var:elgar_composer_progress
	}

	goal_add_value = {
		add = 30
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		custom_tooltip = {
			text = je_elgar_composer_composing_work_tt
			if = {
				limit = {
					var:elgar_composer_var = {
						var:elgar_composer_num_compositions_var >= 3
					}
				}
				random_list = {
					33 = {
						trigger_event = {
							id = elgar.431 #Concerto
							popup = yes
						}
					}
					33 = {
						trigger_event = {
							id = elgar.432 #Symphony
							popup = yes
						}
					}
					33 = {
						trigger_event = {
							id = elgar.433 #Opera
							popup = yes
						}
					}
				}
			}
			else = {
				random_list = {
					50 = {
						trigger_event = {
							id = elgar.431 #Concerto
							popup = yes
						}
					}
					50 = {
						trigger_event = {
							id = elgar.432 #Symphony
							popup = yes
						}
					}
				}
			}
		}
	}

	fail = {
	}

	on_fail = {
		set_variable = {
            name = elgar_composer_progress
            value = 0
		}
	}

	progressbar = yes

	weight = 50
}
