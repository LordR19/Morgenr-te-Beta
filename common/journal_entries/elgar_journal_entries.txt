#Impressionism

je_elgar_impressionism = {
	icon = "gfx/interface/icons/event_icons/event_impressionism.dds"

	group = je_group_elgar_culture

	is_shown_when_inactive = {
		NOT = {
			has_technology_researched = elgar_impressionism_tech
			has_technology_researched = elgar_expressionism_tech
			has_technology_researched = elgar_art_nouveau_tech
		}
	}

	possible = {
		has_technology_researched = elgar_classicism_tech
	}

	immediate = {
		set_variable = {
			name = je_elgar_impressionism_var
			days = 1
		}
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:elgar_music_tradition #value or variable you use to meassure progress
		}
	}

	goal_add_value = {
		value = 850
	}

	complete = {
		any_scope_building = {
			is_building_type = building_elgar_opera
		}
		has_technology_researched = elgar_irrationalism_tech
		has_technology_researched = realism
		ig:ig_landowners = {
			is_powerful = yes
		}
		NOT = {
			ig:ig_industrialists = {
				is_powerful = yes
			}
		}
		ig:ig_intelligentsia = {
			ig_approval >= 5
	    }
		custom_tooltip = {
			text = je_elgar_impressionism_success
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
		hidden_trigger = {
			NOT = { has_variable = je_elgar_impressionism_var }
		}
	}

	on_complete = {
		add_technology_researched = elgar_impressionism_tech
	}

	fail = {
		OR = {
			has_technology_researched = elgar_expressionism_tech
			has_technology_researched = elgar_art_nouveau_tech
		}
	}

	invalid = {
		has_technology_researched = elgar_impressionism_tech
	}

	progressbar = yes

	weight = 50
}

	}

		value = 850 #value you want the variable to reach

je_elgar_composer_composing = {
	icon = "gfx/interface/icons/event_icons/event_music.dds"

	group = je_group_elgar_music

	on_yearly_pulse = {
		effect = {
			composer_experience_gain_effect = { AMOUNT = 5 }
			set_variable = {
				name = elgar_composer_num_yearly_compositions
				value = { 1 4 }
				months = 2
			}
			var:elgar_composer_var = {
				change_variable = {
					name = elgar_composer_num_compositions_var
					add = root.var:elgar_composer_num_yearly_compositions
				}
			}

			post_notification = elgar_composer_yearly_compositions_notification
		}
		random_events = {
			5000 = 0
			
			20 = elgar.440 #Elgar 440: Folk music
			20 = elgar.441 #Elgar 441: Sea Shanty
			20 = elgar.442 #Elgar 442: March
			20 = elgar.443 #Elgar 443: Kunstlied
			20 = elgar.444 #Elgar 444: Sonata for Piano
			20 = elgar.445 #Elgar 445: Treatise on Music
			20 = elgar.446 #Elgar 446: Requiem
			20 = elgar.447 #Elgar 447: A daring opera
			40 = elgar.448 #Elgar 448: Composer Duel

			1 = elgar.406 #Elgar 406: Composer dies
		}
	}

	complete = {
		custom_tooltip = {
			text = je_elgar_composer_composing_stop_tt
			always = no
		}
	}

	on_complete = {
	}

	fail = {
		OR = {
			custom_tooltip = {
				text = je_stopped_composing_tt
				has_variable = elgar_stopped_composing_remover_var
			}
			custom_tooltip = {
				text = je_composer_dies_tt
				NOT = {
					has_variable = elgar_composer_var
				}
			}
		}
	}

	on_fail = {
		remove_variable = elgar_stopped_composing_remover_var
	}

	scripted_button = elgar_stop_composing_button

	progressbar = no

	weight = 50
}
