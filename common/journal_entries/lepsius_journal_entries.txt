#Lepsius' letter to Rosellini, access to Upper Egypt and Lower Egypt

je_lepsius_letter_to_rosellini = { 
	icon = "gfx/interface/icons/event_icons/event_archaeology.dds"

	group = je_group_lepsius_archaeology

	is_shown_when_inactive = {
		exists = c:PRU
		c:PRU = this
		basileia_is_active = no #Built in compatibility to Basileia
		NOT = { has_global_variable = lepsius_egyptology_invented_country_definition_global_var }
	}

	possible = {
		has_technology_researched = lepsius_antiquarianism_tech
	}

	complete = {
		game_date >= 1837.6.1
	}

		add_technology_researched = lepsius_specialization_egypt_tech
		trigger_event = { id = lepsius.2 popup = yes }
	}

		OR = {
			NOT = { exists = c:PRU }
			has_global_variable = lepsius_egyptology_invented_country_definition_global_var
		}
	}

	weight = 10

	should_be_pinned_by_default = yes
}

#Decipherment of the Akkadian language, access to Babylonia and Assyria (early archaeology is needed)

je_lepsius_decipherment_akkadian = {
	icon = "gfx/interface/icons/event_icons/event_archaeology.dds"

	group = je_group_lepsius_archaeology

	is_shown_when_inactive = {
		has_technology_researched = lepsius_antiquarianism_tech
		game_date >= 1850.1.1
		NOT = {
			has_global_variable = lepsius_decipherment_akkadian_var
		}
	}

	possible = {
		has_technology_researched = lepsius_early_archaeology_tech
	}

	complete = {
		any_scope_building = {
			is_building_type = building_lepsius_archaeological_museum
		}
		country_has_building_type_levels = {  
			target = bt:building_university  
			value  >= 10
		}
		lepsius_has_archaeologist = yes
		hidden_trigger = {
			NOT = {
				has_variable = lepsius_akkadian_activator_var
			}
		}
	}

	immediate = {
		set_variable = {
			name = lepsius_akkadian_activator_var
			days = 2
		}
	}

	fail = {
		has_global_variable = lepsius_decipherment_akkadian_var
    }

	on_complete = {
		add_technology_researched = lepsius_specialization_mesopotamia_tech
		add_modifier = {
			name = lepsius_decipherment_akkadian_modifier
		}
		trigger_event = { id = lepsius.4 popup = yes }
	}

	weight = 10

	should_be_pinned_by_default = yes
}

		}

			lepsius_has_major_artifacts = yes
		}
		any_scope_state = {
			is_capital = yes
			OR = {
				is_production_method_active = {
					building_type = building_lepsius_archaeological_museum
					production_method = pm_lepsius_large_egyptian_exhibition
				}

				}
				is_production_method_active = {
					building_type = building_lepsius_archaeological_museum
					production_method = pm_lepsius_large_precolumbian_exhibition
