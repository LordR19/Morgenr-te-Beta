on_monthly_pulse_country = { 
	on_actions = {
		GUI_update_on_monthly_pulse_country
	}
}

GUI_update_on_monthly_pulse_country = {
	effect = {
		if = {
			limit = { NOT = { has_global_variable = verrier_global_list_created_global_var } }
			set_global_variable = {
				name = verrier_global_list_created_global_var
				days = 10
			}
			verrier_remove_astronomical_objects_ranking_variables = yes
			verrier_create_astronomical_objects_ranking_global_effect = yes
			dubois_remove_specimen_ranking_variables = yes
			dubois_create_specimen_ranking_global_effect = yes
		}
	}
}
