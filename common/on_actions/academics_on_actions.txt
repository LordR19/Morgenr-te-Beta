on_monthly_pulse_country = { 
	on_actions = {
		academics_unhappy_academic_random_on_monthly_pulse_country
		academics_retirement_for_scientists
		academics_academy_progress_for_player
		academics_set_country_advancement
	}
}

academics_unhappy_academic_random_on_monthly_pulse_country = {
	random_events = {
		50 = 0
		2 = academics.1000
		2 = academics.1001
		2 = academics.1002
		2 = academics.1003
		2 = academics.1004
		2 = academics.1005
	}
}

academics_retirement_for_scientists = {
	random_events = {
		1 = academics.1
		1 = academics.2
		1 = academics.3
		1 = academics.4
		1 = academics.5
		1 = academics.6
		40 = 0
	}
}

academics_academy_progress_for_player = {
	effect = {
		if = {
			limit = {
				is_player = yes
				NOT = { has_game_rule = academics_no_gui_rule }
				any_scope_building = {
					is_building_type = building_university
					level > 0
				}
			}
			if = {
				limit = {
					NOT = { has_variable = dubois_academy_progress }
				}
				set_variable = {
					name = dubois_academy_progress
					value = 0
				}
			}
			if = {
				limit = {
					var:dubois_academy_progress >= 360
				}
				academics_academic_recruited_effect = yes
			}
			trigger_event = academics.50
			if = {
				limit = {
					var:dubois_academy_progress >= 360
				}
				academics_academic_recruited_effect = yes
			}
		}
	}
}

academics_set_country_advancement = {
	effect = {
		set_variable = {
			name = academics_country_tech_advance_var
			value = gaudi_percentage_more_advanced_countries_value
		}
		if = {
			limit = { has_variable = gaudi_engineer_var }
			remove_modifier = gaudi_engineer_technology_spread_modifier
			add_modifier = {
				name = gaudi_engineer_technology_spread_modifier
				multiplier = var:academics_country_tech_advance_var
			}
		}
	}
}
