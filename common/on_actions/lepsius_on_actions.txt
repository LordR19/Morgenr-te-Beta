lepsius_on_yearly_pulse_country = {
	events = {
		lepsius.50 #Yearly artifacts from excavations

		#Historical Fossils
		lepsius.1331 #Neanderthaler
		lepsius.1332 #Cro-Magnon
		lepsius.1333 #Peking Man
		lepsius.1334 #Taung Child
		lepsius.1335 #Piltdown Man

		lepsius.1341 #Piltdown Man Fraud
	}

	#Excavation Events
	random_events = {
		500 = 0

		5 = lepsius.712 #Nefertiti
		5 = lepsius.713 #Narmer
		5 = lepsius.723 #Helmet of Meskalamdug
		5 = lepsius.732 #Diskos of Phaistos
		5 = lepsius.733 #Snake Goddess
	}
	#Local Excavation Events
	random_events = {
		120 = 0

		15 = lepsius.1211 #Cult objects
		15 = lepsius.1212 #Ceramics
		5 = lepsius.1213 #Neolithic village
		10 = lepsius.1214 #Human Bones
		20 = lepsius.1215 #An ambitious successor
	}

	#Cave Exploration Events
	random_events = {
		120 = 0

		15 = lepsius.1411 ##Lepsius 1411: Stone Tools
		5 = lepsius.1412 #Lepsius 1412: Strange Bone
		10 = lepsius.1413 #Lepsius 1413: Fruitful studies
		10 = lepsius.1414 #Lepsius 1414: Assistant finds Fossils
		20 = lepsius.1415 #Lepsius 1415: Human Bones
		5 = lepsius.1416 #Lepsius 1416: Drawings
		10 = lepsius.1417 #Lepsius 1417: Altamira Paintings
	}

	#Local Archaeology Events
	#This uses random_lists to ensure that events get fired equally often no matter how many different archaeological regions you own.
	#The first random list fires if you have an active archaeologist. It decides whether you find anything this year.
	#The second random list decides what (of the possible regions you own) you find.
	#The third random list decides, which specific Event triggers.
	#Make sure that all Event Triggers are also used somewhere in this construct, otherwise the event will not fire anyways.
	#E.g. the Roman Tablets event needs Modern Archaeology, if you do not have it, the trigger does not allow that option to be taken.
	effect = {
		if = {
			limit = { lepsius_has_active_archaeologist = yes }
			random_list = {
				14 = {}
				1 = {
					random_list = {
						2 = { #Aztec
							trigger = {
								any_scope_state = {
									lepsius_atztec_empire_archaeology_region = yes
								}
							}
							random_list = {
								2 = { trigger_event = lepsius.2131 } #Lepsius 2131: Skull Fragmets found in Aztect Settlement
								2 = { trigger_event = lepsius.2132 } #Lepsius 2132: Aztec Inscription saved by local Scholar
								5 = { trigger_event = lepsius.2133 } #Lepsius 2133: Aztec Ruins discovered
								1 = { trigger_event = lepsius.2134 } #Lepsius 2134: Obsidian Mirror discovered
							}
						}
						2 = { #Persian
							trigger = {
								any_scope_state = {
									lepsius_persian_archaeology_region = yes
								}
							}
							random_list = {
								3 = { trigger_event = lepsius.2111 } #Lepsius 2111: Ancient Persian Sarcophagus
								1 = { trigger_event = lepsius.2112 } #Lepsius 2112: Mesopotamian Cuneiform Tablets in Persian Palace
								1 = { trigger_event = lepsius.2113 } #Lepsius 2113: Egyptian Statue in Persian Tomb
								1 = { trigger_event = lepsius.2114 } #Lepsius 2114: Monumental Inscription
							}
						}
						2 = { #Roman
							trigger = {
								any_scope_state = {
									lepsius_imperium_romanum_archaeology_region = yes
								}
							}
							random_list = {
								3 = { trigger_event = lepsius.2101 } #Lepsius 2101: Remains of a Roman Battle
								5 = { trigger_event = lepsius.2102 } #Lepsius 2102: Villa Rustica discovered
								1 = {
									trigger = {
										has_technology_researched = lepsius_modern_archaeology_tech
									}
									trigger_event = lepsius.2103 #Lepsius 2103: Ancient Roman tablets discovered
								}
								1 = { trigger_event = lepsius.2104 } #Lepsius 2104: Roman Coin Treasure discovered
							}
						}
						1 = { #Stone Age
							random_list = {
								1 = { trigger_event = lepsius.2121 } #Lepsius 2121: A Prehistoric Bone Flute
								1 = { trigger_event = lepsius.2122 } #Lepsius 2122: Cave Painting found
								1 = { trigger_event = lepsius.2123 } #Lepsius 2123: Prehistoric Figurine discovered
								1 = { trigger_event = lepsius.2124 } #Lepsius 2124: Bones and Stone Tools recovered
							}
						}
					}
				}
			}
		}
	}
}

lepsius_on_monthly_pulse_country = {
	events = {
		#Excavation sacked
		lepsius.61

		#Anthropology
		lepsius.2000 #Thomsen invents the Three-Age System
		lepsius.1301 #Cave Found
		lepsius.1302 #Neanderthal
		lepsius.1303 #Altamira
		lepsius.1304 #Taung
		lepsius.1305 #Kromdraai
		lepsius.1306 #Zhoukoudian
		lepsius.1307 #Ngandong
		lepsius.1308 #Vézère Valley (Cro-Magnon)

	}
	effect = { #Deactivates Archaeological Museums if there is more than 1
		if = {
			limit = {
				any_scope_state = {
					is_capital = yes
					has_building = building_lepsius_archaeological_museum
					NOT = {
						has_active_building = building_lepsius_archaeological_museum
					}
				}
			}
			capital = {
				activate_building = {
					building = building_lepsius_archaeological_museum
				}
			}
		}
		if = {
			limit = {
				any_scope_state = {
					has_active_building = building_lepsius_archaeological_museum
					count > 1
				}
			}
			if = {
				limit = {
					any_scope_state = {
						is_capital = yes
						has_active_building = building_lepsius_archaeological_museum
					}
				}
				every_scope_state= {
					limit = {
						is_capital = no
						has_active_building = building_lepsius_archaeological_museum
					}
					activate_building = {
						building = building_lepsius_archaeological_museum
					}
				}
			}
			else = {
				while = {
					limit = {
						any_scope_state = {
							has_active_building = building_lepsius_archaeological_museum
							count > 1
						}
					}
					random_scope_state = {
						deactivate_building = {
							building = building_lepsius_archaeological_museum
						}
					}
				}
			}
		}
		if = {
			limit = {
				any_scope_building = {
					is_building_type = building_lepsius_archaeological_museum
				}
				NOT = {
					any_scope_state = {
						has_active_building = building_lepsius_archaeological_museum
					}
				}
			}
			if = {
				limit = {
					any_scope_state = {
						is_capital = yes
						has_building = building_lepsius_archaeological_museum
					}
				}
				capital = {
					activate_building = {
						building = building_lepsius_archaeological_museum
					}
				}
			}
			else = {
				random_scope_state = {
					limit = {
						has_building = building_lepsius_archaeological_museum
					}
					activate_building = {
						building = building_lepsius_archaeological_museum
					}
				}
			}
		}
	}
}

lepsius_on_acquired_technology = {
	events = {
		lepsius.80 #Lepsius 80: A new Age of Archaeology! (if Early Archaeology is discovered)
		lepsius.81 #Lepsius 81: Modern Excavation Methods (if Modern Archaeology is discovered)
	}
}

lepsius_on_building_built = {
	effect = {
		if = {
			limit = {
				owner = {
					NOT = {
						has_variable = lepsius_museum_inauguration_happened
					}
					lepsius_has_archaeological_museum = yes
				}
			}
			owner = {
				trigger_event = {
					id = lepsius.10 #Lepsius 10: Museum inauguration
					popup = yes
				}
			}
		}
	}
}
