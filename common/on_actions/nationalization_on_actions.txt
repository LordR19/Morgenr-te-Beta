on_yearly_pulse_country = { 
	on_actions = {
		natinalization_on_yearly_pulse_country
	}
}

natinalization_on_yearly_pulse_country = {
	effect = {
		if = {
			limit = {
				country_has_primary_culture = cu:united_swiss
				any_scope_state = {
					state_region = {
						NOT = { is_homeland = cu:alemannic }
					}
				}
			}
			every_scope_pop = {
				limit = {
					culture = cu:alemannic
				}
				change_pop_culture = {
					target = cu:united_swiss
					value = 1
				}
			}
		}
		if = {
			limit = {
				has_global_variable = nationalization_north_german_integrated_global_var
				any_scope_pop = {
					has_pop_culture = north_german
				}
			}
			every_scope_pop = {
				limit = {
					has_pop_culture = north_german
				}
				change_pop_culture = {
					target = cu:united_german
					value = 1
				}
			}
		}
		if = {
			limit = {
				has_global_variable = nationalization_south_german_integrated_global_var
				any_scope_pop = {
					has_pop_culture = south_german
				}
			}
			every_scope_pop = {
				limit = {
					has_pop_culture = south_german
				}
				change_pop_culture = {
					target = cu:united_german
					value = 1
				}
			}
		}
		if = {
			limit = {
				has_global_variable = manzoni_united_italian_nation_var
				any_scope_pop = {
					OR = {
						has_pop_culture = south_italian
						has_pop_culture = north_italian
					}
				}
			}
			every_scope_pop = {
				limit = {
					has_pop_culture = north_italian
				}
				change_pop_culture = {
					target = cu:united_italian
					value = 1
				}
			}
			every_scope_pop = {
				limit = {
					has_pop_culture = south_italian
				}
				change_pop_culture = {
					target = cu:united_italian
					value = 1
				}
			}
		}
	}
}

on_monthly_pulse_country = {
	on_actions = {
		natinalization_on_monthly_pulse_country
	}
}

natinalization_on_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
				has_global_variable = nationalization_north_german_integrated_global_var
				country_has_primary_culture = cu:north_german
			}
			remove_primary_culture = north_german
			add_primary_culture = united_german
			every_scope_pop = {
				limit = {
					has_pop_culture = north_german
				}
				change_pop_culture = {
					target = cu:united_german
					value = 1
				}
			}
		}
		if = {
			limit = {
				has_global_variable = nationalization_south_german_integrated_global_var
				country_has_primary_culture = cu:south_german
			}
			remove_primary_culture = south_german
			add_primary_culture = united_german
			every_scope_pop = {
				limit = {
					has_pop_culture = south_german
				}
				change_pop_culture = {
					target = cu:united_german
					value = 1
				}
			}
		}
		if = {
			limit = {
				has_global_variable = manzoni_united_italian_nation_var
				OR = {
					country_has_primary_culture = cu:north_italian
					country_has_primary_culture = cu:south_italian
				}
			}
			remove_primary_culture = north_italian
			remove_primary_culture = south_italian
			add_primary_culture = united_italian
			every_scope_pop = {
				limit = {
					has_pop_culture = north_italian
				}
				change_pop_culture = {
					target = cu:united_german
					value = 1
				}
			}
			every_scope_pop = {
				limit = {
					has_pop_culture = south_italian
				}
				change_pop_culture = {
					target = cu:united_german
					value = 1
				}
			}
		}
	}
}
