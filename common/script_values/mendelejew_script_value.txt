#This value determines the rating of a chemist
mendelejew_chemist_character_rating_value = {
	### Experience: 1 point per 50 Experience ###
	value = var:mendelejew_chemist_experience
	divide = 50
	### Positive Traits ###
	#Experience Trait
	if = {
		limit = {
			has_trait = mendelejew_chemist_famous
		}
		add = 3
	}
	else_if = {
		limit = {
			has_trait = mendelejew_chemist_experienced
		}
		add = 2
	}
	else = {
		add = 1
	}
	#General
	if = {
		limit = {
			has_trait = mendelejew_chemist_theorist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = mendelejew_chemist_analyst
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = mendelejew_chemist_periodic_table_inventor
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = mendelejew_chemist_inspector
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = engineer
		}
		add = 2
	}
	if = {
		limit = {
			OR = {
				has_trait = verrier_astronomer
				has_trait = verrier_astronomer_experienced
				has_trait = verrier_astronomer_famous
			}
		}
		add = 1
	}
	#if = {
	#	limit = {
	#		OR = {
	#			has_trait = dubois_geologist
	#			has_trait = dubois_geologist_experienced
	#			has_trait = dubois_geologist_famous
	#		}
	#	}
	#	add = 1
	#}
	if = {
		limit = {
			has_trait = innovative
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = meticulous
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = persistent
		}
		add = 1
	}
	### Negative Traits ###
	if = {
		limit = {
			has_trait = alcoholic
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = opium_addiction
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = cocaine_addiction
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = senile
		}
		subtract = 3
	}
	### Age Factor, the older the worse for the ranking ###
	if = {
		limit = { age > 30 }
		subtract = 1
	}
	if = {
		limit = { age > 40 }
		subtract = 2
	}
	if = {
		limit = { age > 50 }
		subtract = 3
	}
	if = {
		limit = { age > 60 }
		subtract = 4
	}
	if = {
		limit = { age > 70 }
		subtract = 5
	}
}
