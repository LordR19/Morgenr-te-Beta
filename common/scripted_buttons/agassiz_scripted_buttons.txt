	name = "agassiz_stop_geologist_project_button"
	desc = "agassiz_stop_geologist_project_button_desc"

	visible = {  always = yes }

	possible = {
		NOT = { has_variable = agassiz_stop_geologist_improving_project_var }
	}

		set_variable = agassiz_stop_geologist_improving_project_var
	}

		base = 0 #It's just not worth for you, Mr AI
	}
}

######SEISMOLOGY######

agassiz_build_seismometer_button = {
	name = "agassiz_build_seismometer_button"
	desc = "agassiz_build_seismometer_button_desc"

	visible = { 
		NOT = {
			has_variable = agassiz_seismometer_done_var
		}
	}

	possible = {
		gaudi_has_engineer = yes
		dubois_has_geologist = yes
	}

	effect = {
		custom_tooltip = {
			text = agassiz_seismometer_done_tt
			set_variable = agassiz_seismometer_done_var
			change_variable = {
				name = agassiz_build_seismometer_var
				add = 1
		   }
		}
	}

	ai_chance = {
		base = 100
	}
}
