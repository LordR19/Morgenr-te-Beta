ballooning_button = {
	name = "ballooning_button"
	desc = "ballooning_button_desc"

	visible = { always = yes }

	possible = {
		NOT = {
			has_variable = curtiss_ballooning_window_open_var
		}
	}

	effect = {
		every_country = {
			limit = {
				is_player = yes
			}

		scope:button_presser_scope = {
			set_variable = curtiss_ballooning_window_open_var
		
			trigger_event = {
				id = curtiss.651
				popup = yes
			}
		}
	}
}
