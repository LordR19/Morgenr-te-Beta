#Calculates the Amount of Academy Progress
academics_calculate_progress_effect = {
	### Actual calculation done in script_value ###
	if = {
		limit = { academics_academic_progress_add_value > 0 }
		set_variable = {
			name = academics_estimated_monthly_academy_progress_var
			value = academics_academic_progress_add_value
		}
	}
	else = {
		set_variable = {
			name = academics_estimated_monthly_academy_progress_var
			value = 0
		}
	}
}

		limit = {
			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking

#This effect creates a a variable to display the rank of each country for astronomical objects
verrier_astronomical_rank_effect = {
	set_local_variable = { #This is used to keep track of the current rank
		name = current_rank
		value = 0
	}
	ordered_country = {
		limit = {
			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}
		order_by = var:verrier_astronomical_objects
		max = 1000 #Apparently without this it just takes one country.
		check_range_bounds = no #prevents errors

		remove_variable = verrier_astronomical_rank_country_var
		change_local_variable = {
			name = current_rank
			add = 1
		}
		set_variable = { #uses the current rank to set the rank variable for the country.
			name = verrier_astronomical_rank_country_var
			value = local_var:current_rank
		}
	}
}

	#Removes the GLOBAL ranking variables
	remove_global_variable = manzoni_literary_tradition_rank_1_global_var
	remove_global_variable = manzoni_literary_tradition_rank_2_global_var
	remove_global_variable = manzoni_literary_tradition_rank_3_global_var
	remove_global_variable = manzoni_literary_tradition_rank_4_global_var
	remove_global_variable = manzoni_literary_tradition_rank_5_global_var
	remove_global_variable = manzoni_literary_tradition_rank_6_global_var
	remove_global_variable = manzoni_literary_tradition_rank_7_global_var
	remove_global_variable = manzoni_literary_tradition_rank_8_global_var
	remove_global_variable = manzoni_literary_tradition_rank_9_global_var
	remove_global_variable = manzoni_literary_tradition_rank_10_global_var

	ordered_country = {
		limit = {
			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}
		order_by = var:manzoni_literary_tradition_var
		max = 10 #Limits the list to the ten top countires
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_1_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_1_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_2_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_2_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_3_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_3_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_4_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_4_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_5_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_5_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_6_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_6_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_7_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_7_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_8_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_8_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_9_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_9_global_var
				value = this
			}
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = manzoni_literary_tradition_rank_10_global_var }
			}
			set_global_variable = {
				name = manzoni_literary_tradition_rank_10_global_var
				value = this
			}
		}
	}
}

#This effect creates a a variable to display the rank of each country for Literary Tradition
manzoni_literary_rank_effect = {
	set_local_variable = { #This is used to keep track of the current rank
		name = current_rank
		value = 0
	}
	ordered_country = {
		limit = {
			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}
		order_by = var:manzoni_literary_tradition_var
		max = 1000 #Apparently without this it just takes one country.
		check_range_bounds = no #prevents errors

		remove_variable = manzoni_literary_rank_country_var
		change_local_variable = {
			name = current_rank
			add = 1
		}
		set_variable = { #uses the current rank to set the rank variable for the country.
			name = manzoni_literary_rank_country_var
			value = local_var:current_rank
		}
	}
}
