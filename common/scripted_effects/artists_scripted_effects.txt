#This effect is used to add the monthly Artistic Progress
artists_monthly_artistic_progress_effect = {
	if = { #Effect only gets executed if there is either an Arts Academy or an Opera
		limit = {
			any_scope_building = {
				OR = {
					is_building_type = building_arts_academy
					is_building_type = building_elgar_opera
				}
				level >= 1
			}
		}

		#Set up
		if = {
			limit = { NOT = { has_variable = artists_artistic_progress_var } }
			set_variable = {
				name = artists_artistic_progress_var
				value = 0
			}
		}

		### Actual calculation done in script_value ###
		if = {
			limit = { monthly_artistic_progress_gain_value > 0 }
			set_variable = {
				name = artists_estimated_monthly_artistic_progress_var
				value = monthly_artistic_progress_gain_value
			}
		}
		else = {
			set_variable = {
				name = artists_estimated_monthly_artistic_progress_var
				value = 0
			}
		}
		
		#Add monthly variable to the total variable
		if = {
			limit = { var:artists_estimated_monthly_artistic_progress_var > 0 }
			change_variable = {
				name = artists_artistic_progress_var
				add = var:artists_estimated_monthly_artistic_progress_var
			}
		}

		if = {
			limit = { var:artists_artistic_progress_var >= 1000 }
			change_variable = {
				name = artists_artistic_progress_var
				subtract = 1000
			}
			if = {
				limit = { has_variable = artists_promote_composer_active_var }
				trigger_event = {
					id = artists.51
					popup = yes
				}
			}
			else_if = {
				limit = { has_variable = artists_promote_writer_active_var }
				trigger_event = {
					id = artists.52
					popup = yes
				}
			}
			else_if = {
				limit = { has_variable = artists_promote_painter_active_var }
				trigger_event = {
					id = artists.53
					popup = yes
				}
			}
			else = {
				random_list = {
					10 = {
						trigger_event = {
							id = artists.51
							popup = yes
						}
						modifier = {
							if = {
								limit = { elgar_has_composer_no_tooltip_trigger = yes }
								add = -7
							}
						}
					}
					10 = {
						trigger_event = {
							id = artists.52
							popup = yes
						}
						modifier = {
							if = {
								limit = { manzoni_has_writer_no_tooltip = yes }
								add = -7
							}
						}
					}
					10 = {
						trigger_event = {
							id = artists.53
							popup = yes
						}
						modifier = {
							if = {
								limit = { klimt_has_painter_no_tooltip = yes }
								add = -7
							}
						}
					}
				}
			}
		}
	}
	else = {
		set_variable = {
			name = artists_estimated_monthly_artistic_progress_var
			value = 0
		}
	}
}

#Effect to add $AMOUNT$ Artistic Progress
artists_add_x_artistic_progress = {
	# Amount = amount of Artistic Progress to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = academy_progress_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = artists_add_x_artistic_progress_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = artists_add_x_artistic_progress_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = artists_add_x_artistic_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = artists_add_x_artistic_progress_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = artists_add_x_artistic_progress_plus_x_tt
	}

	#This part adds the AMOUNT of Artistic Progress
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = artists_artistic_progress_var
			}
		}
		set_variable = {
			name = artists_artistic_progress_var
			value = 0
		}
	}
	change_variable = {
		name = artists_artistic_progress_var
		add = $AMOUNT$
	}
}

#This effect is used to set up the list of travelling composers
artists_set_up_travelling_composer_list_effect = {
	clear_global_variable_list = mr_travelling_composer_global_list
	ordered_character_in_void = {
		max = 1000
		order_by = var:elgar_composer_experience
		check_range_bounds = no

		limit = {
			has_variable = mr_is_travelling_composer_var
		}
		add_to_global_variable_list = {
			name = mr_travelling_composer_global_list
			target = this
		}
	}
}

#This effect is used to remove the travelling composers as there shouldn't be too many.
artists_remove_travelling_composer_from_void_effect = {
	if = {
		#This part is used to limit the effect to only one country
		limit = {
			NOT = { has_global_variable = artists_travelling_composer_remove_global_var }
		}
		set_global_variable = {
			name = artists_travelling_composer_remove_global_var
			days = 15
		}

		#This is used to keep track of the current number
		set_local_variable = {
			name = current_number
			value = 0
		}
		
		#This part gets the travelling composers ordered by experience.
		# The Composers that are not historical and are over the limit will be deleted.
		ordered_character_in_void = {
			limit = { has_variable = mr_is_travelling_composer_var }
			order_by = var:elgar_composer_experience
			max = 1000
			check_range_bounds = no

			change_local_variable = {
				name = current_number
				add = 1
			}

			if = {
				limit = {
					local_var:current_number > 20
					is_historical = no
				}
				free_character_from_void = no
			}
		}
	}
}

#This effect is used to set up the list of local writers

		check_range_bounds = no

		limit = {

				this = root
			}
		}
		root = {
			add_to_variable_list = {
				name = mr_local_writers_list
				target = prev
			}
		}
	}

#This effect kills the current composer
# !!! Change stuff in both of the following effects !!!
kill_composer_effect = {
	var:elgar_composer_var = {
		free_character_from_void = no
	}

	remove_variable = elgar_composer_var
	elgar_remove_composer_modifier_effect = yes
}
#This Effect retires the current Composer
# Use "set_retirement_variable_effect" in the immediate effect when using this effect.
retire_composer_effect = {
	custom_tooltip = {
		text = retire_composer_effect_tt

		var:elgar_composer_var = {
			free_character_from_void = no
		}
	}

	remove_variable = elgar_composer_var
	elgar_remove_composer_modifier_effect = yes
}

#This effect makes the current composer a travelling Composer
set_composer_travelling_composer_effect = {
	if = {
		limit = { has_variable = elgar_composer_var }

		custom_tooltip = {
			text = elgar_set_composer_travelling_composer_tt
	
			var:elgar_composer_var = {
				remove_variable = mr_is_composer_var
				set_variable = mr_is_travelling_composer_var
				if = {
					limit = {
						NOT = { has_variable = mr_travelling_composer_year_counter_var }
					}
					set_variable = {
						name = mr_travelling_composer_year_counter_var
						value = 0
					}
				}
				add_to_global_variable_list = {
					name = mr_travelling_composer_global_list
					target = this
				}
			}
		
			remove_variable = elgar_composer_var
		}
	}
}

# Used to set up the Experience of a character 
composer_experience_set_up_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = elgar_composer_experience
			}
		}
		#Age: 2 Reputation for each year older than 25
		if = {
			limit = {
				age > 25
			}
			set_variable = {
				name = elgar_composer_experience
				value = age
			}
			change_variable = {
				name = elgar_composer_experience
				add = -25
			}
			change_variable = {
				name = elgar_composer_experience
				multiply = 2
			}
		}
		else = {
			set_variable = {
				name = elgar_composer_experience
				value = 0
			}
		}
		#Random Reputation factor
		set_local_variable = {
			name = random_amount_local_var
			value = { 25 50 }
		}
		change_variable = {
			name = elgar_composer_experience
			add = local_var:random_amount_local_var
		}
		change_variable = {
			name = elgar_composer_experience
			add = artists_composer_experience_set_up_value
		}
		if = { #Adds the Famous Composer Trait if Experience is above 500
			limit = {
				var:elgar_composer_experience >= 500
				NOT = {
					has_trait = elgar_great_composer
				}
			}
			remove_trait = elgar_composer
			add_trait = elgar_great_composer
			owner = {
				elgar_add_great_composer_count_effect = yes
			}
		}
	}
	set_variable = {
		name = elgar_composer_num_compositions_var
		value = 0
	}
	set_variable = {
		name = elgar_composer_num_seasons_var
		value = 0
	}
	set_variable = {
		name = elgar_composer_num_tours_var
		value = 0
	}
	set_variable = {
		name = elgar_composer_num_masterworks_var
		value = 0
	}
}

			}

				order_by = var:elgar_composer_experience
				check_range_bounds = no

					has_variable = mr_is_travelling_composer_var

					NOT = {
						var:elgar_composer_unavailable_for_this_var ?= root
					}
				}
				
				save_scope_as = chosen_composer
	
				root = {
					set_variable = {
						name = elgar_composer_var
						value = scope:chosen_composer
					}
				}
				
				free_character_from_void = yes
				if = {
					limit = {
						NOT = {
							owner ?= root
						}
					}
					transfer_character = root
				}

			}

# Used to set up the Experience of a character 
writer_experience_set_up_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = manzoni_writer_experience
			}
		}
		#Age: 2 Reputation for each year older than 25
		if = {
			limit = {
				age > 25
			}
			set_variable = {
				name = manzoni_writer_experience
				value = age
			}
			change_variable = {
				name = manzoni_writer_experience
				add = -25
			}
			change_variable = {
				name = manzoni_writer_experience
				multiply = 2
			}
		}
		else = {
			set_variable = {
				name = manzoni_writer_experience
				value = 0
			}
		}
		#Random Reputation factor
		set_local_variable = {
			name = random_amount_local_var
			value = { 25 50 }
		}
		change_variable = {
			name = manzoni_writer_experience
			add = local_var:random_amount_local_var
		}
		change_variable = {
			name = manzoni_writer_experience
			add = artists_writer_experience_set_up_value
		}
		if = { #Adds the Great Writer Trait if Experience is above 500
			limit = {
				has_variable = manzoni_writer_experience
				var:manzoni_writer_experience >= 500
				NOT = {
					has_trait = manzoni_great_writer
				}
			}
			remove_trait = literary
			add_trait = manzoni_great_writer
		}
	}

	#Number of Bestsellers
	if = {
		limit = {
			NOT = { has_variable = manzoni_num_bestsellers_var }
		}
		set_variable = {
			name = manzoni_num_bestsellers_var
			value = 0
		}
	}
	#Number of Book Licenses
	if = {
		limit = {
			NOT = { has_variable = manzoni_num_book_licenses_var }
		}
		set_variable = {
			name = manzoni_num_book_licenses_var
			value = 0
		}
	}
	#Number of Book Licenses
	if = {
		limit = {
			NOT = { has_variable = manzoni_num_book_licenses_var }
		}
		set_variable = {
			name = manzoni_num_book_licenses_var
			value = 0
		}
	}
	#Number of Publications
	if = {
		limit = {
			NOT = { has_variable = manzoni_num_publications_var }
		}
		set_variable = {
			name = manzoni_num_publications_var
			value = 0
		}
	}
}

#Writer Experience gain
writer_experience_gain_effect = { #adds XP to the Writer
	# Amount = amount of XP to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = xp_amount
		value = $AMOUNT$
	}

	if = {
		limit = {
			has_variable = manzoni_writer_var #should only add experience if there actually is a Writer
		}
		custom_tooltip = {
			text = manzoni_writer_experience_gain_tt
			if = {
				limit = {
					NOT = {
						var:manzoni_writer_var = { #Fallback to make sure there are no errors if the Writer does not have the variable yet
							has_variable = manzoni_writer_experience
						}
					}
				}
				var:manzoni_writer_var = { #Sets the experience in case it was not set yet
					writer_experience_set_up_effect = yes
				}
			}
			var:manzoni_writer_var = {
				change_variable = {
					name = manzoni_writer_experience
					add = $AMOUNT$
				}
				if = {
					limit = { var:manzoni_writer_experience > 500 }
					set_variable = {
						name = manzoni_writer_experience
						value = 500
					}
				}
			}
		}
		var:manzoni_writer_var = {
			if = {
				limit = {
					var:manzoni_writer_experience >= 500
					NOT = {
						has_trait = manzoni_great_writer
					}
				}
				remove_trait = literary
				add_trait = manzoni_great_writer
				root = {
					trigger_event = {
						id = artists.102
						popup = yes
					}
				}
			}
		}
	}
}

# Used to set up the Experience of a character 
painter_experience_set_up_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = klimt_painter_experience
			}
		}
		#Age: 5 Reputation per year over 25
		if = {
			limit = {
				age > 25
			}
			set_variable = {
				name = klimt_painter_experience
				value = age
			}
			change_variable = {
				name = klimt_painter_experience
				subtract = 25
			}
			change_variable = {
				name = klimt_painter_experience
				multiply = 5
			}
			if = {
				limit = { var:klimt_painter_experience > 500 }
				set_variable = {
					name = klimt_painter_experience
					value = 500
				}
			}
		}
		else = {
			set_variable = {
				name = klimt_painter_experience
				value = 0
			}
		}
		#Random Reputation factor
		set_local_variable = {
			name = random_amount_local_var
			value = { 25 50 }
		}
		change_variable = {
			name = klimt_painter_experience
			add = local_var:random_amount_local_var
		}
		if = {
			limit = {
				owner = {
					klimt_has_gallery = yes
				}
			}
			change_variable = {
				name = klimt_painter_experience
				add = 20
			}
		}
		if = { #Adds the Great Painter Trait if Experience is above 500
			limit = {
				var:klimt_painter_experience >= 500
				NOT = {
					has_trait = klimt_great_painter
				}
			}
			remove_trait = klimt_painter
			add_trait = klimt_great_painter
		}
	}
}

	# Amount = amount of XP to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = xp_amount

	if = {
		limit = {
			has_variable = klimt_painter_var #should only add experience if there actually is a Painter
		}
		custom_tooltip = {
			text = klimt_painter_experience_gain_tt
			if = {
				limit = {
					NOT = {
						var:klimt_painter_var = { #Fallback to make sure there are no errors if the Painter does not have the variable yet
							has_variable = klimt_painter_experience
						}
					}
				}
				var:klimt_painter_var = { #Sets the experience in case it was not set yet
					painter_experience_set_up_effect = yes
				}
			}
			var:klimt_painter_var = {
				change_variable = {
					name = klimt_painter_experience
					add = $AMOUNT$
				}
				if = {
					limit = { var:klimt_painter_experience > 500 }
					set_variable = {
						name = klimt_painter_experience
						value = 500
					}
				}
			}
		}
		var:klimt_painter_var = {
			if = {
				limit = {
					var:klimt_painter_experience >= 500
					NOT = {
						has_trait = klimt_great_painter
					}
				}
				remove_trait = klimt_painter
				add_trait = klimt_great_painter
				root = {
					trigger_event = {
						id = artists.103
						popup = yes
					}
				}
			}
		}
	}
}

klimt_independent_painter_character_set_up_cost_effect = {

		add_treasury = {
			if = {

				value = prev.var:klimt_painter_experience
				divide = 2
				multiply = money_amount_multiplier_small
				max = {
					value = prev.var:klimt_painter_experience
					multiply = 10000
				}
				multiply = -1
			}
			else = {
				value = 1
				divide = 2
				multiply = money_amount_multiplier_small
				max = 100000
				multiply = -1
			}
		}
	}
}

#Give the three countries with most artist reputation a prize
artists_biennale_winner_effect = {
	#Removes the GLOBAL ranking variables
	remove_global_variable = artists_biennale_place_1_global_var 
	remove_global_variable = artists_biennale_place_2_global_var 
	remove_global_variable = artists_biennale_place_3_global_var 

	ordered_country = {
		limit = {
			OR = {
				has_technology_researched = elgar_impressionism_tech
				has_technology_researched = elgar_expressionism_tech
				has_technology_researched = elgar_art_nouveau_tech
				NOT = {
					has_modifier = artists_biennale_triumph_modifier
				}
			}
		}
		order_by = artists_biennale_value
		max = 3 #Limits the list to the three top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = { 
					has_global_variable = artists_biennale_place_1_global_var  
				}
			}
			set_global_variable = {
				name = artists_biennale_place_1_global_var  
				value = this
			}
			save_scope_as = artists_biennale_first_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = artists_biennale_place_2_global_var   }
			}
			set_global_variable = {
				name = artists_biennale_place_2_global_var 
				value = this
			}
			save_scope_as = artists_biennale_second_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = artists_biennale_place_3_global_var   }
			}
			set_global_variable = {
				name = artists_biennale_place_3_global_var 
				value = this
			}
			save_scope_as = artists_biennale_third_winner_scope
		}
	}
}

klimt_ai_hire_painter_effect = {
	if = {

			NOT = {
				has_modifier = klimt_recently_hired_independent_painter_modifier
			}
			has_technology_researched = elgar_classicism_tech
			elgar_has_opera_size_x_trigger = { LEVEL = 1 }
		}

		#This if construct chooses the experience level of possible Composers
		if = { #All Composers possible
			limit = {

			}
			klimt_hire_independent_painter_effect = { EXPERIENCE = 10000 }
		}
		else_if = { #Up to 500 Reputation
			limit = {
				klimt_tier_4_painter_trigger = yes
			}

			klimt_hire_independent_painter_effect = { EXPERIENCE = 400 }			
		}
		else_if = { #Up to 450 Reputation
			limit = {
				klimt_tier_3_painter_trigger = yes
			}

			klimt_hire_independent_painter_effect = { EXPERIENCE = 300 }		
		}
		else_if = { #Up to 350 Reputation
			limit = {
				klimt_tier_2_painter_trigger = yes
			}

			klimt_hire_independent_painter_effect = { EXPERIENCE = 200 }		
		}
		else_if = { #Up to 250 Reputation
			limit = {
				klimt_tier_1_painter_trigger = yes
			}

			klimt_hire_independent_painter_effect = { EXPERIENCE = 100 }		
		}
	}
}

#This Effect is used to either directly choose a Independent Painter or replace an existing one.
klimt_hire_independent_painter_effect = {
	# EXPERIENCE = maximum experience of the composer
	if = {
		limit = { has_variable = klimt_painter_var }
		if = {
			limit = {
				any_character_in_void = {
					has_variable = mr_is_independent_painter
					var:klimt_painter_experience < $EXPERIENCE$
					var:klimt_painter_experience >= klimt_painter_experience_plus_50_value
					culture = { #This limits Painters to countries with primary cultures
						shares_non_heritage_trait_with_any_primary_culture = root
						shares_heritage_trait_with_any_primary_culture = root
					}
					NOT = {
						var:klimt_painter_unavailable_for_this_var ?= root
					}
				}
			}
			ordered_character_in_void = {
				order_by = var:klimt_painter_experience
				check_range_bounds = no

				limit = {
					has_variable = mr_is_independent_painter
					var:klimt_painter_experience < $EXPERIENCE$
					culture = { #This limits Painters to countries with primary cultures
						shares_non_heritage_trait_with_any_primary_culture = root
						shares_heritage_trait_with_any_primary_culture = root
					}
					NOT = {
						var:klimt_painter_unavailable_for_this_var ?= root
					}
				}

				save_scope_as = chosen_painter

				root = {
					set_painter_independent_painter_effect = yes
					set_variable = {
						name = klimt_painter_var
						value = scope:chosen_painter
					}
				}

				free_character_from_void = yes
				if = {
					limit = {
						NOT = {
							owner ?= root
						}
					}
					transfer_character = root
				}
				remove_variable = mr_is_independent_painter
				set_variable = mr_is_painter_var
				remove_list_global_variable = {
					name = mr_independent_painter_global_list
					target = this
				}
				place_character_in_void = 1200
	
				root = {
					add_modifier = {
						name = klimt_recently_hired_independent_painter_modifier
						years = 3
					}
				}
				
				klimt_independent_painter_character_set_up_cost_effect = yes
			}
		}
	}
	else = {
		if = {
			limit = {
				any_character_in_void = {
					has_variable = mr_is_independent_painter
					var:klimt_painter_experience < $EXPERIENCE$
					culture = { #This limits Painters to countries with primary cultures
						shares_non_heritage_trait_with_any_primary_culture = root
						shares_heritage_trait_with_any_primary_culture = root
					}
					NOT = {
						var:klimt_painter_unavailable_for_this_var ?= root
					}
				}
			}
			ordered_character_in_void = {
				order_by = var:klimt_painter_experience
				check_range_bounds = no
	
				limit = {
					has_variable = mr_is_independent_painter
					var:klimt_painter_experience < $EXPERIENCE$
					culture = { #This limits Painters to countries with primary cultures
						shares_non_heritage_trait_with_any_primary_culture = root
						shares_heritage_trait_with_any_primary_culture = root
					}
					NOT = {
						var:klimt_painter_unavailable_for_this_var ?= root
					}
				}
				
				save_scope_as = chosen_painter
	
				root = {
					set_variable = {
						name = klimt_painter_var
						value = scope:chosen_painter
					}
				}
				
				free_character_from_void = yes
				if = {
					limit = {
						NOT = {
							owner ?= root
						}
					}
					transfer_character = root
				}
				remove_variable = mr_is_independent_painter
				set_variable = mr_is_composer_var
				remove_list_global_variable = {
					name = mr_independent_painter_global_list
					target = this
				}
				place_character_in_void = 1200
	
				root = {
					add_modifier = {
						name = klimt_recently_hired_independent_painter_modifier
						years = 3
					}
				}
				scope:chosen_painter = {
					klimt_independent_painter_character_set_up_cost_effect = yes
				}
			}
		}
	}
}
