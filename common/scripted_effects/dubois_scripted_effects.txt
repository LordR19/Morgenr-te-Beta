#Effect to add $AMOUNT$ Specimen
dubois_add_x_specimen_effect = {
	# Amount = amount of specimens to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = specimen_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = dubois_add_x_specimen_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = dubois_add_x_specimen_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = dubois_add_x_specimen_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = dubois_add_x_specimen_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = dubois_add_x_specimen_plus_x_tt
	}

	#Yearly Specimenss are used for the botanical report, they are added with each normal specimen
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = dubois_yearly_specimen_var
			}
		}
		set_variable = {
			name = dubois_yearly_specimen_var
			value = 0
		}
	}
	change_variable = {
		name = dubois_yearly_specimen_var
		add = $AMOUNT$
	}
	change_global_variable = {
		name = dubois_global_specimen
		add = $AMOUNT$
	}

	#This part adds the AMOUNT of Specimens
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = dubois_specimen
			}
		}
		set_variable = {
			name = dubois_specimen
			value = 0
		}
	}
	change_variable = {
		name = dubois_specimen
		add = $AMOUNT$
	}

	#dubois_naturalist_total_specimens_var is used to keep track of all the specimens that a specific Naturalist found.
	#It is used to show how successful the Naturalist is.
	#Every time a normal Naturalist object is added, the same number gets added here.
	if = {
		limit = { #Only works if there actually is a Naturalist
			has_variable = dubois_naturalist_var
		}
		var:dubois_naturalist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = dubois_naturalist_total_specimens_var
					}
				}
				set_variable = {
					name = dubois_naturalist_total_specimens_var
					value = 0
				}
			}
			change_variable = { #The amount of Specimens that this Naturalist discovered
				name = dubois_naturalist_total_specimens_var
				add = $AMOUNT$
			}
		}
	}
}

#Effect to add $AMOUNT$ Academy Progress
dubois_add_x_academy_progress = {
	# Amount = amount of Academy Progress to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = academy_progress_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = academy_add_x_academy_progress_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = academy_add_x_academy_progress_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = academy_add_x_academy_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = academy_add_x_academy_progress_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = academy_add_x_academy_progress_plus_x_tt
	}

	#This part adds the AMOUNT of Academy Progress
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = dubois_academy_progress
			}
		}
		set_variable = {
			name = dubois_academy_progress
			value = 0
		}
	}
	change_variable = {
		name = dubois_academy_progress
		add = $AMOUNT$
	}
}

dubois_change_dinosaur_rating = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = add_dinosaur_prestige
		value = $AMOUNT$
	}

	#Used to display the right tooltip
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = dubois_change_dinousaur_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = dubois_change_dinousaur_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = dubois_change_dinousaur_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = dubois_change_dinousaur_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = dubois_change_dinousaur_plus_x_tt
	}

	#Fallback in case the variable does not exist yet
	if = {
		limit = {
			NOT = {
				has_variable = dubois_dinosaur_rating
			}
		}
		set_variable = {
			name = dubois_dinosaur_rating
			value = 0
		}
	}

	#Actual Variable Change
	change_variable = { 
		name = dubois_dinosaur_rating
		add = $AMOUNT$
	}

	#Adds the correct modifier
	hidden_effect = {
		if = {
			limit = {
				var:dubois_dinosaur_rating <= -5
				NOT = {
					has_modifier = dubois_dinosaur_rating_cabinet_of_scientific_horros_modifier
				}
			}
			dubois_remove_dinosaur_prestige_modifiers_effect = yes
			add_modifier = {
				name = dubois_dinosaur_rating_cabinet_of_scientific_horros_modifier
			}
		}
		if = {
			limit = {
				var:dubois_dinosaur_rating <= -2
				NOR = {
					var:dubois_dinosaur_rating <= -5
					has_modifier = dubois_dinosaur_rating_skeletons_of_ridiculousness_modifier
				}
			}
			dubois_remove_dinosaur_prestige_modifiers_effect = yes
			add_modifier = {
				name = dubois_dinosaur_rating_skeletons_of_ridiculousness_modifier
			}
		}
		if = {
			limit = {
				var:dubois_dinosaur_rating < 2
				NOR = {
					var:dubois_dinosaur_rating <= -2
					has_modifier = dubois_dinosaur_rating_mediocre_exhibition_modifier
				}
			}
			dubois_remove_dinosaur_prestige_modifiers_effect = yes
			add_modifier = {
				name = dubois_dinosaur_rating_mediocre_exhibition_modifier
			}
		}
		if = {
			limit = {
				var:dubois_dinosaur_rating >= 2
				NOR = {
					var:dubois_dinosaur_rating >= 5
					has_modifier = dubois_dinosaur_rating_little_paleontological_curiosities_modifier
				}
			}
			dubois_remove_dinosaur_prestige_modifiers_effect = yes
			add_modifier = {
				name = dubois_dinosaur_rating_little_paleontological_curiosities_modifier
			}
		}
		if = {
			limit = {
				var:dubois_dinosaur_rating >= 5
				NOR = {
					var:dubois_dinosaur_rating >= 10

######Dinosaur Remount######

dubois_remount_dinosaur_effect = {
	random_list = {
		25 = {
			modifier = {
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							var:dubois_paleontologist_experience >= 175
						}
					}
					add = 25
				}
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							var:dubois_paleontologist_experience >= 200
						}
					}
					add = 25
				}
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							var:dubois_paleontologist_experience >= 250
						}
					}
					add = 50
				}
			}
			#Better Remount
			trigger_event = {
				id = dubois.2101
				days = { 180 200 }
			}
		}
		40 = {
			modifier = {
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							has_trait = dubois_paleontologist_famous
						}
					}
					add = -100
				}
			}
			#No change Remount
			trigger_event = {
				id = dubois.2102
				days = { 180 200 }
			}
		}
		35 = {
			modifier = {
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							var:dubois_paleontologist_experience <= 99
						}
					}
					add = 50
				}
				if = {
					limit = {
						var:dubois_paleontologist_var = {
							OR = {
								has_trait = dubois_paleontologist_experienced
								has_trait = dubois_paleontologist_famous
							}
						}
					}
					add = -100
				}
			}
			#Worse Remount
			trigger_event = {
				id = dubois.2103
				days = { 180 200 }
			}
		}
	}
	custom_tooltip = dubois_remount_dinosaur_tt
}

dubois_set_paleontologist_location = {
	if = {
		limit = {
			has_variable = dubois_paleontologist_in_north_america_var
		}
		random_state = {
			limit = {
				has_state_trait = dubois_state_trait_great_north_american_formation
			}
			save_scope_as = dubois_paleontology_state
		}
	}
	if = {
		limit = {
			has_variable = dubois_paleontologist_in_south_america_var
		}
		random_state = {
			limit = {
				has_state_trait = dubois_state_trait_south_american_formation
			}
			save_scope_as = dubois_paleontology_state
		}
	}
	if = {
		limit = {
			has_variable = dubois_paleontologist_in_europe_var
		}
		random_state = {
			limit = {
				has_state_trait = dubois_state_trait_great_european_formation
			}
			save_scope_as = dubois_paleontology_state
		}
	}
	if = {
		limit = {
			has_variable = dubois_paleontologist_in_africa_var
		}
		random_state = {
			limit = {
				has_state_trait = dubois_state_trait_great_african_formation
			}
			save_scope_as = dubois_paleontology_state
		}
	}
	if = {
		limit = {
			has_variable = dubois_paleontologist_in_asia_var
		}
		random_state = {
			limit = {
				has_state_trait = dubois_state_trait_great_asian_formation
			}
			save_scope_as = dubois_paleontology_state
		}
	}
	if = {
		limit = {
			has_variable = dubois_paleontologist_in_australia_var
		}
		random_state = {
			limit = {
				has_state_trait = dubois_state_trait_great_australian_formation
			}
			save_scope_as = dubois_paleontology_state
		}
	}
	if = {
		limit = {
			has_variable = dubois_paleontologist_in_local_var
		}
		random_state = {
			limit = {
				OR = {
					AND = {
						has_state_trait = dubois_state_trait_normal_formation
						owner = THIS
					}
					AND = {
						has_state_trait = dubois_state_trait_normal_formation
						owner = {
							has_diplomatic_pact = {
								who = ROOT
								type = paleontological_license
							}
						}
					}
				}
			}
			save_scope_as = dubois_paleontology_state
		}
	}
}

set_dubois_promote_mountain_expedition_leader_effect = {
	if = {
		limit = {
			scope:dubois_mountain_expedition_leader = {
				has_trait = dubois_mountaineer
				has_trait = explorer
			}
		}
		scope:dubois_mountain_expedition_leader = {
			add_trait = mountain_commander
		}
	}
	if = {
		limit = {
			scope:dubois_mountain_expedition_leader = {
				has_trait = dubois_mountaineer
				NOT = {
					has_trait = explorer
				}
			}
		}
		scope:dubois_mountain_expedition_leader = {
			add_trait = explorer
		}
	}
	if = {
		limit = {
			scope:dubois_mountain_expedition_leader = {
				NOT = {
					has_trait = dubois_mountaineer
				}
			}
		}
		scope:dubois_mountain_expedition_leader = {
			add_trait = dubois_mountaineer
		}
	}
	if = {
		limit = {
			scope:dubois_mountain_expedition_leader = {
				has_trait = psychological_affliction
			}
		}
		scope:dubois_mountain_expedition_leader = {
			remove_trait = psychological_affliction
		}
	}
}

### Paleontological Progress ###

#Effect to add $AMOUNT$ Paleontological Progress
dubois_add_x_paleontological_progress_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = progress_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = dubois_add_x_progress_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = dubois_add_x_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = dubois_add_x_progress_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = dubois_dinosaur_excavation_finding_progress
			}
		}
		set_variable = {
			name = dubois_dinosaur_excavation_finding_progress
			value = 0
		}
	}
	change_variable = {
		name = dubois_dinosaur_excavation_finding_progress
		add = $AMOUNT$
	}

	if = {
		limit = { var:dubois_dinosaur_excavation_finding_progress < 0 }
		set_variable = {
			name = dubois_dinosaur_excavation_finding_progress
			value = 0
		}
	}
}
