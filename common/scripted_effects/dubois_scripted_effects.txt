######NATURALIST

create_naturalist_effect = {
	create_character = {
		add_trait = dubois_naturalist
		save_scope_as = dubois_naturalist_scope
	}
		
	scope:dubois_naturalist_scope = {
		place_character_in_void = 1200	
	}
		
	set_variable = {
		name = dubois_naturalist_var
		value = scope:dubois_naturalist_scope
	}
}

###Add Technology###

#Small Add

dubois_small_add_technology = {
	if = {
		limit = {
			NOT = {
				has_technology_researched = pharmaceuticals
			}
			can_research = pharmaceuticals
		}
		add_technology_progress = { progress = 1500 technology = pharmaceuticals }
	}
	if = {
		limit = {
			has_technology_researched = dialectics
			has_technology_researched = pharmaceuticals
			NOT = {
				has_technology_researched = dubois_theory_of_evolution_tech
			}
			can_research = dubois_theory_of_evolution_tech
		}
		add_technology_progress = { progress = 1500 technology = dubois_theory_of_evolution_tech }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			NOT = {
				has_technology_researched = dubois_paleontology_tech
			}
			can_research = dubois_paleontology_tech
		}
		add_technology_progress = { progress = 3000 technology = dubois_paleontology_tech }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			has_technology_researched = dubois_paleontology_tech
			NOT = {
				has_technology_researched = quinine
			}
			can_research = quinine
		}
		add_technology_progress = { progress = 1500 technology = quinine }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			has_technology_researched = dubois_paleontology_tech
			has_technology_researched = quinine
			NOT = {
				has_technology_researched = malaria_prevention
			}
			can_research = malaria_prevention
		}
		add_technology_progress = { progress = 3000 technology = quinine }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			has_technology_researched = dubois_paleontology_tech
			has_technology_researched = quinine
			has_technology_researched = malaria_prevention
		}
	}
}
