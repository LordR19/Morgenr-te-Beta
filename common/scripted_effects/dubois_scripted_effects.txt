create_dubois_eugene_effect = {
	create_character = {
		dna = mr_net_dubois_eugene
		first_name = dubois_eugene
		last_name = dubois_dubois
		culture = cu:dutch
		religion = protestant
		interest_group = ig_intelligentsia
		female = no
		birth_date = 1858.1.28
		traits = { 
			dubois_paleontologist
			innovative
			persistent
			meticulous
		}

		save_scope_as = dubois_new_paleontologist_scope
	}
		
	scope:dubois_new_paleontologist_scope = {
		place_character_in_void = 1200
		set_variable = dubois_dubois_character_var
	}
}

			persistent

		}

		save_scope_as = dubois_naturalist_scope
	}
		
	scope:dubois_naturalist_scope = {
		add_trait = dubois_naturalist
		place_character_in_void = 1200
		set_variable = dubois_darwin_character_var
	}
		
	set_variable = {
		name = dubois_naturalist_var
		value = scope:dubois_naturalist_scope
	}
}

###Add Technology###

#Small Add

dubois_small_add_technology = {
	if = {
		limit = {
			NOT = {
				has_technology_researched = pharmaceuticals
			}
			can_research = pharmaceuticals
		}
		add_technology_progress = { progress = 1500 technology = pharmaceuticals }
	}
	if = {
		limit = {
			has_technology_researched = dialectics
			has_technology_researched = pharmaceuticals
			NOT = {
				has_technology_researched = dubois_theory_of_evolution_tech
			}
			can_research = dubois_theory_of_evolution_tech
		}
		add_technology_progress = { progress = 1500 technology = dubois_theory_of_evolution_tech }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			NOT = {
				has_technology_researched = dubois_paleontology_tech
			}
			can_research = dubois_paleontology_tech
		}
		add_technology_progress = { progress = 3000 technology = dubois_paleontology_tech }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			has_technology_researched = dubois_paleontology_tech
			NOT = {
				has_technology_researched = quinine
			}
			can_research = quinine
		}
		add_technology_progress = { progress = 1500 technology = quinine }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			has_technology_researched = dubois_paleontology_tech
			has_technology_researched = quinine
			NOT = {
				has_technology_researched = malaria_prevention
			}
			can_research = malaria_prevention
		}
		add_technology_progress = { progress = 3000 technology = malaria_prevention }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			has_technology_researched = dubois_paleontology_tech
			has_technology_researched = quinine
			has_technology_researched = malaria_prevention
			NOT = {
				has_technology_researched = antibiotics
			}
			can_research = antibiotics
		}
		add_technology_progress = { progress = 5500 technology = antibiotics }
	}
	if = {
		limit = {
			has_technology_researched = dubois_theory_of_evolution_tech
			has_technology_researched = dubois_paleontology_tech
			has_technology_researched = quinine
			has_technology_researched = malaria_prevention
			has_technology_researched = antibiotics
		}
	}
}

#Effect to add a specific amount of fossils to the paleontologist and the the yearly discovered fossils
#NUMBER = How many fossils will be added
dubois_set_fossils_paleontologist_yearly = {
	if = {
		limit = {
			NOT = {
				has_variable = dubois_yearly_fossils_var
			}
		}
		set_variable = {
			name = dubois_yearly_fossils_var
			value = 0
		}
	}
	if = {
		limit = {
			NOT = {
				has_variable = dubois_paleontologist_var
				var:dubois_paleontologist_var = {
					has_variable = dubois_paleontologist_total_fossils_var
				}
			}
		}
		var:dubois_paleontologist_var = {
			set_variable = {
				name = dubois_paleontologist_total_fossils_var
				value = 0
			}
		}
	}
	change_variable = {
		name = dubois_yearly_fossils_var
		add = $NUMBER$
	}
	if = {
		limit = {
			has_variable = dubois_paleontologist_var
		}
		var:dubois_paleontologist_var = {
			change_variable = {
				name = dubois_paleontologist_total_fossils_var
				add = $NUMBER$
			}
		}
	}
}
