	if = {
		limit = {
			NOT = {
				has_variable = gaudi_splendid_architecture
			}
		}
		set_variable = {
			name = gaudi_splendid_architecture
			value = 0
		}
	}
	change_variable = {
		name = gaudi_splendid_architecture
		add = 1
	}

}

######Neuschwanstein######

#Add Neuschwanstein progress

#Effect to add $AMOUNT$ Neuschwanstein Progress
gaudi_add_x_neuschwanstein_progress = {
	# Amount = amount of Neuschwanstein Progress to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = neuschwanstein_progress_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_plus_x_tt
	}

	#This part adds the AMOUNT of Neuschwanstein Progress
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_global_variable = gaudi_neuschwanstein_progress
			}
		}
		set_global_variable = {
			name = gaudi_neuschwanstein_progress
			value = 0
		}
	}
	change_global_variable = {
		name = gaudi_neuschwanstein_progress
		add = $AMOUNT$
	}
}

#Effect to add $AMOUNT$ Magnetic Data
gaudi_add_x_magnetic_data = {
	# Amount = amount of Magnetic Data to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = magnetic_data_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = gaudi_add_x_magnetic_data_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = gaudi_add_x_magnetic_data_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = gaudi_add_x_magnetic_data_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = gaudi_add_x_magnetic_data_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = gaudi_add_x_magnetic_data_plus_x_tt
	}

	#This part adds the AMOUNT of National Magnetic Data
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = gaudi_magnetic_data
			}
		}
		set_variable = {
			name = gaudi_magnetic_data
			value = 0
		}
	}
	change_variable = {
		name = gaudi_magnetic_data
		add = $AMOUNT$
	}

	#This part adds the AMOUNT of Global Magnetic Data
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_global_variable = gaudi_total_magnetic_data
			}
		}
		set_global_variable = {
			name = gaudi_total_magnetic_data
			value = 0
		}
	}
	change_global_variable = {
		name = gaudi_total_magnetic_data
		add = $AMOUNT$
	}
}

gaudi_set_magnetic_association = {
	set_global_variable = {
		name = gaudi_magnetic_association_members
		value = 1
	}
	set_variable = {
		name = gaudi_magnetic_data
		value = 0
	}
	set_global_variable = {
		name = gaudi_total_magnetic_data
		value = 0
	}
	set_variable = gaudi_magnetic_association_member_var

	custom_tooltip = gaudi_set_magnetic_association_tt
}

gaudi_add_magnetic_association_member = {
	change_global_variable = {
		name = gaudi_magnetic_association_members
		add = 1
	}
	set_variable = {
		name = gaudi_megnetic_station_progress
		value = 0
	}
	set_variable = gaudi_magnetic_association_member_var

	custom_tooltip = gaudi_add_magnetic_association_member_tt
}

gaudi_civil_engineering_progress = {
	set_variable = {
		name = gaudi_monthly_civil_engineering_progress_var
		value = 1 #Base Value
	}
	### Safety to make sure the engineer has the right experienc type ###
	if = {
		limit = {
			has_variable = gaudi_engineer_var
			var:gaudi_engineer_var = {
				NOT = {
					has_variable = gaudi_civil_engineer_experience
				}
			}
		}
		var:gaudi_engineer_var = {
			set_variable = {
				name = gaudi_civil_engineer_experience
				value = 0
			}
		}
	}

	if = { #Has Civil Engineer
		limit = {
			gaudi_has_civil_engineer_trigger = yes
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 1
		}
	}
	if = { #Experience of the Engineer divided by 100
		limit = {
			gaudi_has_civil_engineer_trigger = yes
		}
		if = {
			limit = {
				var:gaudi_engineer_var.var:gaudi_civil_engineer_experience < 300
			}
			set_local_variable = {
				name = engineer_experience_add
				value = var:gaudi_engineer_var.var:gaudi_civil_engineer_experience
			}
			change_local_variable = {
				name = engineer_experience_add
				divide = 300
			}
			change_variable = {
				name = gaudi_monthly_civil_engineering_progress_var
				add = local_var:engineer_experience_add
			}
		}
		else = {
			change_variable = {
				name = gaudi_monthly_civil_engineering_progress_var
				add = 1
			}
		}

				has_trait = gaudi_civil_engineer
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 1
		}
	}

			gaudi_has_engineer = yes
			var:gaudi_engineer_var = {
				has_trait = meticulous
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 0.1
		}
	}
	if = { #Engineer has Persistent Trait
		limit = {
			gaudi_has_engineer = yes
			var:gaudi_engineer_var = {
				has_trait = persistent
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 0.1

				}
			}
		}
		change_variable = {
			name = gaudi_monthly_aeronautical_engineering_progress_var
			add = 0.2
		}
	}
	if = { #Engineer has Pilot Trait
		limit = {
			gaudi_has_engineer = yes
			var:gaudi_engineer_var = {
				OR = {
					has_trait = curtiss_pilot
					has_trait = curtiss_pilot_ace
				}
			}
		}
		change_variable = {
			name = gaudi_monthly_aeronautical_engineering_progress_var
			add = 0.2
		}
	}
	if = {
		limit = {
			has_modifier = gaudi_additional_support_for_civil_engineer_modifier
		}
		change_variable = {
			name = gaudi_monthly_aeronautical_engineering_progress_var
			add = 1
		}
	}
	#Specialized Universities
	change_variable = {
		name = gaudi_monthly_aeronautical_engineering_progress_var
		add = academics_mathematical_university_multiplier_tiny
	}
}

#Effect used to add or remove progress from the scripted progress bar.
# AMOUNT = amount of progress to add/subtract
gaudi_add_x_mechanical_computer_progress_effect = {
	# Amount = amount of Progress to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = amount
		value = $AMOUNT$
	}
	
	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = gaudi_add_x_mechanical_computer_progress_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = gaudi_add_x_mechanical_computer_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = gaudi_add_x_mechanical_computer_progress_plus_x_tt
	}

	hidden_effect = {
		je:je_gaudi_mechanical_computer_project = {
			add_progress = { 
				name = gaudi_mechanical_computer_project_progress_bar
				value = $AMOUNT$
			}
		}
	}
}
