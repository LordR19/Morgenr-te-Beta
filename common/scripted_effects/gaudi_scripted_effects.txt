gaudi_add_1_splendid_architecture = {
	if = {
		limit = {
			NOT = {
				has_variable = gaudi_splendid_architecture
			}
		}
		set_variable = {
			name = gaudi_splendid_architecture
			value = 0
		}
	}
	change_variable = {
		name = gaudi_splendid_architecture
		add = 1
	}
	custom_tooltip = gaudi_add_1_splendid_architecture_tt
}

create_gaudi_antoni_effect = {
	create_character = {
		first_name = gaudi_antoni
		last_name = gaudi_gaudi
		culture = cu:catalan
		female = no
		religion = rel:catholic
		interest_group = ig:ig_intelligentsia
		dna = mr_cat_gaudi_antoni
		birth_date = 1852.6.25
		historical = yes
		traits = { 
			erudite
			reserved
			meticulous
			pious
			engineer
		}

		save_scope_as = gaudi_new_engineer_scope
	}

	scope:gaudi_new_engineer_scope = {
		set_variable = gaudi_gaudi_character_var
		set_variable = mr_is_engineer_var
		place_character_in_void = 1200
	}

	set_global_variable = gaudi_gaudi_created
}

#Historical Eiffel
create_eiffel_gustave_effect = {
	create_character = {
		first_name = gaudi_gustave
		last_name = gaudi_eiffel
		culture = cu:french
		female = no
		religion = rel:catholic
		interest_group = ig_industrialists
		dna = mr_fra_eiffel_gustave
		birth_date = 1832.12.15
		historical = yes
		traits = { 
			erudite
			ambitious
			engineer
		}

		save_scope_as = gaudi_engineer_scope
	}

	scope:gaudi_engineer_scope = {
		set_variable = mr_is_engineer_var
		set_variable = gaudi_eiffel_character_var
		place_character_in_void = 1200
	}
		
	set_variable = {
		name = gaudi_engineer_var
		value = scope:gaudi_engineer_scope
	}

	set_global_variable = gaudi_eiffel_created
}
