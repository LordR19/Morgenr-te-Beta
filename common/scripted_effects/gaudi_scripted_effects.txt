#Add Neuschwanstein progress

#Effect to add $AMOUNT$ Neuschwanstein Progress
gaudi_add_x_neuschwanstein_progress = {
	# Amount = amount of Neuschwanstein Progress to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = neuschwanstein_progress_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = gaudi_add_x_neuschwanstein_progress_plus_x_tt
	}

	#This part adds the AMOUNT of Neuschwanstein Progress
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_global_variable = gaudi_neuschwanstein_progress
			}
		}
		set_global_variable = {
			name = gaudi_neuschwanstein_progress
			value = 0
		}
	}
	change_global_variable = {
		name = gaudi_neuschwanstein_progress
		add = $AMOUNT$
	}
}

#Adds a small amount of civil engineering progress
gaudi_civil_engineering_small_add_technology = {
	if = {
		limit = {
			can_research = urbanization
		}
		add_technology_progress = { progress = 1500 technology = urbanization }
	}
	else_if = {
		limit = {
			can_research = urban_planning
		}
		add_technology_progress = { progress = 1500 technology = urban_planning }
	}
	else_if = {
		limit = {
			can_research = modern_sewerage
		}
		add_technology_progress = { progress = 1500 technology = modern_sewerage }
	}
	else_if = {
		limit = {
			can_research = steel_frame_buildings
		}
		add_technology_progress = { progress = 3000 technology = steel_frame_buildings }
	}
	else_if = {
		limit = {
			can_research = elevator
		}
		add_technology_progress = { progress = 3000 technology = elevator }
	}
	else_if = {
		limit = {
			can_research = paved_roads
		}
		add_technology_progress = { progress = 5500 technology = paved_roads }
	}
	else = {
		civil_engineer_experience_gain_effect = { AMOUNT = 5 }
		dubois_add_x_academy_progress = { AMOUNT = 2 }
	}
}

#This Scripted Effect calculates the Civil Engineering Progress
gaudi_civil_engineering_progress = {
	set_variable = {
		name = gaudi_monthly_civil_engineering_progress_var
		value = 1 #Base Value
	}
	if = { #Has Civil Engineer
		limit = {
			gaudi_has_civil_engineer_trigger = yes
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 1
		}
	}
	if = { #Experience of the Engineer divided by 100
		limit = {
			gaudi_has_civil_engineer_trigger = yes
		}
		if = {
			limit = {
				var:gaudi_engineer_var.var:gaudi_civil_engineer_experience < 300
			}
			set_local_variable = {
				name = engineer_experience_add
				value = var:gaudi_engineer_var.var:gaudi_civil_engineer_experience
			}
			change_local_variable = {
				name = engineer_experience_add
				divide = 300
			}
			change_variable = {
				name = gaudi_monthly_civil_engineering_progress_var
				add = local_var:engineer_experience_add
			}
		}
		else = {
			change_variable = {
				name = gaudi_monthly_civil_engineering_progress_var
				add = 1
			}
		}
	}
	if = { #Engineer has Civil Engineer Trait
		limit = {
			gaudi_has_civil_engineer_trigger = yes
			var:gaudi_engineer_var = {
				has_trait = gaudi_civil_engineer
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 1
		}
	}
	if = { #Engineer has Meticulous Trait
		limit = {
			gaudi_has_civil_engineer_trigger = yes
			var:gaudi_engineer_var = {
				has_trait = meticulous
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 0.1
		}
	}
	if = { #Engineer has Persistent Trait
		limit = {
			gaudi_has_civil_engineer_trigger = yes
			var:gaudi_engineer_var = {
				has_trait = persistent
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 0.1
		}
	}
	if = { #Engineer has Innovative Trait
		limit = {
			gaudi_has_civil_engineer_trigger = yes
			var:gaudi_engineer_var = {
				has_trait = innovative
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 0.1
		}
	}
	if = { #Engineer has Surveyor Trait
		limit = {
			gaudi_has_civil_engineer_trigger = yes
			var:gaudi_engineer_var = {
				has_trait = surveyor
			}
		}
		change_variable = {
			name = gaudi_monthly_civil_engineering_progress_var
			add = 0.1
		}
	}
}
