###Check Archaeologist Rank###

check_lepsius_archaeologist_rank = {
	if = {
		limit = {
			var:lepsius_archaeologist_var = {
				var:lepsius_archaeologist_experience >= 3
				NOT = {
					var:lepsius_archaeologist_experience >= 5
				}
				NOR = {
					has_trait = lepsius_archaeologist_experienced
					has_trait = lepsius_archaeologist_famous
				}
			}
		}
		remove_trait = lepsius_archaeologist
		add_trait = lepsius_archaeologist_experienced
	}
	if = {
		limit = {
			var:lepsius_archaeologist_var = {
				var:lepsius_archaeologist_experience >= 5
				NOT = {
					has_trait = lepsius_archaeologist_famous
				}
			}
		}
		remove_trait = lepsius_archaeologist_experienced
		add_trait = lepsius_archaeologist_famous
	}
}

#Effect to add $AMOUNT$ Minor Egyptian Artifacts
lepsius_add_x_egyptian_artifacts_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = artifact_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_total_artifacts
			}
		}
		set_variable = {
			name = lepsius_total_artifacts
			value = 0
		}
	}
	change_variable = {
		name = lepsius_total_artifacts
		add = $AMOUNT$
	}
	change_global_variable = {
		name = lepsius_global_artifacts
		add = $AMOUNT$
	}
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_egyptian_artifact_minor
			}
		}
		set_variable = {
			name = lepsius_egyptian_artifact_minor
			value = 0
		}
	}
	change_variable = {
		name = lepsius_egyptian_artifact_minor
		add = $AMOUNT$
	}
	if = {
		limit = { #Only works if there actually is an archaeologist
			has_variable = lepsius_archaeologist_var
		}
		var:lepsius_archaeologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = lepsius_egyptian_artifact_minor_archaeologist_var
					}
				}
				set_variable = {
					name = lepsius_egyptian_artifact_minor_archaeologist_var
					value = 0
				}
			}
			change_variable = { #The amount of Artifacts that this archaeologist discovered
				name = lepsius_egyptian_artifact_minor_archaeologist_var
				add = $AMOUNT$
			}
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = lepsius_total_artifact_minor_archaeologist_var
					}
				}
				set_variable = {
					name = lepsius_total_artifact_minor_archaeologist_var
					value = 0
				}
			}
			change_variable = { #The amount of Total Artifacts that this archaeologist discovered
				name = lepsius_total_artifact_minor_archaeologist_var
				add = $AMOUNT$
			}
		}
	}
}

#Add Unspecific Human Fossils

lepsius_add_x_human_fossils_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = fossil_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = lepsius_add_x_human_fossils_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = lepsius_add_x_human_fossils_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = lepsius_add_x_human_fossils_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = lepsius_add_x_human_fossils_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = lepsius_add_x_human_fossils_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_total_human_fossil
			}
		}
		set_variable = {
			name = lepsius_total_human_fossil
			value = 0
		}
	}
	change_variable = {
		name = lepsius_total_human_fossil
		add = $AMOUNT$
	}
	change_global_variable = {
		name = lepsius_global_human_fossil
		add = $AMOUNT$
	}
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_human_fossil
			}
		}
		set_variable = {
			name = lepsius_human_fossil
			value = 0
		}
	}
	change_variable = {
		name = lepsius_human_fossil
		add = $AMOUNT$
	}
	if = {
		limit = { #Only works if there actually is an anthropologist
			has_variable = lepsius_anthropologist_var
		}
		var:lepsius_anthropologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = lepsius_human_fossil_anthropologist_var
					}
				}
				set_variable = {
					name = lepsius_human_fossil_anthropologist_var
					value = 0
				}
			}
			change_variable = { #The amount of Artifacts that this archaeologist discovered
				name = lepsius_human_fossil_anthropologist_var
				add = $AMOUNT$
			}
		}
	}
}

#As this effect is called in an effect, it creates the list with normal variables
#Look at the customizable localizations to see how the countries are displayed.
#Basically the custom loc uses the variable to display both the countries in their order as well as the variables of those countries
lepsius_create_total_artifacts_ranking_effect = {
	ordered_country = {
		limit = {
			NOT = { #Hopefully keeps revolutions to appear in the ranking
				is_revolutionary = yes
			}
		}
		order_by = var:lepsius_total_artifacts
		max = 10 #Limits the list to the ten top countires
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = {
					root = {
						has_variable = lepsius_total_artifacts_rank_1_var
					}
				}
			}
			root = { #The countries are actually saved to the country that uses the effect in the event window
