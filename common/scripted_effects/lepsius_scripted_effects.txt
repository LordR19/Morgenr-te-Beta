###Determine the artifacts at the end of an Expedition###

lepsius_archaeological_expedition_artifacts_effect = {
	if = {
		limit = {
			var:lepsius_expedition_success < 3
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 1 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 3
			var:lepsius_expedition_success < 5
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 2 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 5
			var:lepsius_expedition_success < 8
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 4 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 8
			var:lepsius_expedition_success < 12
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 6 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 12
			var:lepsius_expedition_success < 15
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 8 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 15
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 10 }
	}
}

###Check Archaeologist Rank###

check_lepsius_archaeologist_rank = {
	if = {
		limit = {
			var:lepsius_archaeologist_var = {
				var:lepsius_archaeologist_experience >= 3
				NOT = {
					var:lepsius_archaeologist_experience >= 5
				}
				NOR = {
					has_trait = lepsius_archaeologist_experienced
					has_trait = lepsius_archaeologist_famous
				}
			}
		}
		remove_trait = lepsius_archaeologist
		add_trait = lepsius_archaeologist_experienced
	}
	if = {
		limit = {
			var:lepsius_archaeologist_var = {
				var:lepsius_archaeologist_experience >= 5
				NOT = {
					has_trait = lepsius_archaeologist_famous
				}
			}
		}
		remove_trait = lepsius_archaeologist_experienced
		add_trait = lepsius_archaeologist_famous
	}
}

#Effect to add $AMOUNT$ Minor Egyptian Artifacts
lepsius_add_x_egyptian_artifacts_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = artifact_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = lepsius_add_x_egyptian_artifacts_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_total_artifacts
			}
		}
		set_variable = {
			name = lepsius_total_artifacts
			value = 0
		}
	}
	change_variable = {
		name = lepsius_total_artifacts
		add = $AMOUNT$
	}
	change_global_variable = {
		name = lepsius_global_artifacts
		add = $AMOUNT$
	}
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_egyptian_artifact_minor
			}
		}
		set_variable = {
			name = lepsius_egyptian_artifact_minor
			value = 0
		}
	}
	change_variable = {
		name = lepsius_egyptian_artifact_minor
		add = $AMOUNT$
	}
	if = {
		limit = { #Only works if there actually is an archaeologist
			has_variable = lepsius_archaeologist_var
		}
		var:lepsius_archaeologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = lepsius_egyptian_artifact_minor_archaeologist_var
					}
				}
				set_variable = {
					name = lepsius_egyptian_artifact_minor_archaeologist_var
					value = 0
				}
			}
			change_variable = { #The amount of Artifacts that this archaeologist discovered
				name = lepsius_egyptian_artifact_minor_archaeologist_var
				add = $AMOUNT$
			}
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = lepsius_total_artifact_minor_archaeologist_var
					}
				}
				set_variable = {
					name = lepsius_total_artifact_minor_archaeologist_var
					value = 0
				}
			}
			change_variable = { #The amount of Total Artifacts that this archaeologist discovered
				name = lepsius_total_artifact_minor_archaeologist_var
				add = $AMOUNT$
			}
		}
	}
}

	custom_tooltip = lepsius_update_exhibition_effect_tt

	######EGYPTIAN EXHIBITION######
	if = {
		limit = {
			var:lepsius_egyptian_artifact_minor >= 10
			any_scope_state = {
				is_capital = yes
				is_production_method_active = {
					building_type = building_lepsius_archaeological_museum
					production_method = pm_lepsius_no_egyptian_exhibition
				}
			}
		}
		random_scope_state = {
			limit = {
				has_building = building_lepsius_archaeological_museum
				is_capital = yes
			}
			activate_production_method = {
				building_type = building_lepsius_archaeological_museum
				production_method = pm_lepsius_small_egyptian_exhibition
			}
		}

#Add Unspecific Human Fossils

lepsius_add_x_human_fossils_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = fossil_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = lepsius_add_x_human_fossils_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = lepsius_add_x_human_fossils_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = lepsius_add_x_human_fossils_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = lepsius_add_x_human_fossils_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = lepsius_add_x_human_fossils_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_total_human_fossil
			}
		}
		set_variable = {
			name = lepsius_total_human_fossil
			value = 0
		}
	}
	change_variable = {
		name = lepsius_total_human_fossil
		add = $AMOUNT$
	}
	change_global_variable = {
		name = lepsius_global_human_fossil
		add = $AMOUNT$
	}
	if = {
		limit = { #Only works if there actually is an anthropologist
			has_variable = lepsius_anthropologist_var
		}
		var:lepsius_anthropologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = lepsius_human_fossil_anthropologist_var
					}
				}
				set_variable = {
					name = lepsius_human_fossil_anthropologist_var
					value = 0
				}
			}
			change_variable = { #The amount of Artifacts that this archaeologist discovered
				name = lepsius_human_fossil_anthropologist_var
				add = $AMOUNT$
			}
		}
	}
}

	if = {

			if = {
				limit = { NOT = { has_variable = lepsius_total_human_fossils_anthropologist_var } }
				set_variable = {
					name = lepsius_total_human_fossils_anthropologist_var

			}
		}
		set_variable = {
			name = lepsius_cromagnon_human_fossil
			value = 0
		}

###ANTHROPOLOGICAL RATING######

lepsius_change_anthropological_rating = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = add_anthropological_prestige
		value = $AMOUNT$
	}

	#Used to display the right tooltip
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = lepsius_change_anthropological_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = lepsius_change_anthropological_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = lepsius_change_anthropological_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = lepsius_change_anthropological_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = lepsius_change_anthropological_plus_x_tt
	}

	#Fallback in case the variable does not exist yet
	if = {
		limit = {
			NOT = {
				has_variable = lepsius_anthropological_rating
			}
		}
		set_variable = {
			name = lepsius_anthropological_rating
			value = 0
		}
	}

	#Actual Variable Change
	change_variable = { 
		name = lepsius_anthropological_rating
		add = $AMOUNT$
	}

	#Adds the correct modifier
	hidden_effect = {
		lepsius_remove_anthropological_prestige_modifiers_effect = yes
		if = {
			limit = {
				var:lepsius_anthropological_rating >= 50
			}
			if = {
				limit = {
					var:lepsius_anthropological_rating < 150
				}
				add_modifier = {
					name = lepsius_anthropological_rating_level_1_modifier
				}
			}
			else_if = {
				limit = {
					var:lepsius_anthropological_rating < 250
				}
				add_modifier = {
					name = lepsius_anthropological_rating_level_2_modifier
				}
			}
			else_if = {
				limit = {
					var:lepsius_anthropological_rating < 400
				}
				add_modifier = {
					name = lepsius_anthropological_rating_level_3_modifier
				}
			}
			else_if = {
				limit = {
					var:lepsius_anthropological_rating < 600
				}
				add_modifier = {
					name = lepsius_anthropological_rating_level_4_modifier
				}
			}
			else_if = {
				limit = {
					var:lepsius_anthropological_rating >= 600
				}
				add_modifier = {
					name = lepsius_anthropological_rating_level_5_modifier
				}
			}
		}
	}
}

lepsius_remove_anthropological_prestige_modifiers_effect = {
	if = {
