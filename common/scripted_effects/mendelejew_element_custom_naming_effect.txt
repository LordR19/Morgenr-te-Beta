mendelejew_element_31_gallium_custom_name_effect = { 
	if = {
		#Only sets global variables if the dynamic element name rule is active.
		limit = { has_game_rule = mendelejew_dynamic_element_names_rule }

		#French Countries
		if = {
			limit = {
				any_primary_culture = {
					OR = {
						cu:french ?= this
						cu:occitan ?= this
					}
				}
				NOT = { 
					any_in_global_list
						custom_element_31_gallium_global_var}

			set_global_variable = custom_element_31_gallium_global_var
		}

		#German Countries
		else_if = {
			limit = {
				any_primary_culture = {
					OR = {
						cu:north_german ?= this
						cu:south_german ?= this
					}
				}
			}

			set_global_variable = custom_element_31_alemannium_global_var
		}

		#Alemannic Countries
		else_if = {
			limit = {
				any_primary_culture = {
					cu:alemannic ?= this
				}
			}

			set_global_variable = custom_element_31_helvetium_global_var
		}

		#Flemish and Walonian Countries
		else_if = {
			limit = {
				any_primary_culture = {
					OR = {
						cu:flemish ?= this
						cu:wallonian ?= this
					}
				}
			}

			set_global_variable = custom_element_31_belgicum_global_var
		}

		#Italian Countries
		else_if = {
			limit = {
				any_primary_culture = {
					OR = {
						cu:north_italian ?= this
						cu:south_italian ?= this
					}
				}
			}

			set_global_variable = custom_element_31_florentium_global_var
		}

		#English and Welsh Countries
		else_if = {
			limit = {
				any_primary_culture = {
					OR = {
						cu:british ?= this
						cu:welsh ?= this
					}
				}
			}

			set_global_variable = custom_element_31_icenium_global_var
		}

		#Irish Countries
		else_if = {
			limit = {
				any_primary_culture = {
					cu:irish ?= this
				}
			}

			set_global_variable = custom_element_31_druidium_global_var
		}

		#Japanese Countries
		else_if = {
			limit = {
				any_primary_culture = {
					cu:japanese ?= this
				}
			}

			set_global_variable = custom_element_31_nihonium_global_var
		}

		#Greek Countries
		else_if = {
			limit = {
				any_primary_culture = {
					cu:greek ?= this
				}
			}

			set_global_variable = custom_element_31_laconium_global_var
		}

		#Turkish Countries
		else_if = {
			limit = {
				any_primary_culture = {
					cu:turkish ?= this
				}
			}

			set_global_variable = custom_element_31_gumus_global_var
		}

		#Default no rule
		else = {
			set_global_variable = custom_element_31_luminor_global_var
		}
	}
