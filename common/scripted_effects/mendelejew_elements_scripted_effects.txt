#Effect to add a theorized late game element
mendelejew_add_lategame_theorized_element = {
	if = {
		limit = {
			NOT = {
				has_global_variable = mendelejew_theorized_element_$NUMBER$_technetium_var
			}
		}
		set_global_variable = mendelejew_theorized_element_$NUMBER$_technetium_var
		set_global_variable = {
			name = mendelejew_theorized_element_$NUMBER$_technetium_country_global_var
			value = this
		}
		country_definition = {
			set_global_variable = {
				name = mendelejew_theorized_element_$NUMBER$_technetium_global_var
				value = this
			}
		}
		change_global_variable = {
			name = mendelejew_theorized_elements_global_number
			add = 1
		}
		var:mendelejew_chemist_var = {
			change_variable = {
				name = mendelejew_theorized_elements_character_number
				add = 1
			}
		}
	}
}
