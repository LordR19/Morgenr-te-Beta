######THEORY, DISCOVERY AND ISOLATION ELEMENTS######

#Effect to add a specific thought element
mendelejew_add_thought_element_effect = {
	#NUMBER --> Number of the Element in the Periodic Table

	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = value
		value = $NUMBER$
	}

	#Actual Effect
	if = {
		limit = {
			NOT = {
				any_in_global_list = {
					variable = mendelejew_theorized_elements_global_list
					this = $NUMBER$
				}
			}
		}
		set_variable = {
			name = mendelejew_thought_variable
			value = $NUMBER$
		}
	}
}
