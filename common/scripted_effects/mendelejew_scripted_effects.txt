			}
		}
		mendelejew_add_postgame_theorized_element = yes
	}
	###Late game theoretical Elements (With Modern Physics)###
	else_if = {
		limit = {
			any_scope_building = {
				has_active_production_method = pm_mendelejew_spectroscopy
				has_active_production_method = pm_curie_physics_modern
			}
		}
		mendelejew_add_lategame_theorized_element = yes
	}
	###Mid game theoretical Element###
	else_if = {
		limit = {
			any_scope_building = {
				has_active_production_method = pm_mendelejew_spectroscopy
			}
			NOT = {
				any_scope_building = {
					has_active_production_method = pm_curie_physics_modern
				}
			}
		}
		mendelejew_add_midgame_theorized_element = yes
	}
	###Early game theoretical Element###
	else_if = {
		limit = {
			any_scope_building = {
				has_active_production_method = pm_mendelejew_chemistry
			}
		}
		mendelejew_add_earlygame_theorized_element = yes
	}
	###Pre-game theoretical Element###
	else_if = {
		limit = {
			any_scope_building = {
				has_active_production_method = pm_mendelejew_chemistry_classical
			}
		}
		mendelejew_add_pregame_theorized_element = yes
	}
}
