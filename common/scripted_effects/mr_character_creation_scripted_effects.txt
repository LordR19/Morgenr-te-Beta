create_gaudi_antoni_effect = {
	create_character = {
		first_name = gaudi_antoni
		last_name = gaudi_gaudi
		culture = cu:catalan
		female = no
		religion = rel:catholic
		interest_group = ig:ig_intelligentsia
		dna = mr_cat_gaudi_antoni
		birth_date = 1852.6.25
		historical = yes
		traits = { 
			engineer
			gaudi_civil_engineer
			erudite
			reserved
			meticulous
			pious
		}

		save_scope_as = gaudi_new_engineer_scope
	}

	scope:gaudi_new_engineer_scope = {
		set_variable = mr_is_engineer_var
		engineer_experience_set_up_effect = yes
		place_character_in_void = 1200
	}
}
