#Simply adds variables to new countries
mr_add_variables = {
	if = {
		limit = {
			NOT = {
				has_variable = lepsius_variables_var
			}
		}
		lepsius_add_variables = yes
	}
	if = {
		limit = {
			NOT = {
				has_variable = verrier_variables_var
			}
		}
		verrier_add_variables = yes
	}
	if = {
		limit = {
			NOT = {
				has_variable = elgar_variables_var
			}
		}
		elgar_add_variables = yes
	}
	if = {
		limit = {
			NOT = {
				has_variable = dubois_variables_var
			}
		}
		dubois_add_variables = yes
	}
}

#Copies certain percentages of Variables to a newly released country
mr_copy_percentage_of_variables = {
	#Adds 90% of the releasing nations music tradition to the new nation
	set_variable = {
		name = elgar_ninety_percent_music_tradition
		value = root.var:elgar_music_tradition
	}
	change_variable = {
		name = elgar_ninety_percent_music_tradition
		multiply = 0.9
	}
	set_variable = {
		name = elgar_music_tradition
		value = var:elgar_ninety_percent_music_tradition
	}
	round_variable = {
		name = elgar_music_tradition
		nearest = 1
	}

	#Adds 90% of the releasing nations astronomical objects to the new nation
	set_variable = {
		name = elgar_ninety_percent_astronomical_objects
		value = root.var:verrier_astronomical_objects
	}
	change_variable = {
		name = elgar_ninety_percent_astronomical_objects
		multiply = 0.9
	}
	set_variable = {
		name = verrier_astronomical_objects
		value = var:elgar_ninety_percent_astronomical_objects
	}
	round_variable = {
		name = verrier_astronomical_objects
		nearest = 1
	}

	#Adds 20% of the releasing nations specimen to the new nation
	set_variable = {
		name = dubois_twenty_percent_specimen
		value = root.var:dubois_specimen
	}
	change_variable = {
		name = dubois_twenty_percent_specimen

	}

	#Adds 20% of the releasing nations fossils to the new nation

		name = dubois_twenty_percent_fossils

	}
	change_variable = {
