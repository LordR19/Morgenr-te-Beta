### Cholera ###

# This effect adds the Cholera Pathogen
theiler_add_cholera_pathogen_effect = {
	custom_tooltip = {
		text = theiler_add_cholera_pathogen_effect_tt
		set_variable = theiler_cholera_pathogen_var
		if = {
			limit = {
				NOT = {
					has_global_variable = theiler_cholera_pathogen_invented_global_var
				}
			}
			set_global_variable = {
				name = theiler_cholera_pathogen_year_global_var
				value = year
			}
			country_definition = {
				set_global_variable = {
					name = theiler_cholera_pathogen_invented_global_var
					value = this
				}
			}
			var:theiler_biologist_var = {
				set_variable = theiler_cholera_pathogen_inventor_var
			}
			add_modifier = {
				name = panum_medical_avant_garde_modifier
				years = 5
			}
			biologist_experience_gain_effect = { AMOUNT = 40 }
			panum_medium_add_technology = yes
			hidden_effect = {
				every_country = {
					limit = {
						is_player = yes
					}
					post_notification = theiler_pathogen_cholera_notification
				}
			}
			#For each First Pathogen, add 6 to the medical score
			panum_add_x_medical_score_points_effect = { AMOUNT = 6 }
		}
		else = {
			biologist_experience_gain_effect = { AMOUNT = 20 }
			panum_small_add_technology = yes
			#For each Pathogen, add 3 to the medical score
			panum_add_x_medical_score_points_effect = { AMOUNT = 3 }
		}
	}
}

#Effect to add $AMOUNT$ Cholera Pathogen Clues
theiler_add_x_cholera_pathogen_clues_effect = {
	# Amount = amount of Clues to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = cholera_pathogen_clues_amount
		value = $AMOUNT$
	}
	
	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = theiler_add_x_cholera_pathogen_clues_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = theiler_add_x_cholera_pathogen_clues_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = theiler_add_x_cholera_pathogen_clues_plus_x_tt
	}

	#This part adds the AMOUNT of Clues
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = theiler_cholera_pathogen_clues_var
			}
		}
		set_variable = {
			name = theiler_cholera_pathogen_clues_var
			value = 0
		}
	}
	change_variable = {
		name = theiler_cholera_pathogen_clues_var
		add = scope:cholera_pathogen_clues_amount
	}
}
