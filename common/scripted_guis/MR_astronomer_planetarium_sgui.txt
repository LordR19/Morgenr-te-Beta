	scope = country
	is_shown = {
		OR = {
			has_global_variable = verrier_active_comet_global_var
			has_global_variable = verrier_1843_comet_discoverer_global_var
			has_global_variable = verrier_1858_comet_discoverer_global_var
			has_global_variable = verrier_1861_comet_discoverer_global_var
			has_global_variable = verrier_1874_comet_discoverer_global_var
			has_global_variable = verrier_1882_comet_discoverer_global_var
		}
	}
}
