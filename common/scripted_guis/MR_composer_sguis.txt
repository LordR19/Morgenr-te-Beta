#MR_MOZART_MONUMENT_BUTTON
mr_mozart_monument_button = {
    scope = country
	is_shown = {
		any_scope_state = {
			state_region = s:STATE_TYROL
		}
		is_player = yes
		NOT = {
			has_global_variable = elgar_mozart_monument_var
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_NO_WAR_TRIGGER_TT
			is_at_war = no
		}
		custom_tooltip = {
            text = MR_ROMANTICISM_TECH_TRIGGER_TT
            has_technology_researched = romanticism
        }
		custom_tooltip = {
            text = MR_OPERA_LEVEL_4_HIGHER_TRIGGER_TT
            any_scope_building = {
				is_building_type = building_elgar_opera
				level >= 4
			}
        }
		custom_tooltip = {
			text = MR_MUSIC_TRADITION_50_TRIGGER_TT
			var:elgar_music_tradition >= 50
		}
    }
    effect = {
		trigger_event = { 
			id = elgar.111
			popup = yes
		}
		set_global_variable = elgar_mozart_monument_var
    }
}

#MR_INVITE_COMPOSER_BUTTON
mr_invite_composer_button = {
    scope = country
	saved_scopes = {
        chosen_composer
    }
    is_valid = {
		#not hired a Travelling Composer recently
		custom_tooltip = {
			text = MR_NOT_HIRED_TRAVELLING_COMPOSER_RECENTLY_TRIGGER_TT
			NOT = {
				has_modifier = elgar_recently_hired_travelling_composer_modifier
			}
		}

		#Opera Triggers
		trigger_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 500 }
			custom_tooltip = {
				text = MR_OPERA_LEVEL_10_HIGHER_TRIGGER_TT
				any_scope_building = {
					is_building_type = building_elgar_opera
					level >= 10
				}
			}
		}
		trigger_else_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 450 }
			custom_tooltip = {
				text = MR_OPERA_LEVEL_5_HIGHER_TRIGGER_TT
				any_scope_building = {
					is_building_type = building_elgar_opera
					level >= 5
				}
			}
		}
		trigger_else_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 350 }
			custom_tooltip = {
				text = MR_OPERA_LEVEL_4_HIGHER_TRIGGER_TT
				any_scope_building = {
					is_building_type = building_elgar_opera
					level >= 4
				}
			}
		}
		trigger_else_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 250 }
			custom_tooltip = {
				text = MR_OPERA_LEVEL_3_HIGHER_TRIGGER_TT
				any_scope_building = {
					is_building_type = building_elgar_opera
					level >= 3
				}
			}
		}
		trigger_else_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 150 }
			custom_tooltip = {
				text = MR_OPERA_LEVEL_2_HIGHER_TRIGGER_TT
				any_scope_building = {
					is_building_type = building_elgar_opera
					level >= 2
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = MR_OPERA_BUILDING_TRIGGER_TT
				any_scope_building = {
					is_building_type = building_elgar_opera
					level >= 1
				}
			}
		}

		#Tech Triggers
		custom_tooltip = {
			text = MR_CLASSICISM_TECH_TRIGGER_TT
			has_technology_researched = elgar_classicism_tech
		}
		trigger_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 150 }
			custom_tooltip = {
				text = MR_ROMANTICISM_TECH_TRIGGER_TT
				has_technology_researched = romanticism
			}
		}
		trigger_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 250 }
			custom_tooltip = {
				text = MR_IRRATIONALISM_TECH_TRIGGER_TT
				has_technology_researched = elgar_irrationalism_tech
			}
		}
		trigger_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 350 }
			custom_tooltip = {
				text = MR_REALISM_TECH_TRIGGER_TT
				has_technology_researched = realism
			}
		}
		trigger_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 450 }
			custom_tooltip = {
				text = MR_TIER_1_TECH_TRIGGER_TT
				OR = {
					has_technology_researched = elgar_impressionism_tech
					has_technology_researched = elgar_expressionism_tech
					has_technology_researched = elgar_art_nouveau_tech
				}
			}
		}

		#Musical Tradition Triggers
		trigger_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 450 }
			custom_tooltip = {
				text = MR_MUSIC_TRADITION_300_TRIGGER_TT
				var:elgar_music_tradition >= 300
			}
		}
		trigger_else_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 350 }
			custom_tooltip = {
				text = MR_MUSIC_TRADITION_200_TRIGGER_TT
				var:elgar_music_tradition >= 200
			}
		}
		trigger_else_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 250 }
			custom_tooltip = {
				text = MR_MUSIC_TRADITION_100_TRIGGER_TT
				var:elgar_music_tradition >= 100
			}
		}

		#Power Rank Triggers
		trigger_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 450 }
			custom_tooltip = {
				text = MR_GREAT_POWER_TRIGGER_TT
				country_rank >= rank_value:great_power
			}
		}
		trigger_else_if = {
			limit = { scope:chosen_composer.var:elgar_composer_experience >= 350 }
			custom_tooltip = {
				text = MR_MAJOR_POWER_HIGHER_TRIGGER_TT
				country_rank >= rank_value:major_power
			}
		}

		#National Composer
		trigger_if = {
			limit = {
				scope:chosen_composer = { has_trait = elgar_national_composer }
			}
			custom_tooltip = {
				text = MR_NATIONAL_COMPOSER_TRIGGER_TT
				scope:chosen_composer = {
					culture = {
						shares_non_heritage_trait_with_any_primary_culture = root
						shares_heritage_trait_with_any_primary_culture = root
					}
				}
			}
		}
    }
    effect = {
		custom_tooltip = {
			text = MR_INVITE_COMPOSER_BUTTON_EFFECT_TT
			set_variable = {
				name = elgar_composer_var
				value = scope:chosen_composer
			}

			set_local_variable = mr_invite_composer_button_local_var #Used fro Error prevention
			if = { #Used for error prevention
				limit = { has_local_variable = mr_invite_composer_button_local_var }
				var:elgar_composer_var = {
					free_character_from_void = yes
					transfer_character = root
					remove_variable = mr_is_travelling_composer_var
					remove_trait = elgar_travelling_composer
					set_variable = mr_is_composer_var
					place_character_in_void = 1200
				}
		
				artists_set_up_travelling_composer_list_effect = yes
			}
		}

		add_modifier = {
			name = elgar_recently_hired_travelling_composer_modifier
			years = 5
		}
		scope:chosen_composer = {
			elgar_composer_character_set_up_cost_effect = yes
		}
    }
}
