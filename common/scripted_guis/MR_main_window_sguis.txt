#Window Variable Trigger and Toggle
mr_main_window_toggle_sgui = {
	is_shown = {
		has_variable = mr_main_window_is_active_var
	}
	effect = {
		if = {
			limit = {
				NOT = {
					has_variable = mr_main_window_is_active_var
				}
			}
			set_variable = mr_main_window_is_active_var
		}
		else = {
			remove_variable = mr_main_window_is_active_var
		}
	}
}
