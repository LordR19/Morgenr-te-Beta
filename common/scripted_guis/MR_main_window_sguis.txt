#Button visibility
# Checks if the Community GUI Framework is active
mr_main_button_visibility_sgui = {
	is_shown = {
		community_gui_is_active = no
	}
}

#Window Variable to Toggle through the main windows
mr_main_window_toggle_through_sgui = {
	effect = {
		mr_main_window_toggle_through_effect = yes
	}
}

mr_main_window_toggle_off_sgui = {
	effect = {
		set_variable = {
			name = mr_main_window_last_active_tab_var
			value = var:mr_main_window_is_active_var
		}
		set_variable = {
			name = mr_main_window_is_active_var
			value = 0
		}
		GUI_remove_unnecessary_variables_effect = yes
	}
}

mr_main_window_visible_sgui = {
	is_shown = {
		has_variable = mr_main_window_is_active_var
		NOT = {
			var:mr_main_window_is_active_var = 0
		}
	}
}
