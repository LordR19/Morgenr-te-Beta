#Window Variable to Toggle through the main windows
mr_main_window_toggle_through_sgui = {
	effect = {
		if = {
			limit = {
				NOT = {
					has_variable = mr_main_window_is_active_var
				}
			}
			set_variable = {
				name = mr_main_window_is_active_var
				value = 1
			}
		}
		if = {
			limit = {
				has_variable = mr_main_window_last_active_tab_var
				var:mr_main_window_is_active_var = 0
			}
			set_variable = {
				name = mr_main_window_is_active_var
				value = var:mr_main_window_last_active_tab_var
			}
		}
		else = {
			if = {
				limit = {
					var:mr_main_window_is_active_var < 3
				}
				change_variable = {
					name = mr_main_window_is_active_var
					add = 1
				}
			}
			else = {
				set_variable = {
					name = mr_main_window_is_active_var
					value = 0
				}
			}
		}
		remove_variable = mr_main_window_last_active_tab_var
	}
}

mr_main_window_toggle_off_sgui = {
	effect = {
		set_variable = {
			name = mr_main_window_last_active_tab_var
			value = var:mr_main_window_is_active_var
		}
		set_variable = {
			name = mr_main_window_is_active_var
			value = 0
		}
	}
}

mr_main_window_visible_sgui = {
	is_shown = {
		has_variable = mr_main_window_is_active_var
		NOT = {
			var:mr_main_window_is_active_var = 0
		}
	}
}
