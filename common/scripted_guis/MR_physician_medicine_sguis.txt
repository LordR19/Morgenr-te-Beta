#Activates the Information for Cholera
mr_physician_cholera_catalogue_button_sgui = {
	scope = country
	is_valid = {
		NOT = {
			has_variable = GUI_panum_cholera_information_display_visible_var
		}
	}
	effect = {
		panum_remove_all_diseases_catalogue_variables_effect = yes
		set_variable = GUI_panum_cholera_information_display_visible_var
	}
}

mr_physician_cholera_catalogue_active_sgui = {
	is_shown = {
		has_variable = GUI_panum_cholera_information_display_visible_var
	}
}
