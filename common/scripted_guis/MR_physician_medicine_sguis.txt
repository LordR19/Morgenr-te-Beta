#Activates the Information for Cholera
mr_physician_cholera_catalogue_button_sgui = {
	scope = country
	is_valid = {
		NOT = {
			has_variable = GUI_panum_cholera_information_display_visible_var
		}
	}
	effect = {
		panum_remove_all_diseases_catalogue_variables_effect = yes
		set_variable = GUI_panum_cholera_information_display_visible_var
	}
}
#Checks if this is button is active
mr_physician_cholera_catalogue_active_sgui = {
	is_shown = {
		has_variable = GUI_panum_cholera_information_display_visible_var
	}
}
#Is Cholera described?
mr_physician_cholera_first_description_active_sgui = {
	is_shown = {
		has_global_variable = panum_cholera_description_invented_global_var
	}
}
#Is Cholera Treatment discovered?
mr_physician_cholera_first_treatment_active_sgui = {
	is_shown = {
		has_global_variable = panum_cholera_treatment_invented_global_var
	}
}
#Is pathogen discovered?
mr_physician_cholera_pathogen_discovered_active_sgui = {
	is_shown = {
		has_global_variable = theiler_cholera_pathogen_invented_global_var
	}
}

mr_physician_cholera_vaccine_invented_active_sgui = {
