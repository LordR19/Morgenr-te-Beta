### PROMOTE SCIENTIST BUTTONS ###
mr_academics_promote_engineer = {
    scope = country
    is_valid = {
        hidden_trigger = {
            NOT = {
                has_variable = academics_promote_engineer_active
                has_variable = academics_academy_promotion_active
            }
        }
        custom_tooltip = {
            text = MR_ACADEMICS_PROMOTE_UNIVERSITY_MATHEMATICS_TT
            any_scope_state = {
                is_production_method_active = {
                    building_type = building_university
                    production_method = pm_dubois_university_mathematics
                }
            }
        }
    }

    effect = {
        custom_tooltip = {
            text = MR_ACADEMICS_PROMOTE_ENGINEER_TT
            set_variable = academics_promote_engineer_active
            set_variable = academics_academy_promotion_active
        }
    }

    ai_chance = {
        base = 100
        modifier = {
            trigger = {
                OR = {
                    c:PRU ?= ROOT
                    c:ITA ?= ROOT
                    c:SWI ?= ROOT
                    c:GER ?= ROOT
                    c:AUS ?= ROOT
                    c:SPA ?= ROOT
                    c:BRZ ?= ROOT
                }
            }
            add = 50
        }
        modifier = {
            trigger = {
                verrier_astronomical_observation = yes
            }
            add = 50
        }
        modifier = {
            trigger = {
                country_rank < rank_value:major_power
            }
            add = 50
        }
        modifier = {
            trigger = {
                NOR = {
                    dubois_has_formation = yes
                    dubois_can_discover_formation = yes
                    any_country = {
                        has_diplomatic_pact = {
                            who = ROOT
                            type = paleontological_license
                        }
                    }
                    dubois_has_rainforest = yes
                    any_country = {
                        has_diplomatic_pact = {
                            who = ROOT
                            type = botanical_license
                        }
                    }
                    lepsius_has_excavation_site = yes
                    any_country = {
                        has_diplomatic_pact = {
                            who = ROOT
                            type = excavation_license
                        }
                    }
                }
            }
            add = 200
        }
        modifier = {
            trigger = {
                gaudi_has_engineer = yes
            }
            add = -1000
        }   
    }
}

mr_academics_academy_investment = {
    scope = country

    is_valid = {
        custom_tooltip = {
            text = MR_ACADEMICS_PROMOTE_UNIVERSITY_INVESTMENT_TT
            NOT = {
                has_Modifier = dubois_academy_investment_modifier
            }
        }
        custom_tooltip = {
            text = MR_ACADEMICS_PROMOTE_UNIVERSITY_NEUTRAL_TT
            any_scope_state = {
                is_production_method_active = {
                    building_type = building_university
                    production_method = pm_dubois_university_standard
                }
            }
        }
    }

    effect = {
        if = {
			limit = {
				has_modifier = dubois_removed_academy_investment_modifier
			}
			remove_modifier = dubois_removed_academy_investment_modifier
		}
		add_modifier = {
			name = dubois_academy_investment_modifier
			multiplier = money_amount_multiplier_small
			years = 10
		}
        custom_tooltip = MR_ACADEMICS_PROMOTE_UNIVERSITY_INVESTMENT_EFFECT_TT
    }

    ai_chance = {
        base = 0 #It's just not worth for you, Mr AI  
    }
}
