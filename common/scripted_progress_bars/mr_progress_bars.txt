verrier_satellite_discovery_progress_bar = {
	name = "verrier_satellite_discovery_progress_bar"
	desc = "verrier_satellite_discovery_progress_bar_desc"

	default = yes

	monthly_progress = {
		if = {
			limit = {
				# Astronomer exist (basic)
				owner = {
					var:verrier_astronomer_var = {
						has_trait = verrier_astronomer
					}
				}
			}
			add = {
				desc = "verrier_satellite_discovery_progress_from_astronomer_normal_tt"
				value = 1
			}
		}
		if = {
			limit = {
				# Astronomer is experienced or famous
				var:verrier_astronomer_var = {
					OR = {
						has_trait = verrier_astronomer_experienced
						has_trait = verrier_astronomer_famous
					}
				}
			}
			add = {
				desc = "verrier_satellite_discovery_progress_from_astronomer_experienced_tt"
				value = 2
			}
		}
		if = {
			limit = {
				# Astronomer is moon observer
				var:verrier_astronomer_var = {
					has_trait = verrier_moon_observer
				}
			}
			add = {
				desc = "verrier_satellite_discovery_progress_from_moon_observer_tt"
				value = 1
			}
		}
		if = {
			limit = {
				# Astronomer is mathematician
				var:verrier_astronomer_var = {
					has_trait = verrier_mathematician
				}
			}
			add = {
				desc = "verrier_satellite_discovery_progress_from_mathematician_tt"
				value = 1
			}
		}
		if = {
			limit = {
				# Astronomer is meticulous
				var:verrier_astronomer_var = {
					has_trait = meticulous
				}
			}
			add = {
				desc = "verrier_satellite_discovery_progress_from_meticulous_tt"
				value = 1
			}
		}
		if = {
			limit = {
				# Astronomer has a vice
				var:verrier_astronomer_var = {
					OR = {
						has_trait = alcoholic
						has_trait = opium_addiction
						has_trait = cocaine_addiction
						has_trait = senile
					}
				}
			}
			subtract = {
				desc = "verrier_satellite_discovery_progress_from_moon_vice_tt"
				value = 1
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 48

}
