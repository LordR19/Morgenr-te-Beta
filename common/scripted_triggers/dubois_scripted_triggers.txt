dubois_has_zoo = {
	any_scope_building = {
		is_building_type = building_dubois_zoo
	}
}

dubois_has_zoo_minimum = {
	var:dubois_zoo_animal_total >= 5
}

	OR = {
		######AFRICA######

		#Hippopotamus
		AND = {
			NOT = {
				has_variable = dubois_zoo_animal_hippopotamus
			}
			any_state = {
				has_modifier = dubois_hippopotamus_habitat_modifier
				OR = {
					owner = {
