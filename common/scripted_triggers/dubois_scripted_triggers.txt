dubois_can_discover_formation = {
	OR = {
		dubois_has_paleontologist = yes
		has_technology_researched = dubois_paleontology_tech
		any_scope_building = {
			is_building_type = building_university
			has_active_production_method = pm_dubois_university_natural_history
		}
	}
}

dubois_has_formation = {
	any_scope_state = {
		OR = {
			has_modifier = dubois_normal_formation_modifier
			has_modifier = dubois_great_north_american_formation_modifier
			has_modifier = dubois_great_south_american_formation_modifier
			has_modifier = dubois_great_european_formation_modifier
			has_modifier = dubois_great_african_formation_modifier
			has_modifier = dubois_great_asian_formation_modifier
			has_modifier = dubois_great_australian_formation_modifier
		}
	}
