#Has an Engineer that is not busy
gaudi_has_active_engineer = {
	custom_tooltip = {
		text = lepsius_has_active_engineer_tt
		trigger_if = {
			limit = { has_variable = gaudi_engineer_var }
			var:gaudi_engineer_var = {
				is_busy = no
			}
		}
		trigger_else = {
			always = no
		}
	}
}
