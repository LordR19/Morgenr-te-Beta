#This is used to prevent some historical characters from wearing hats
mr_no_hat_historic_person_trigger = {
	scope:character ?= {
		OR = {
			has_variable = mr_no_headgear_var
			AND = {
				is_female = yes
				OR = {
					has_variable = mr_is_engineer_var
					has_variable = mr_is_astronomer_var
					has_variable = mr_is_chemist_var
					has_variable = mr_is_naturalist_var
					has_variable = mr_is_paleontologist_var
					has_variable = mr_is_archaeologist_var
					has_variable = mr_is_anthropologist_var
					has_variable = mr_is_balloonist_var
					has_variable = mr_is_composer_var
					has_variable = mr_is_travelling_composer_var
					has_variable = mr_is_painter_var
					has_variable = mr_is_writer_var
					has_variable = mr_is_local_writer_var
					has_variable = mr_is_exiled_writer_var
					has_variable = verrier_einstein_character_var
				}
			}
		}
	}
}

mr_european_plus_clothes_trigger = {
	OR = {
		european_clothes_trigger = yes
		AND = {
			scope:character ?= {
				OR = {
					culture = cu:cherokee
					culture = cu:muskogean
				}
			}
		}
		african_diaspora_clothes_trigger = yes
	}
}

mr_academic_clothes_trigger = {
	OR = {
		has_variable = mr_is_engineer_var
		has_variable = mr_is_astronomer_var
		has_variable = mr_is_chemist_var
		has_variable = mr_is_naturalist_var
		has_variable = mr_is_paleontologist_var
		has_variable = mr_is_archaeologist_var
		has_variable = mr_is_anthropologist_var
	}
}

mr_academic_expedition_clothes_trigger = {
	scope:character ?= {
		is_busy = yes
		OR = {
			has_variable = mr_is_anthropologist_var
			AND = {
				has_variable = mr_is_astronomer_var
				has_variable = rapanui_character_on_pacific_expedition_var
			}
			AND = {
				has_variable = mr_is_naturalist_var
				OR = {
					has_variable = dubois_naturalist_on_botanical_expedition_var
					has_variable = dubois_naturalist_on_safari_var
					has_variable = dubois_naturalist_on_zoological_expedition_var
					has_variable = dubois_scientist_on_mountain_expedition_var
					has_variable = rapanui_character_on_pacific_expedition_var
				}
			}
			AND = {
				has_variable = mr_is_paleontologist_var
				OR = {
					has_variable = dubois_scientist_on_mountain_expedition_var
					has_variable = rapanui_character_on_pacific_expedition_var
					has_variable = dubois_paleontologist_in_south_america_var
					has_variable = dubois_paleontologist_in_south_america_var
					has_variable = dubois_paleontologist_in_europe_var
					has_variable = dubois_paleontologist_in_africa_var
					has_variable = dubois_paleontologist_in_asia_var
					has_variable = dubois_paleontologist_in_australia_var
					has_variable = dubois_paleontologist_in_local_var
				}
			}
			AND = {
				has_variable = mr_is_archaeologist_var
				OR = {
					has_variable = lepsius_archaeologist_in_egypt_var
					has_variable = lepsius_archaeologist_in_mesopotamia_var
					has_variable = lepsius_archaeologist_in_greece_var
					has_variable = lepsius_archaeologist_in_yucatan_var
					has_variable = lepsius_archaeologist_in_peru_var
					has_variable = lepsius_archaeologist_in_local_var
				}
			}
		}
	}
}
