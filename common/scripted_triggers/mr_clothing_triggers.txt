######Swiss Army Triggers######
mr_swiss_army_clothing_trigger = {
	OR = {
		AND = {
			exists = scope:character
			scope:character = {
				AND = {
					exists = c:SWI
					exists = home_country
					OR = {
						home_country = c:SWI
						home_country = {
							was_formed_from = SWI
						}
					}
					is_interest_group_type = ig_armed_forces
					AND = {
						has_role = politician
						is_ruler = no
					}
				}
			}
		}
		AND = {
			exists = scope:pop
			scope:pop = {
				is_pop_type = soldiers
				AND = {
					exists = c:SWI
					exists = home_country
					OR = {
						home_country = c:SWI
						home_country = {
							was_formed_from = SWI
						}
					}
				}
			}
		}
		AND = {
			exists = scope:pop
			scope:pop = {
				is_pop_type = officers
				AND = {
					exists = c:SWI
					exists = home_country
					OR = {
						home_country = c:SWI
						home_country = {
							was_formed_from = SWI
						}
					}
				}
			}
		}
	}
}
