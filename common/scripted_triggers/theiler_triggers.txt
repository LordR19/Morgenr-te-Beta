######Biologist CHARACTER TRIGGERS######

theiler_has_biologist = {
	custom_tooltip = {
		text = theiler_has_biologist_tt
		theiler_has_biologist_no_tooltip_trigger = yes
	}
}
theiler_has_biologist_no_tooltip_trigger = {
	has_variable = theiler_biologist_var
}

theiler_has_active_biologist = {
	custom_tooltip = {
		text = theiler_has_active_biologist_tt
		theiler_has_active_biologist_no_tooltip_trigger = yes
	}
}
theiler_has_active_biologist_no_tooltip_trigger = {
	has_variable = theiler_biologist_var
	var:theiler_biologist_var ?= {
		is_busy = no
	}
}

### Triggers for the rating display Biologist ###
theiler_biologist_fantastic_rating_trigger = {
	theiler_biologist_character_rating_value > 8
}

theiler_biologist_very_good_rating_trigger = {
	theiler_biologist_character_rating_value > 5
}

theiler_biologist_good_rating_trigger = {
	theiler_biologist_character_rating_value > 2
}

theiler_biologist_mediocre_rating_trigger = {
	theiler_biologist_character_rating_value > 0
}
