curtiss_seaplanes_tech = {

	era = era_5
	texture = "gfx/interface/icons/invention_icons/curtiss_seaplanes.dds"
	category = military
	
	unlocking_technologies = {
		military_aviation
	}

	modifier = {
		unit_navy_defense_mult = 0.1
		unit_convoy_raiding_mult = 0.05
	}

	ai_weight = {
		value = 1
		
		# Important for naval powers
		if = {
			limit = { navy_size >= 20 }
			add = 0.5
		}		
		#Seaplane countries
		if = {
			limit = { 
				exists = c:FRA
				c:FRA = ROOT 
			}
			add = 0.5
		}	
		if = {
			limit = { 
				exists = c:USA
				c:USA = ROOT 
			}
			add = 0.5
		}		
		if = {
			limit = { 
				exists = c:GBR
				c:GBR = ROOT 
			}
			add = 0.5
		}
		if = {
			limit = { 
				exists = c:ITA
				c:ITA = ROOT 
			}
			add = 0.5
		}
	}
}
