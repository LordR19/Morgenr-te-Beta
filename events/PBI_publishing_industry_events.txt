#PBI 100: Whatever you like
pbi_publishing_industry_events.100 = {
	type = country_event
	placement = scope:target_state

	title = pbi_publishing_industry_events.100.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = scope:target_party_1
					exists = scope:target_ig_3
				}
				desc = pbi_publishing_industry_events.100.d.1
			}
			triggered_desc = {
				trigger = {
					exists = scope:target_party_1
				}
				desc = pbi_publishing_industry_events.100.d.2
			}
			triggered_desc = {
				trigger = {
					exists = scope:target_ig_3
				}
				desc = pbi_publishing_industry_events.100.d.3
			}
			desc = pbi_publishing_industry_events.100.d.4
		}
	}
  flavor = pbi_publishing_industry_events.100.f
  event_image = {
    video = "europenorthamerica_capitalists_meeting"
  }
  on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
  on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/capitalists_meeting"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3
	
	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				has_building = building_manzoni_publishing_industry
			}
			save_scope_as = target_state
		}
		if = {
			limit = {
				any_interest_group = {
					count > 1
					has_party = yes
				}
			}
			random_interest_group = {
				limit = {
					is_in_government = yes
					has_party = yes
				}
				party = {
					save_scope_as = target_party_1
				}
			}
			if = {
				limit = {
					NOT = {
						exists = scope:target_party_1
					}
				}
				random_interest_group = {
					limit = {
						has_party = yes
					}
					save_scope_as = target_party_1
				}
			}
			random_interest_group = {
				limit = {
					has_party = yes
					party = {
						NOT = {
							THIS = scope:target_party_1
						}
					}
				}
				party = {
					save_scope_as = target_party_2
				}
			}
		}
		else = {
			random_interest_group = {
				limit = {
					is_in_government = yes
				}
				save_scope_as = target_ig_1
			}
			random_interest_group = {
				limit = {
					is_in_government = no
					is_marginal = no
				}
				save_scope_as = target_ig_2
			}
		}
		random_interest_group = {
			limit = {
				is_marginal = yes
			}
			save_scope_as = target_ig_3
		}
	}

	option = {
		name = pbi_publishing_industry_events.100.a
		default_option = yes
		add_modifier = {
			name = pbi_country_modifier_promote_social_change
			months = normal_modifier_time
			is_decaying = yes
		}
		every_interest_group = {
			limit = {
				is_in_government = no
			}
			add_modifier = {
				name = pbi_ig_modifier_political_diversity
				months = normal_modifier_time
				is_decaying = yes
			}
		}
		add_radicals = {
			value = medium_radicals
		}
	}
	option = {
		name = pbi_publishing_industry_events.100.b
		add_modifier = {
			name = pbi_country_modifier_suppress_social_change
			months = normal_modifier_time
			is_decaying = yes
		}
		every_interest_group = {
			add_modifier = {
				name = pbi_ig_modifier_resolving_political_discontent
				months = normal_modifier_time
				is_decaying = yes
			}
		}
		add_loyalists = {
			value = medium_radicals
		}
	}
}
