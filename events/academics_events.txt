#Academics 1: Engineer retires
academics.1 = {
	type = country_event

	title = academics.1.t

	desc = academics.1.d

#	flavor = academics.1.f
	
	duration = 3

	gui_window = event_window_1char_tabloid

	left_icon = scope:academics_retired_person_scope

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		NOT = {
			has_variable = academics_engineer_emeritus_delayed_var
		}
		gaudi_has_engineer = yes
		var:gaudi_engineer_var = {
			age >= 65
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			var:gaudi_engineer_var = {
				age >= 70
			}
			factor = 10000
		}
	}

	immediate = {
		var:gaudi_engineer_var = {
			save_scope_as = academics_retired_person_scope
		}
	}

	option = {	#He earned his retirement
		name = academics.1.g
		default_option = yes

		kill_engineer_effect = yes

		if = {
			limit = { has_modifier = academics_engineer_emeritus_delayed_modifier }
			remove_modifier = academics_engineer_emeritus_delayed_modifier
		}
	}

	option = {	#We need him! Pay him extra!
		name = academics.1.h

		trigger = {
			NOT = {
				has_modifier = academics_engineer_emeritus_delayed_modifier
			}
		}

		add_modifier = {
			name = academics_engineer_emeritus_delayed_modifier
		}
		set_variable = {
			name = academics_engineer_emeritus_delayed_var
			years = { 1 5 }
		}

		ai_chance = {
			base = 0
		}
	}
}

#Academics 50: Academy Progress

academics.50 = {
	type = country_event

	hidden = yes

	trigger = {
		any_scope_building = { #No academy progress without academy
			is_building_type = building_university
		}
	}

	immediate = {
		dubois_add_academy_progress = yes #basic progress

		if = { #Literacy 25%
			limit = {
				literacy_rate >= 0.25
			}
			dubois_add_academy_progress = yes
		}
		if = { #Literacy 50%
			limit = {
				literacy_rate >= 0.5
			}
			dubois_add_academy_progress = yes
		}
		if = { #Literacy 75%
			limit = {
				literacy_rate >= 0.75
			}
			dubois_add_academy_progress = yes
		}
		if = { #Academics at least 0.5%
			limit = {
				pop_type_percent_country = {
					pop_type = academics
					percent >= 0.005
				}
			}
			dubois_add_academy_progress = yes
		}
		if = { #Academics at least 1%
			limit = {
				pop_type_percent_country = {
					pop_type = academics
					percent >= 0.01
				}
			}
			dubois_add_academy_progress = yes
		}
		if = { #Academics at least 2%
			limit = {
				pop_type_percent_country = {
					pop_type = academics
					percent >= 0.02
				}
			}
			dubois_add_academy_progress = yes
		}
		if = { #Institution Level 2
			limit = {
            	institution_investment_level = {
                	institution	= institution_schools
                	value = 2
            	}
        	}
			dubois_add_academy_progress = yes
		}
		if = { #Institution Level 4
			limit = {
            	institution_investment_level = {
                	institution	= institution_schools
                	value = 4
            	}
        	}
			dubois_add_academy_progress = yes
		}
		if = { #Intelligentsia is content
			limit = {
            	ig:ig_intelligentsia = {
					ig_approval >= 5
				}
        	}
			dubois_add_academy_progress = yes
		}
		if = { #Special Tech School
			limit = {
				OR = {
					has_modifier = academics_avantgarde_intelligentsia_modifier
					has_modifier = academics_practical_academia_modifier
					has_modifier = academics_useful_ivory_tower_modifier
				}
        	}
			dubois_add_academy_progress = yes
		}

		if = { #Observatory gives mathematical advantage
			limit = {
				has_variable = dubois_promote_mathematics_active
				any_scope_building = {
					is_building_type = building_verrier_observatory
				}
        	}
			dubois_add_academy_progress = yes
		}

		if = { #Zoo gives naturalistic advantage
			limit = {
				has_variable = dubois_promote_natural_history_active
				any_scope_building = {
					is_building_type = building_dubois_zoo
				}
        	}
			dubois_add_academy_progress = yes
		}

		if = { #Archaoelogical Museum gives anthropological advantage
			limit = {
				has_variable = dubois_promote_humanities_active
				any_scope_building = {
					is_building_type = building_lepsius_archaeological_museum
				}
        	}
			dubois_add_academy_progress = yes
		}

		if = { #Academy Investment
			limit = {
            	has_modifier = dubois_academy_investment_modifier
        	}
			dubois_add_academy_progress = yes
		}
	}
}

	flavor = academics.2009.f

		texture = "gfx/event_pictures/verrier_pictures/verrier_milky_way.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	cooldown = {
		years = { 50 100 }
	}

	trigger = {
		academics_astronomical_mission_possible_trigger = yes #does not have an astronomical mission or 3 or more missions already
		has_building = building_verrier_observatory
		any_country = {
			NOT = {
				this = root
			}
			var:verrier_astronomical_objects >= academics_has_10_or_more_astronomical_objects
			var:verrier_astronomical_objects <= academics_has_100_or_less_astronomical_objects_more
		}
		ig:ig_intelligentsia = {
			is_marginal = no
		}
	}

	immediate = {
		random_country = {
			limit = {
				NOT = {
					this = root
				}
				var:verrier_astronomical_objects >= academics_has_10_or_more_astronomical_objects
				var:verrier_astronomical_objects <= academics_has_100_or_less_astronomical_objects_more
			}
			save_scope_as = rival_scope
		}

		
		set_variable = {
			name = je_rival_nation_var
			value = scope:rival_scope
		}
	}

	option = { #Let us beat them!
		name = academics.2009.g

		add_journal_entry = {
			type = je_academics_rivalry_for_stars
			target = scope:rival_scope
		}

		custom_tooltip = academics.2009_goal_tt
		custom_tooltip = academics.2009_reward_tt
	}

	option = { #This is not a space race...
		name = academics.2009.h
		default_option = yes

		add_modifier = {
			name = academics_unhappy_intelligentsia
			years = 2
		}

		remove_variable = je_rival_nation_var
	}
}

			AND = {
				var:lepsius_egyptian_artifact_minor >= 25
				var:lepsius_egyptian_artifact_minor <= 75
			}
			AND = {
				var:lepsius_mesopotamian_artifact_minor >= 35
				var:lepsius_mesopotamian_artifact_minor <= 75
			}
			AND = {
				var:lepsius_aegean_artifact_minor >= 35
				var:lepsius_aegean_artifact_minor <= 75
			}
			AND = {
				var:lepsius_precolumbian_artifact_minor >= 35
				var:lepsius_precolumbian_artifact_minor <= 75
			}
		}

			is_marginal = no

#Academics 2116: We need to find more artifacts than [Scope Rival Nation]
academics.2116 = {
	type = country_event

	title = academics.2116.t

	desc = academics.2116.d

	flavor = academics.2116.f

	placement = ROOT

	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/lepsius_pictures/museum/lepsius_museum_world_cultures_collection.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	cooldown = {
		years = { 50 100 }
	}

	trigger = {
		academics_archaeological_mission_possible_trigger = yes #does not have an Archaeological mission
		lepsius_has_archaeological_museum = yes
		lepsius_has_archaeologist = yes
		var:lepsius_total_artifacts > 10
		any_country = {
			NOT = {
				this = root
			}
			var:lepsius_total_artifacts >= academics_has_10_or_more_total_artifacts
			var:lepsius_total_artifacts <= academics_has_50_or_less_total_artifacts_more
		}
		ig:ig_intelligentsia = {
			is_marginal = no
		}
	}

	immediate = {
		random_country = {
			limit = {
				NOT = {
					this = root
				}
				var:lepsius_total_artifacts >= academics_has_10_or_more_total_artifacts
				var:lepsius_total_artifacts <= academics_has_50_or_less_total_artifacts_more
			}
			save_scope_as = rival_scope
		}
		
		set_variable = {
			name = je_artifacts_rival_nation_var
			value = scope:rival_scope
		}
	}

	option = { #Let us beat them!
		name = academics.2116.g

		add_journal_entry = {
			type = je_academics_rivalry_for_artifacts
			target = scope:rival_scope
		}

		custom_tooltip = academics.2116_goal_tt
		custom_tooltip = academics.2116_reward_tt
	}

	option = { #I am an Archaeologist and I'm digging a hole!
		name = academics.2116.h
		default_option = yes

		add_modifier = {
			name = academics_unhappy_intelligentsia
			years = 2
		}

		remove_variable = je_artifacts_rival_nation_var
	}
}

#Academics 2300: We need to find more fossils than [Scope Rival Nation]
academics.2300 = {
	type = country_event

	title = academics.2300.t

	desc = academics.2300.d

	flavor = academics.2300.f

	placement = ROOT

	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/dubois_pictures/dubois_fossil_mixed.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	cooldown = {
		years = { 50 100 }
	}

	trigger = {
		academics_paleontologist_mission_possible_trigger = yes #does not have an astronomical mission
		dubois_has_paleontologist = yes
		var:dubois_fossil > 5
		any_country = {
			NOT = {
				this = root
			}
			var:dubois_fossil >= academics_has_15_or_more_total_fossils
			var:dubois_fossil <= academics_has_35_or_less_total_fossils_more
		}
		ig:ig_intelligentsia = {
			is_marginal = no
		}
	}

	immediate = {
		random_country = {
			limit = {
				NOT = {
					this = root
				}
				var:dubois_fossil >= academics_has_15_or_more_total_fossils
				var:dubois_fossil <= academics_has_35_or_less_total_fossils_more
			}
			save_scope_as = rival_scope
		}
		
		set_variable = {
			name = je_fossil_rival_nation_var
			value = scope:rival_scope
		}
	}

	option = { #Ah, a fossil feud!
		name = academics.2300.g

		add_journal_entry = {
			type = je_academics_rivalry_for_fossils
			target = scope:rival_scope
		}

		custom_tooltip = academics.2300_goal_tt
		custom_tooltip = academics.2300_reward_tt
	}

	option = { #We do not need any dusty rocks...
		name = academics.2300.h
		default_option = yes

		add_modifier = {
			name = academics_unhappy_intelligentsia
			years = 2
		}

		remove_variable = je_fossil_rival_nation_var
	}
}

#Academics 2301: Fantastic Fossil Collection
academics.2301 = {
	type = country_event

	title = academics.2301.t

	desc = {
		triggered_desc = { #In case your Rival still exists
			desc = academics.2301.d_rival
			trigger = {
				exists = var:je_fossil_rival_nation_var
			}
		}
		triggered_desc = { #In case your Rival does not exist anymore
			desc = academics.2301.d_rival_gone
			trigger = {
				NOT = {
					exists = var:je_fossil_rival_nation_var
				}
			}
		}
	}
	flavor = academics.2301.f

	placement = ROOT

	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/dubois_pictures/dubois_fossil_mixed.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	trigger = {
		#triggered by successful je
	}

	immediate = {
	}

	option = { #This puts an end to this feud!
		name = academics.2301.g

		add_modifier = {
			name = academics_competition_with_rival_won_fossils
			years = 4
		}

		remove_variable = je_fossil_rival_nation_var
	}
}
