#Academics 1: Scientist retires (before 70)
academics.1 = {
	type = country_event

	title = academics.1.t

	desc = academics.1.d

	flavor = academics.1.f
	
	duration = 3

	gui_window = event_window_1char_tabloid

	left_icon = scope:academics_retired_person_scope

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		OR = {
			has_variable = gaudi_engineer_var
			has_variable = verrier_astronomer_var
			has_variable = dubois_naturalist_var
			has_variable = dubois_paleontologist_var
			has_variable = lepsius_archaeologist_var
			has_variable = curtiss_balloonist_var
		}
	}

	immediate = {
		random_list = {
			1 = {
				trigger = {
					has_variable = gaudi_engineer_var
				}
				set_variable = kill_engineer_var
				var:gaudi_engineer_var = {
					save_scope_as = academics_retired_person_scope
				}
			}
			1 = {
				trigger = {
					has_variable = verrier_astronomer_var
				}
				set_variable = kill_astronomer_var
				var:verrier_astronomer_var = {
					save_scope_as = academics_retired_person_scope
				}
			}
			1 = {
				trigger = {
					has_variable = dubois_naturalist_var
				}
				set_variable = kill_naturalist_var
				var:dubois_naturalist_var = {
					save_scope_as = academics_retired_person_scope
				}
			}
			1 = {
				trigger = {
					has_variable = dubois_paleontologist_var
				}
				set_variable = kill_palentologist_var
				var:dubois_paleontologist_var = {
					save_scope_as = academics_retired_person_scope
				}
			}
			1 = {
				trigger = {
					has_variable = lepsius_archaeologist_var
				}
				set_variable = kill_archaeologist_var
				var:lepsius_archaeologist_var = {
					save_scope_as = academics_retired_person_scope
				}
			}
			1 = {
				trigger = {
					has_variable = curtiss_balloonist_var
				}
				set_variable = kill_ballonist_var
				var:curtiss_balloonist_var = {
					save_scope_as = academics_retired_person_scope
				}
			}
		}
	}

	option = {	#He earned his retirement (non balloonist)
		name = academics.1.g
		default_option = yes

		trigger = {
			NOT = {
				has_variable = kill_ballonist_var
			}
		}

		if = {
			limit = {
				has_variable = kill_engineer_var
			}
			kill_engineer_effect = yes
		}
		else_if = {
			limit = {
				has_variable = kill_astronomer_var
			}
			kill_astronomer_effect = yes
		}
		else_if = {
			limit = {
				has_variable = kill_naturalist_var
			}
			kill_naturalist_effect = yes
		}
		else_if = {
			limit = {
				has_variable = kill_palentologist_var
			}
			kill_paleontologist_effect = yes
		}
		else_if = {
			limit = {
				has_variable = kill_archaeologist_var
			}
			kill_archaeologist_effect = yes
		}
	}

	option = {	#He earned his retirement (balloonist)
		name = academics.1.g
		default_option = yes

		trigger = {
			has_variable = kill_ballonist_var
		}

		kill_balloonist_effect = yes
	}

	option = {	#Get a new Balloonist immediately! (balloonist)
		name = academics.1.h
		default_option = yes

		trigger = {
			has_variable = kill_ballonist_var
		}

		kill_balloonist_effect = yes
		balloonist_cost_effect = yes
		create_balloonist_effect = yes
	}
}
