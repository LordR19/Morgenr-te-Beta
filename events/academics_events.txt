#Academics 1: Engineer retires
academics.1 = {
	type = country_event

	title = academics.1.t

	desc = academics.1.d

#	flavor = academics.1.f
	
	duration = 3

	gui_window = event_window_1char_tabloid

	left_icon = scope:academics_retired_person_scope

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		NOT = {
			has_variable = academics_engineer_emeritus_delayed_var
		}
		gaudi_has_engineer = yes
		var:gaudi_engineer_var = {
			age >= 65
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			var:gaudi_engineer_var = {
				age >= 70
			}
			factor = 10000
		}
	}

	immediate = {
		var:gaudi_engineer_var = {
			save_scope_as = academics_retired_person_scope
		}
	}

	option = {	#He earned his retirement
		name = academics.1.g
		default_option = yes

		kill_engineer_effect = yes

		if = {
			limit = { has_modifier = academics_engineer_emeritus_delayed_modifier }
			remove_modifier = academics_engineer_emeritus_delayed_modifier
		}
	}

	option = {	#We need him! Pay him extra!
		name = academics.1.h

		trigger = {
			NOT = {
				has_modifier = academics_engineer_emeritus_delayed_modifier
			}
		}

		add_modifier = {
			name = academics_engineer_emeritus_delayed_modifier
		}
		set_variable = {
			name = academics_engineer_emeritus_delayed_var
			years = { 1 5 }
		}

		ai_chance = {
			base = 0
		}
	}
}

#Academics 50: Academy Progress

academics.50 = {
	type = country_event

	hidden = yes

	trigger = {
		any_scope_building = { #No academy progress without academy
			is_building_type = building_university
		}
	}

	immediate = {
		set_variable = {
			name = academics_estimated_monthly_academy_progress_var
			value = 0
		}
		dubois_add_academy_progress = yes #basic progress
		change_variable = {
			name = academics_estimated_monthly_academy_progress_var
			add = 1
		}

		if = { #Literacy 25%
			limit = {
				literacy_rate >= 0.25
			}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Literacy 50%
			limit = {
				literacy_rate >= 0.5
			}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Literacy 75%
			limit = {
				literacy_rate >= 0.75
			}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Academics at least 0.5%
			limit = {
				pop_type_percent_country = {
					pop_type = academics
					percent >= 0.005
				}
			}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Academics at least 1%

			}

			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Academics at least 2%
			limit = {
				pop_type_percent_country = {
					pop_type = academics
					percent >= 0.02
				}
			}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Institution Level 2
			limit = {
            	institution_investment_level = {
                	institution	= institution_schools
                	value = 2
            	}
        	}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Institution Level 3
			limit = {
            	institution_investment_level = {
                	institution	= institution_schools
                	value = 3
            	}
        	}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Institution Level 4
			limit = {
            	institution_investment_level = {
                	institution	= institution_schools
                	value = 4
            	}
        	}
			dubois_add_academy_progress = yes
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 2
			}
		}
		if = { #Institution Level 5

                	value = 4
            	}

			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 2
			}
		}
		if = { #Intelligentsia is content
			limit = {
            	ig:ig_intelligentsia = {
					ig_approval >= 5
				}
        	}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}
		if = { #Special Tech School
			limit = {
				OR = {
					has_modifier = academics_avantgarde_intelligentsia_modifier
					has_modifier = academics_practical_academia_modifier
					has_modifier = academics_useful_ivory_tower_modifier
				}
        	}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}

		if = { #Observatory gives mathematical advantage

				OR = {
					has_variable = academics_promote_astronomer_active
					has_variable = academics_promote_engineer_active
				}

					is_building_type = building_verrier_observatory

			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}

		if = { #Zoo gives naturalistic advantage
			limit = {
				OR = {
					has_variable = academics_promote_naturalist_active
					has_variable = academics_promote_paleontologist_active
				}
				any_scope_building = {
					is_building_type = building_dubois_zoo
				}
        	}

			change_variable = {

					is_building_type = building_lepsius_archaeological_museum
				}
        	}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}

		if = { #Academy Investment
			limit = {
            	has_modifier = dubois_academy_investment_modifier
        	}
			dubois_add_academy_progress = yes
			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}
		}

			limit = {
            	has_modifier = lepsius_historical_materialistic_school_modifier
        	}

			change_variable = {

				add = 1
			}

        	}

			change_variable = {
				name = academics_estimated_monthly_academy_progress_var
				add = 1
			}

            	has_modifier = academics_academic_purge_modifier
        	}
			dubois_add_x_academy_progress = {
				AMOUNT = -1
			}
			change_variable = {

		}
		if = {
			limit = {
				scope:gaudi_new_engineer_scope = {
					has_modifier =  academics_is_mechanical_engineer_modifier
				}
			}
			gaudi_mechanical_engineering_medium_add_technology = yes
		}

			limit = {

					has_modifier =  academics_is_military_engineer_modifier
				}
			}
			gaudi_military_engineering_medium_add_technology = yes
		}

			limit = {
				scope:gaudi_new_engineer_scope = {
					has_modifier =  academics_is_electrical_engineer_modifier
				}
			}
			gaudi_electrical_engineering_medium_add_technology = yes
		}
		if = {
			limit = {
				scope:gaudi_new_engineer_scope = {

			base = 10

					has_variable = gaudi_engineer_var

						age <= 60
					}
				}
				add = 10000
			}
		}
	}

	after = {
		dubois_remove_academy_effect = yes
	}

academics.2116 = {
	type = country_event

	title = academics.2116.t

	desc = academics.2116.d

	flavor = academics.2116.f

	placement = ROOT

	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/lepsius_pictures/museum/lepsius_museum_world_cultures_collection.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	cooldown = {
		years = { 50 100 }
	}

	trigger = {
		academics_archaeological_mission_possible_trigger = yes #does not have an Archaeological mission
		lepsius_has_archaeological_museum = yes
		lepsius_has_archaeologist = yes
		var:lepsius_total_artifacts > 10
		any_country = {
			NOT = {
				this = root
			}
			var:lepsius_total_artifacts >= academics_has_10_or_more_total_artifacts
			var:lepsius_total_artifacts <= academics_has_50_or_less_total_artifacts_more
		}
		ig:ig_intelligentsia = {
			is_marginal = no
		}
	}

	immediate = {
		random_country = {
			limit = {
				NOT = {
					this = root
				}
				var:lepsius_total_artifacts >= academics_has_10_or_more_total_artifacts
				var:lepsius_total_artifacts <= academics_has_50_or_less_total_artifacts_more
			}
			save_scope_as = rival_scope
		}
		
		set_variable = {
			name = je_artifacts_rival_nation_var
			value = scope:rival_scope
		}
	}

	option = { #Let us beat them!
		name = academics.2116.g

		add_journal_entry = {
			type = je_academics_rivalry_for_artifacts
			target = scope:rival_scope
		}

		custom_tooltip = academics.2116_goal_tt
		custom_tooltip = academics.2116_reward_tt
	}

	option = { #I am an Archaeologist and I'm digging a hole!
		name = academics.2116.h
		default_option = yes

		add_modifier = {
			name = academics_unhappy_intelligentsia
			years = 2
		}

		remove_variable = je_artifacts_rival_nation_var
	}
}
