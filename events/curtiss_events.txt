#Curtiss 1: Airline status

curtiss.1 = {
	type = country_event
		
	hidden = yes

	trigger = {
		curtiss_has_airline = yes
	}

	immediate = {

		if = {
			limit = {
				var:curtiss_flight_route <= 2
				NOT = {
					has_modifier = curtiss_airline_level_1_modifier
				}
			}
			add_modifier = curtiss_airline_level_1_modifier
			remove_modifier = curtiss_airline_level_2_modifier
			remove_modifier = curtiss_airline_level_3_modifier
			remove_modifier = curtiss_airline_level_4_modifier
			remove_modifier = curtiss_airline_level_5_modifier
			remove_modifier = curtiss_airline_level_6_modifier
		}
		if = {
			limit = {
				var:curtiss_flight_route >= 3
				NOT = {
					has_modifier = curtiss_airline_level_2_modifier
					var:curtiss_flight_route >= 6
				}
			}
			add_modifier = curtiss_airline_level_2_modifier
			remove_modifier = curtiss_airline_level_1_modifier
			remove_modifier = curtiss_airline_level_3_modifier
			remove_modifier = curtiss_airline_level_4_modifier
			remove_modifier = curtiss_airline_level_5_modifier
			remove_modifier = curtiss_airline_level_6_modifier
		}
		if = {
			limit = {
				var:curtiss_flight_route >= 6
				NOT = {
					has_modifier = curtiss_airline_level_3_modifier
					var:curtiss_flight_route >= 9
				}
			}
			add_modifier = curtiss_airline_level_3_modifier
			remove_modifier = curtiss_airline_level_1_modifier
			remove_modifier = curtiss_airline_level_2_modifier
			remove_modifier = curtiss_airline_level_4_modifier
			remove_modifier = curtiss_airline_level_5_modifier
			remove_modifier = curtiss_airline_level_6_modifier
		}
		if = {
			limit = {
				var:curtiss_flight_route >= 9
				NOT = {
					has_modifier = curtiss_airline_level_4_modifier
					var:curtiss_flight_route >= 12
				}
			}
			add_modifier = curtiss_airline_level_4_modifier
			remove_modifier = curtiss_airline_level_1_modifier
			remove_modifier = curtiss_airline_level_2_modifier
			remove_modifier = curtiss_airline_level_3_modifier
			remove_modifier = curtiss_airline_level_5_modifier
			remove_modifier = curtiss_airline_level_6_modifier
		}
		if = {
			limit = {
				var:curtiss_flight_route >= 12
				NOT = {
					has_modifier = curtiss_airline_level_5_modifier
					var:curtiss_flight_route >= 15
				}
			}
			add_modifier = curtiss_airline_level_5_modifier
			remove_modifier = curtiss_airline_level_1_modifier
			remove_modifier = curtiss_airline_level_2_modifier
			remove_modifier = curtiss_airline_level_3_modifier
			remove_modifier = curtiss_airline_level_4_modifier
			remove_modifier = curtiss_airline_level_6_modifier

#Curtiss 10: Seaplanes [Schneider's Trophy JE]

curtiss.10 = {
	type = country_event

    title = curtiss.10.t
		
    desc = curtiss.10.d

	flavor = curtiss.10.f

	duration = 1
		
	event_image = { 
		texture = "gfx/event_pictures/curtiss_pictures/curtiss_schneider_foundation.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/airplane"

	icon = "gfx/interface/icons/event_icons/event_aviation.dds"

	trigger = {
		has_technology_researched = curtiss_seaplanes_tech
		country_rank >= rank_value:major_power
		any_scope_building = {
			is_building_type = building_port
			count >= 3
		}
		NOT = {
			has_global_variable = curtiss_seaplanes_invented
		}
	}

	immediate = {
		set_global_variable = curtiss_seaplanes_invented
		set_global_variable = {
			name = curtiss_seaplane_nations
			value = 0
		}
		set_variable = curtiss_has_seaplanes
	}

	option = { #A noble idea.
		name = curtiss.10.g
		add_journal_entry = {
			type = je_curtiss_found_schneider_trophy
		}
	}
}

curtiss.11 = {
	type = country_event
		
	hidden = yes

	trigger = {
		has_technology_researched = curtiss_seaplanes_tech
		has_global_variable = curtiss_seaplanes_invented
		NOT = {
			has_variable = curtiss_seaplanes_country_var
		}
	}

	immediate = {
		set_variable = curtiss_seaplanes_country_var
		if = {
			limit = {
				any_scope_building = {
					is_building_type = building_port
					count >= 3
				}
				country_rank >= rank_value:major_power
			}
			change_global_variable = {
				name = curtiss_seaplane_nations
				add = 1
			}
		}
	}
}

	type = country_event

    title = curtiss.12.t
		
    desc = curtiss.12.d

	flavor = curtiss.12.f

	duration = 1
		
	event_image = { 
		texture = "gfx/event_pictures/curtiss_pictures/curtiss_fabre.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/airplane"

	icon = "gfx/interface/icons/event_icons/event_aviation.dds"

	trigger = {
		game_date >= 1909.1.1
		NOT = {
			game_date >= 1914.1.1
		}
		NOT = {
			has_global_variable = curtiss_henri_fabre_happened
		}
		OR = {
			country_has_primary_culture = cu:french
			country_has_primary_culture = cu:occitan
			country_has_primary_culture = cu:north_italian
		}
		any_scope_state = {
			state_region = s:STATE_PROVENCE #Marseille
		}
		has_technology_researched = curtiss_early_aviation_tech
	}

	immediate = {
		set_global_variable = curtiss_henri_fabre_happened
	}

	option = { #I love this flying boat!
		name = curtiss.12.g
		default_option = yes
		if = {
			limit = {
				NOT = {
					has_technology_researched = military_aviation
				}
			} 
			add_technology_researched = military_aviation
		}
		if = {
			limit = {
				has_technology_researched = military_aviation
				NOT = {
					has_technology_researched = curtiss_seaplanes_tech
				}
			} 
			add_technology_researched = curtiss_seaplanes_tech
		}
		if = {
			limit = {
				has_technology_researched = curtiss_seaplanes_tech
				NOT = {
					has_technology_researched = curtiss_modern_aviation_tech
				}
			} 
			add_technology_progress = { 
				progress = 8000 technology = curtiss_modern_aviation_tech
			}
		}
	}
}

curtiss.23 = {
	type = country_event

	title = curtiss.23.t

	desc = curtiss.23.d

	flavor = curtiss.23.f
	
	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/curtiss_pictures/curtiss_airline_imperial_1.dds"
	}

	icon = "gfx/interface/icons/event_icons/event_aviation.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/airplane"

	trigger = {
		has_variable = curtiss_private_airline
		OR = { 
			has_law = law_type:law_cooperative_ownership
			has_law = law_type:law_command_economy
		}
	}

	immediate = {
		remove_variable = curtiss_private_airline
		set_variable = curtiss_state_airline
		if = {
			limit = {
				has_modifier = curtiss_spent_money_on_airline_company_modifier
			}
			remove_modifier = curtiss_spent_money_on_airline_company_modifier
		}
		if = {
			limit = {
				has_modifier = curtiss_corrupting_influences_in_politics_modifier
			}
