#Dubois 2: Origin of Species

dubois.2 = {
	type = country_event

	title = dubois.2.t

	desc = dubois.2.d

	flavor = dubois.2.f

	placement = ROOT

	gui_window = event_window_1char_tabloid

	left_icon = scope:dubois_naturalist_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_animals.dds"

	trigger = {
		has_technology_researched = dubois_theory_of_evolution_tech
		NOT = {
			has_global_variable = dubois_theory_of_evolution
		}
	}

	immediate = {
		set_global_variable = dubois_theory_of_evolution
		if = {
			limit ={
				NOT = {
					has_variable = dubois_naturalist_var
				}
			}
			create_naturalist_effect = yes
		}
		var:dubois_naturalist_var = {
			save_scope_as = dubois_naturalist_scope
		}
	}

	option = {
		name = dubois.2.g
		default_option = yes
		add_modifier = {
			name = dubois_theory_of_evolution_invented_modifier
			years = 5
		}
		dubois_large_add_technology = yes

		ai_chance = {
			base = 80
		}
	}

	option = {
		name = dubois.2.h
		add_modifier = {
			name = dubois_theory_of_evolution_invented_modifier
			years = 5
		}
		custom_tooltip = {
			text = state_atheism_conversion_tt
			every_scope_state = {
				limit = {
					is_incorporated = yes
				}
				convert_population = {
					target = rel:atheist
					value = 0.05
				}
			}
		}

		ai_chance = {
			base = 20
		}
	}

	after = {
	}
}

#Dubois 3: Return of the Beagle

dubois.3 = {
	type = country_event

	title = dubois.3.t

	desc = dubois.3.d

	flavor = dubois.3.f

	placement = ROOT

	gui_window = event_window_1char_tabloid

	left_icon = scope:dubois_naturalist_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_animals.dds"

	trigger = {
		NOT = {
			game_date >= 1837.1.1
		}
		ROOT = c:GBR
		NOT = {
			has_variable = dubois_beagle_returned
		}
	}

	immediate = {
		set_variable = dubois_beagle_returned
		create_darwin_effect = yes
		var:dubois_naturalist_var = {
			save_scope_as = dubois_naturalist_scope
		}
	}

	option = {
		name = dubois.3.g
		default_option = yes
		dubois_add_5000_specimen = yes
		dubois_add_3_fossil = yes
		dubois_medium_add_technology = yes
		remove_modifier = rapanui_pacific_expedition_modifier
		add_modifier = {
			name = rapanui_had_pacific_expedition_modifier
			years = 5
		}
	}

	after = {
	}
}

#Dubois 6: Dubois finds the Java Man (alternative to Rapanui.250)

dubois.6 = {
	type = country_event

	title = dubois.6.t
	desc = dubois.6.d
	flavor = dubois.6.f

	placement = scope:dubois_java_man_location

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/dubois_pictures/dubois_fossil_java.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fossil.dds"

	trigger = {
		NOT = {
			has_global_variable = dubois_java_man_found
		}
		game_date >= 1891.1.1
		has_technology_researched = dubois_paleontology_tech
		exists = c:NET
		ROOT = c:NET
	}

	immediate = {
		set_global_variable = dubois_java_man_found
		random_state = {
			limit = {
				state_region = s:STATE_EAST_JAVA
			}
			save_scope_as = dubois_java_man_location
		}

		create_dubois_eugene_effect = yes
	}

	option = {
		name = rapanui.250.g #A historical day
		default_option = yes
		dubois_add_fossil_java = yes
		dubois_medium_add_technology = yes
		post_notification = dubois_java_man_notification

		hidden_effect = {
			dubois_remove_new_paleontologist_effect = yes
		}
	}

	option = {
		name = dubois.6.g #We need this man as our lead palaeontologist!
		dubois_add_fossil_java = yes
		post_notification = dubois_java_man_notification

		dubois_set_new_paleontologist_as_paleontologist_effect = yes

		ai_chance = {
			base = 0
		}
	}

	after = {
	}
}

#Dubois 81: Yearly paleontological report

dubois.81 = {
	type = country_event
		
	hidden = yes

	trigger = {
		NOT = {
			has_variable = dubois_paleontological_report_cooldown_var
		}
		dubois_paleontological_research = yes
	}

	immediate = {
		set_variable = {
			name = dubois_paleontological_report_cooldown_var
			days = 365
		}
		dubois_add_1_fossil = yes #base fossils

		#Additional yearly fossils
		if = {
			limit = {
				has_technology_researched = dubois_paleontology_tech
			}
			dubois_add_1_fossil = yes
		}
		if = {
			limit = {
				any_scope_building = {
					is_building_type = building_university
					has_active_production_method = pm_dubois_university_natural_history
				}
			}
			dubois_add_1_fossil = yes
		}
		if = {
			limit = {
				has_variable = dubois_yearly_fossils_var
			}
			set_variable = {
				name = dubois_fossils_notification_var
				value = var:dubois_yearly_fossils_var
			}
			set_variable = {
				name = dubois_yearly_fossils_var
				value = 0
			}
		}
		if = {
			limit = {
				dubois_paleontological_research = yes #Notification
			}
			post_notification = dubois_paleontological_report_notification
		}
	}
}
