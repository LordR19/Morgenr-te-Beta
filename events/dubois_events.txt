#Dubois 2: Origin of Species

dubois.2 = {
	type = country_event

	title = dubois.2.t

	desc = dubois.2.d

	flavor = dubois.2.f

	placement = ROOT

	gui_window = event_window_1char_tabloid

	left_icon = scope:dubois_naturalist_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_animals.dds"

	trigger = {
		NOT = {
			has_global_variable = dubois_theory_of_evolution
		}
	}

	immediate = {
		set_global_variable = dubois_theory_of_evolution
		if = {
			limit ={
				NOT = {
					has_variable = dubois_naturalist_var
				}
			}
			create_naturalist_effect = yes
		}
		var:dubois_naturalist_var = {
			save_scope_as = dubois_naturalist_scope
		}
	}

	option = {
		name = dubois.2.g
		default_option = yes
		add_modifier = {
			name = dubois_theory_of_evolution_invented_modifier
			years = 5
		}
		dubois_large_add_technology = yes

		ai_chance = {
			base = 80
		}
	}

	option = {
		name = dubois.2.h
		add_modifier = {
			name = dubois_theory_of_evolution_invented_modifier
			years = 5
		}
		custom_tooltip = {
			text = state_atheism_conversion_tt
			every_scope_state = {
				limit = {
					is_incorporated = yes
				}
				convert_population = {
					target = rel:atheist
					value = 0.05
				}
			}
		}

		ai_chance = {
			base = 20
		}
	}

	after = {
	}
}

#Dubois 3: Return of the Beagle

	type = country_event

		NOT = {
			game_date >= 1837.1.1

			save_scope_as = dubois_naturalist_scope

		name = dubois.3.g
		default_option = yes
		dubois_add_5000_specimen = yes
		dubois_medium_add_technology = yes
	}

	after = {
	}
}
