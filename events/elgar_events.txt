#Elgar 1: Yearly music tradition
elgar.1 = {
	dlc = dlc_morgenrote

	type = country_event
		
	hidden = yes

	trigger = {
		NOT = {
			has_variable = elgar_music_tradition_cooldown_var
		}
		has_technology_researched = elgar_classicism_tech
	}

	immediate = {
		set_variable = {
			name = elgar_music_tradition_cooldown_var
			days = 365
		}

		elgar_add_x_music_tradition_unscaled_effect = {
			AMOUNT = elgar_yearly_musical_tradition_value
		}
		#######################################################################
		if = {
			limit = {
				has_modifier = elgar_patron_of_arts_modifier
				NOT = {
					has_law = law_type:law_traditionalism
				}
			}
			remove_modifier = elgar_patron_of_arts_modifier
		}
	}
}

#Elgar 10: Visit the Opera
elgar.10 = {
	dlc = dlc_morgenrote

	type = country_event

    title = elgar.10.t
		
    desc = elgar.10.d

	flavor = elgar.10.f

	duration = 3
		
	event_image = { 
		texture = "gfx/event_pictures/elgar_pictures/elgar_opera.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_music.dds"

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_capital = yes
			}
			save_scope_as = elgar_philharmonics_scope
		}
		random_country = {
			limit = {
				elgar_has_composer = yes
			}
			save_scope_as = elgar_foreign_country_scope
			var:elgar_composer_var = {
				save_scope_as = elgar_foreign_composer_scope
			}
		}

		#Sets up a variable that sets the instrument of the Concerto
		set_variable = {
			name = elgar_concerto_instrument_randomizer_var
			value = { 1 15 }
		}
		round_variable = {
			name = elgar_concerto_instrument_randomizer_var
			nearest = 1
		}
		#Sets up a variable that sets the chord of the Concerto
		set_variable = {
			name = elgar_music_chord_randomizer_var
			value = { 1 7 }
		}
		round_variable = {
			name = elgar_music_chord_randomizer_var
			nearest = 1
		}
		#Sets up a variable that sets Major/Minor
		set_variable = {
			name = elgar_music_major_minor_randomizer_var
			value = { 1 2 }
		}
		round_variable = {
			name = elgar_music_major_minor_randomizer_var
			nearest = 1
		}
	}

	option = { #Bye
		name = elgar.10.g
		default_option = yes
		remove_variable = elgar_visiting_opera_var
	}

	after = {
		remove_variable = elgar_concerto_instrument_randomizer_var
		remove_variable = elgar_music_chord_randomizer_var
		remove_variable = elgar_music_major_minor_randomizer_var
	}
}

elgar.30 = {
	dlc = dlc_morgenrote

	type = country_event

	title = elgar.30.t

	desc = elgar.30.d

	flavor = {
		triggered_desc = {
			trigger = {
				NOR = {
					has_modifier = elgar_composer_on_conducting_modifier

					has_modifier = elgar_composer_on_tour_modifier
					has_modifier = elgar_composer_on_masterpiece_modifier
					has_modifier = elgar_conductor_exhausted_modifier
				}
			}
			desc = elgar.30.free
		}
		triggered_desc = {
			trigger = {
				has_modifier = elgar_conductor_exhausted_modifier
			}
			desc = elgar.30.exhausted
		}
		triggered_desc = {
			trigger = {
				has_modifier = elgar_composer_on_conducting_modifier
			}
			desc = elgar.30.conducting
		}
		triggered_desc = {
			trigger = {
				has_modifier = elgar_composer_on_composing_modifier
			}
			desc = elgar.30.composing
		}
		triggered_desc = {
			trigger = {
				has_modifier = elgar_composer_on_tour_modifier
			}
			desc = elgar.30.touring
		}
		triggered_desc = {
			trigger = {
				has_modifier = elgar_composer_on_masterpiece_modifier
			}
			desc = elgar.30.masterpiece
		}
	}

	gui_window = event_window_1char_tabloid

	left_icon = scope:elgar_conductor

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_music.dds"

	immediate = {
		set_elgar_conductor_scope_effect = yes
	}

	#Concentrate on composing
	option = {
		name = elgar.30.h
		default_option = yes
		trigger = {
			elgar_has_active_composer = yes
			elgar_composer_not_exhausted = yes
		}
		if = {
			limit = { has_variable = elgar_composer_var }
			var:elgar_composer_var = {
				set_character_busy_and_immortal = yes
				add_modifier = {
					name = elgar_conductor_composes_modifier
				}
			}
		}
		else = {
			custom_tooltip = elgar_composer_becomes_busy_tt
		}
		add_modifier = {
			name = elgar_composer_on_composing_modifier
			multiplier = money_amount_multiplier_very_small
		}
		set_elgar_composer_progress = yes
		add_journal_entry = {
			type = je_elgar_composer_composing
		}
		remove_variable = elgar_visiting_opera_var

		ai_chance = {
			base = 5
			modifier = { #As a bgeinne,r compose as much as possible
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_num_compositions_var < 25
					}
				}
				add = 10
			}
			modifier = { #Don't compose too much
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_num_compositions_var >= 35
					}
				}
				add = -1
			}
			modifier = { #Don't compose too much
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_num_compositions_var >= 45
					}
				}
				add = -1
			}
			modifier = { #Don't compose too much
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_num_compositions_var >= 50
					}
				}
				add = -1
			}
			modifier = {
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_experience >= 450
						var:elgar_composer_num_masterworks_var < 3
					}
				}
				add = -1000 #Should not compose when able to do masterpiece
			}
		}
	}

	#Conduct the next seasons
	option = {
		name = elgar.30.g
		trigger = {
			elgar_has_active_composer = yes
			elgar_composer_not_exhausted = yes
			elgar_can_composer_conduct_trigger = yes
			elgar_has_opera_size_x_tooltip_trigger = { LEVEL = 1 }
		}
		if = {
			limit = { has_variable = elgar_composer_var }
			var:elgar_composer_var = {
				set_character_busy_and_immortal = yes
				add_modifier = {
					name = elgar_conductor_conducts_modifier
				}
			}
		}
		else = {
			custom_tooltip = elgar_composer_becomes_busy_tt
		}
		random_scope_state = {
			limit = {
				has_building = building_elgar_opera
				NOT = {
					has_modifier = elgar_music_production_modifier
				}
			}
			add_modifier = {
				name = elgar_music_production_modifier
				years = 5
			}
		}
		add_modifier = {
			name = elgar_composer_on_conducting_modifier
			multiplier = money_amount_multiplier_very_small
		}
		set_elgar_composer_progress = yes
		add_journal_entry = {
			type = je_elgar_composer_conducting
		}
		remove_variable = elgar_visiting_opera_var

		ai_chance = {
			base = 5
			modifier = { #Should try to get trait for concert tour
				trigger = {
					scope:elgar_conductor = {
						NOR = {
							has_trait = elgar_conductor
							has_trait = elgar_virtuoso
						}
					}
				}
				add = 2
			}
			modifier = { #A reserved character can not go on tour, so the composer should try to compose
				trigger = {
					scope:elgar_conductor = {
						has_trait = reserved
					}
				}
				add = -5
			}
			modifier = { #A reserved character can not go on tour, however, it is the only way to improve.
				trigger = {
					scope:elgar_conductor = {
						has_trait = reserved
						var:elgar_composer_experience < 400
						age < 40
					}
				}
				add = 2
			}
			modifier = {
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_experience >= 450
						var:elgar_composer_num_masterworks_var < 3
					}
				}
				add = -1000 #Should not conduct when able to do masterpiece
			}
		}
	}

	#Do a concert tour
	option = {
		name = elgar.30.i
		trigger = {
			elgar_has_active_composer = yes
			elgar_composer_not_exhausted = yes
			elgar_can_composer_tour_trigger = yes
			elgar_composer_experience_100_or_more_trigger = yes
			elgar_composer_virtuoso_or_conductor_trigger = yes
			trigger_if = {
				limit = {
					has_variable = elgar_composer_var
				}
				custom_tooltip = {
					text = MR_COMPOSER_YOUNGER_THAN_65_TRIGGER_TT
					var:elgar_composer_var.age < 65
				}
			}
			trigger_else = {
				hidden_trigger = { always = no }
			}
			trigger_if = {
				limit = {
					has_variable = elgar_composer_var
				}
				custom_tooltip = {
					text = MR_COMPOSER_NOT_RESERVED_TRIGGER_TT
					NOT = {
						var:elgar_composer_var = {
							has_trait = reserved
						}
					}
				}
			}
			trigger_else = {
				hidden_trigger = { always = no }
			}
		}
		if = {
			limit = { has_variable = elgar_composer_var }
			var:elgar_composer_var = {
				set_character_busy_and_immortal = yes
				add_modifier = {
					name = elgar_conductor_touring_modifier
				}
			}
		}
		else = {
			custom_tooltip = elgar_composer_becomes_busy_tt
		}
		add_modifier = {
			name = elgar_composer_on_tour_modifier
		}
		set_elgar_composer_progress = yes
		add_journal_entry = {
			type = je_elgar_composer_touring
		}
		hidden_effect = {
			trigger_event = {
				id = elgar.451
			}
		}
		remove_variable = elgar_visiting_opera_var

		ai_chance = {
			base = 3 #Should prefer tour when possible
			modifier = {
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_experience >= 450
					}
				}
				add = -1000 #Should never go on tour when having 450 reputation. From 450 on, the composer should do masterpieces. After that, he will not need to improve anymore.
			}
			modifier = {
				trigger = {
					scope:elgar_conductor = {
						var:elgar_composer_num_compositions_var <= 25
					}
				}
				add = -1000 #Should never go on tour when having less than 25 compositions
			}
		}
	}

		}
		set_elgar_composer_progress = yes
		add_journal_entry = {
			type = je_elgar_composer_masterpiece
		}
		remove_variable = elgar_visiting_opera_var

		ai_chance = {
			base = 0 #Shouldn't do this if the composer is bad
			modifier = {
