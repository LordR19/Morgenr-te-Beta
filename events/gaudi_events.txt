	type = country_event

    title = gaudi.11.t
		
    desc = gaudi.11.d

		texture = "gfx/event_pictures/elgar_pictures/elgar_art_nouveau.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_music.dds"

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				has_modifier = gaudi_first_secession_state_modifier
			}
			set_variable = gaudi_first_secession_state_var
			save_scope_as = gaudi_secession_state
		}
	}

	option = { #We are a leading cultural nation
		name = gaudi.11.g
		add_modifier = {
			name = gaudi_first_secession_modifier
		}
	}
}

#Gaudí 60: Burning the Summer Palace

gaudi.60 = {
	type = country_event

    title = gaudi.60.t
		
    desc = gaudi.60.d

	flavor = gaudi.60.f

	duration = 1

	placement = scope:gaudi_summer_palace_state
		
	event_image = { 
		texture = "gfx/event_pictures/gaudi_pictures/gaudi_summer_burning.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_BEIJING
			}
			save_scope_as = gaudi_summer_palace_state
		}
	}

	option = { #Burn it down!
		name = gaudi.60.g
		default_option = yes
		custom_tooltip = gaudi.60.tt
		add_treasury = 100000
		lepsius_add_5_fareastern_artifact = yes
		lepsius_add_1_major_fareastern_artifact = yes
		change_infamy = 10
		set_variable = lepsius_major_artifact_summer_found_var
		set_global_variable = gaudi_summer_palace_looted
		c:CHI = {
			change_relations = {
				country = ROOT 
				value = -25
			}
			trigger_event = {
				id = gaudi.61
				popup = yes
			}
		}
		post_notification = gaudi_summer_palace_burned_notification
	}

	option = { #Only barbarians would think of that.
		name = gaudi.60.h
		change_infamy = -2

		ai_chance = {
			base = 0
		}
	}
}

gaudi.100 = { #This will trigger in 1823 in a 1815/1816 mod
	type = country_event

	title = gaudi.100.t
	desc = gaudi.100.d
	flavor = gaudi.100.f

	placement = scope:gaudi_dom_state

	event_image = { 
		texture = "gfx/event_pictures/gaudi_pictures/gaudi_dombau_begin.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_architecture.dds"

	duration = 1

	trigger = {
		NOT = {
			has_global_variable = gaudi_dombau_verein
		}
		game_date >= 1823.1.1
		NOT = {
			game_date >= 1842.1.1
		}
		any_scope_state = {
			state_region = s:STATE_NORTH_RHINE
		}
		has_technology_researched = romanticism
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_NORTH_RHINE
			}
			save_scope_as = gaudi_dom_state
		}
		set_global_variable = gaudi_dombau_verein
	}

	option = {
		name = gaudi.100.g
        default_option = yes
	}
}

#Gaudí 648: Archaeological investigations

gaudi.648 = {
	type = country_event

	title = gaudi.648.t
	desc = gaudi.648.d
	flavor = gaudi.648.f

	placement = scope:gaudi_sagrada_state

	gui_window = event_window_1char_tabloid

	left_icon = scope:gaudi_archaeologist

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_architecture.dds"

	duration = 1

	trigger = {
		NOT = {
			has_variable = gaudi_excavation_barcelona
		}
		any_scope_character = {
			OR = {
				has_trait = lepsius_archaeologist
				has_trait = lepsius_archaeologist_experienced
			}
			is_busy = no
		}
		global_var:gaudi_sagrada_progress >= 12
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_CATALONIA
			}
			save_scope_as = gaudi_sagrada_state
		}
		random_scope_character = {
			limit = {
				OR = {
					has_trait = lepsius_archaeologist
					has_trait = lepsius_archaeologist_experienced
				}
				is_busy = no
			}
			save_scope_as = gaudi_archaeologist
		}
		set_variable = gaudi_excavation_barcelona
	}

	option = {
		name = gaudi.648.g #We cannot afford such a long halt to construction.
        default_option = yes
		gaudi_add_1_sagrada_progress = yes
	}

	option = {
		name = gaudi.648.h #A unique archaeological opportunity.
		gaudi_sub_12_sagrada_progress = yes
		lepsius_archaeologist_experience_add = yes
	}

	after = {
		scope:gaudi_archaeologist = {
			if = {
				limit = {
					var:lepsius_archaeologist_experience >= 3
					NOT = {
						var:lepsius_archaeologist_experience >= 5
						has_trait = lepsius_archaeologist_experienced
						has_trait = lepsius_archaeologist_famous
					}
				}
				remove_trait = lepsius_archaeologist
				add_trait = lepsius_archaeologist_experienced
			}
			if = {
				limit = {
					var:lepsius_archaeologist_experience >= 5
					NOT = {
						has_trait = lepsius_archaeologist_famous
					}
				}
				remove_trait = lepsius_archaeologist_experienced
				add_trait = lepsius_archaeologist_famous
			}
		}
	}
}
