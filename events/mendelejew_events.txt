#Mendelejew 111: An Exemplary Professor
mendelejew.111 = {
	dlc = dlc_morgenrote

	type = country_event

	placement = scope:mendelejew_university_state

	title = mendelejew.111.t

	desc = mendelejew.111.d

	flavor = mendelejew.111.f
	
	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/mendelejew_pictures/mendelejew_element_theory.dds"
	}

	icon = "gfx/interface/icons/event_icons/event_chemistry.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	cooldown = {
		years = 10
	}

	trigger = {
		var:mendelejew_chemist_theory_progress_var >= 6
		any_scope_state = {
			is_production_method_active = {
				building_type = building_university
				production_method = pm_dubois_university_mathematics
			}
		}
	}

	immediate = {
		random_scope_building = {
			limit = {
				is_building_type = building_university
				has_active_production_method = pm_dubois_university_mathematics
			}
			save_scope_as = mendelejew_university_building
			state = {
				save_scope_as = mendelejew_university_state
			}
		}
	}

	option = {	#DOCTOR should concentrate on research
		name = mendelejew.111.g
		mendelejew_add_x_chemist_progress_effect = { AMOUNT = 12 }
		scope:mendelejew_university_building = {
			add_modifier = {
				name = mendelejew_absent_chemist_modifier 
				years = 3
			}
		}
	}

	option = {	#DOCTOR should concentrate on lectureship
		name = mendelejew.111.h
		dubois_add_x_academy_progress = { AMOUNT = 9 }
		mendelejew_add_x_chemist_progress_effect = { AMOUNT = -6 }
	}

	option = {	#DOCTOR should find a balance
		name = mendelejew.111.i
		default_option = yes
		mendelejew_add_x_chemist_progress_effect = { AMOUNT = -3 }
	}

	after = {
	}
}

#Mendelejew 313: Neglected Duties
mendelejew.313 = {
	dlc = dlc_morgenrote

	type = country_event

	placement = scope:mendelejew_university_state

	title = mendelejew.313.t

	desc = mendelejew.313.d

	flavor = mendelejew.313.f
	
	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/mendelejew_pictures/mendelejew_element_isolation.dds"
	}

	icon = "gfx/interface/icons/event_icons/event_chemistry.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	cooldown = {
		years = 15
	}

	trigger = {
		var:mendelejew_chemist_theory_progress_var >= 12
		any_scope_state = {
			is_production_method_active = {
				building_type = building_university
				production_method = pm_dubois_university_mathematics
			}
		}
	}

	immediate = {
		random_scope_building = {
			limit = {
				is_building_type = building_university
				has_active_production_method = pm_dubois_university_mathematics
			}
			save_scope_as = mendelejew_university_building
			state = {
				save_scope_as = mendelejew_university_state
			}
		}
	}

	option = {	#Dr. Chemist has more important things to do!
		name = mendelejew.313.g
		mendelejew_small_add_technology = yes
		dubois_add_x_academy_progress = { AMOUNT = -12 }
		scope:mendelejew_university_building = {
			add_modifier = {
				name = mendelejew_absent_chemist_modifier
				years = 3
			}
		}
	}

	option = {	#Dr. Chemist must fulfill his other obligations.
		name = mendelejew.313.h
		default_option = yes
		mendelejew_add_x_chemist_progress_effect = { AMOUNT = -12 }
	}

	after = {
	}
}
