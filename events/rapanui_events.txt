#Rapanui 101: Mission to Oceania

rapanui.101 = {
	type = country_event

	title = {
		triggered_desc = {
			trigger = {
				country_has_state_religion = rel:catholic
			}
			desc = rapanui.101.t_cat
		}
		triggered_desc = {
			trigger = {
				country_has_state_religion = rel:protestant
			}
			desc = rapanui.101.t_pro
		}
	}

	desc = {
		triggered_desc = {
			trigger = {
				country_has_state_religion = rel:catholic
			}
			desc = rapanui.101.d_cat
		}
		triggered_desc = {
			trigger = {
				country_has_state_religion = rel:protestant
			}
			desc = rapanui.101.d_pro
		}
	}

	flavor = rapanui.101.f

	duration = 3

	gui_window = event_window_1char_tabloid

	left_icon = scope:rapanui_missionary_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_mission.dds"

	trigger = {
	}

	immediate = {
		create_missionary_effect = yes
		var:rapanui_missionary_var = {
			save_scope_as = rapanui_missionary_scope
		}
		random_country = {
			limit = {
				OR = {
					has_diplomatic_pact = {
						who = ROOT
						type = catholic_mission_to_oceania
					}
					has_diplomatic_pact = {
						who = ROOT
						type = protestant_mission_to_oceania
					}
				}
			}
			save_scope_as = rapanui_mission_destination_country
		}
	}

	option = {
		name = rapanui.101.g_cat
		trigger = {
			country_has_state_religion = rel:catholic
		}
	}

	option = {
		name = rapanui.101.g_pro
		trigger = {
			country_has_state_religion = rel:protestant
		}
	}

	after = {
		if = {
			limit = {
				country_has_state_religion = rel:catholic
			}
			scope:rapanui_mission_destination_country = {
				trigger_event = {
					id = rapanui.102
					days = { 3 9 }
				}
			}
		}
		if = {
			limit = {
				country_has_state_religion = rel:protestant
			}
			scope:rapanui_mission_destination_country = {
				trigger_event = {
					id = rapanui.103
					days = { 3 9 }
				}
			}
		}
	}
}

			type = catholic_mission_to_oceania

	}

		name = rapanui.104.g_pro

			country_has_state_religion = rel:protestant
		}
		remove_diplomatic_pact = {
			country = scope:rapanui_mission_destination_country
			type = protestant_mission_to_oceania
		}
		remove_modifier = rapanui_mission_to_oceania_modifier

	flavor = rapanui.1001.f

	trigger = {
		NOT = {
			has_variable = rapanui_moai_search
		}
		NAND = {
			has_global_variable = rapanui_hava_found
			has_global_variable = rapanui_hoa_found
		}
		exists = c:RAP
		lepsius_has_archaeological_museum = yes
		has_technology_researched = lepsius_early_archaeology_tech
		has_interest_marker_in_region = region_oceania
	}

	immediate = {
		create_adventurer_effect = yes
		var:rapanui_adventurer_var = {
			save_scope_as = rapanui_adventurer_scope
		}
	}

	option = {
		name = rapanui.1001.g #What a mysterious place...
	}

	after = {
		set_variable = {
			name = rapanui_moai_search
			years = 20
		}
		trigger_event = {
			id = rapanui.1002
			days = { 3 12 }
		}
	}
}
