#Yearly astronomical report

verrier.1 = {
	dlc = dlc_morgenrote

	type = country_event
		
	hidden = yes

	trigger = {
		NOT = {
			has_variable = verrier_astronomical_objects_cooldown_var
		}
		verrier_astronomical_observation = yes
	}

	immediate = {
		set_variable = {
			name = verrier_astronomical_objects_cooldown_var
			days = 365
		}
		#Adds the yearly astronomical objects.
		# See script value for calculation.
		verrier_add_x_astronomical_objects_effect = {
			AMOUNT = verrier_yearly_astronomical_objects_value
		}
		###Yearly Notification###
		if = {
			limit = {
				has_variable = verrier_yearly_astronomical_objects_var
			}
			set_variable = {
				name = verrier_astronomical_objects_notification_var
				value = var:verrier_yearly_astronomical_objects_var
			}
			set_variable = {
				name = verrier_yearly_astronomical_objects_var
				value = 0
			}
		}
		if = {
			limit = {
				verrier_astronomical_observation = yes #Notification
			}
			post_notification = verrier_astronomical_observation_notification
		}
	}
}

#Verrier 2: Astronomical Prestige Set up event
verrier.2 = {
	hidden = yes

	immediate = {
		while = {
			limit = {
				has_modifier = verrier_astronomical_prestige_modifier
			}
			remove_modifier = verrier_astronomical_prestige_modifier
		}
		#Adds the Astronomical Prestige Modifier if the country has any astronomical prestige
		add_modifier = {
			name = verrier_astronomical_prestige_modifier
			multiplier = verrier_astronomical_prestige
		}
	}
}

verrier.6 = {
	dlc = dlc_morgenrote

	type = country_event

	title = verrier.6.t
	desc = verrier.6.d
	flavor = verrier.6.f

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_comet_1881.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_astronomy.dds"

	trigger = {
		any_scope_state = {
			is_capital = no
			has_building = building_verrier_observatory
		}
	}

	immediate = {
	}

	option = {
		name = verrier.6.g
		default_option = yes

		trigger = {
			any_scope_state = {
				is_capital = yes
				has_building = building_verrier_observatory
			}
		}
		verrier_add_x_astronomical_objects_effect = {
			AMOUNT = 3
		}
		verrier_small_add_technology = yes
		every_scope_state = {
    		limit = { 
				has_building = building_verrier_observatory
				is_capital = no
			}
    		remove_building = building_verrier_observatory
  		}
	}

	option = {
		name = verrier.6.h

		trigger = {
			any_scope_state = {
				is_capital = yes
				NOT = {
					has_building = building_verrier_observatory
				}
			}
		}
		every_scope_state = {
    		limit = { 
				is_capital = yes
			}
    		create_building={
				building="building_verrier_observatory"
				level=1
				reserves=1
			}
			activate_building = {
				building = "building_verrier_observatory"
			}
  		}
		if = {
			limit = {
				any_scope_state = {
					is_production_method_active = {
						building_type = building_verrier_observatory
						production_method = pm_gaudi_magnetic_observatory
					}
				}
			}
			capital = {
				activate_production_method = {
					building_type = building_verrier_observatory
					production_method = pm_gaudi_magnetic_observatory
				}
			}
		}
		every_scope_state = {
    		limit = { 
				has_building = building_verrier_observatory
				is_capital = no
			}
    		remove_building = building_verrier_observatory
  		}
	}
}

#Verrier 10: Neptune discovered

verrier.10 = {
	dlc = dlc_morgenrote

	type = country_event

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_neptune.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_astronomy.dds"

	immediate = {
		set_global_variable = verrier_neptune_discovered_var

		set_global_variable = {
			name = verrier_neptune_year_global_var
			value = year
		}
		country_definition = {
			set_global_variable = {
				name = verrier_discoverer_neptune_global_var
				value = this
			}
		}
		set_variable = verrier_neptun_achievement
		if = {
			limit = {
				has_variable = verrier_astronomer_var
			}
			var:verrier_astronomer_var = {
				save_scope_as = verrier_astronomer_scope
			}
		}
		else = {
			create_new_astronomer_effect = yes
			verrier_set_new_astronomer_as_astronomer_effect = yes
			var:verrier_astronomer_var = {
				save_scope_as = verrier_astronomer_scope
			}
		}
	}

	title = verrier.10.t
	desc = verrier.10.d
	flavor = verrier.10.f

	duration = 3

	option = {
		name = verrier.10.g
		custom_tooltip = {
			text = verrier_discovered_neptune_tt
			set_variable = verrier_discovered_neptune_variable
		}
		scope:verrier_astronomer_scope = {
			add_modifier = {
				name = verrier_neptune_discoverer_modifier
			}
			set_variable = verrier_neptune_discoverer_var
		}
		change_global_variable = {
 			name = verrier_planets_number
 			add = 1
		}
		astronomer_experience_gain_effect = {
			AMOUNT = 100
		}
	}

	after = {
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { this = ROOT }
					is_player = yes
				}
				trigger_event = {
					id = verrier.11
					days = 3
				}
			}
		}
		random_list = {
			1 = {}
			1 = {
				var:verrier_astronomer_var = {
					save_scope_as = manzoni_scientist_scope
				}
				trigger_event = {
					id = manzoni.13
					days = { 7 21 }
				}
			}
		}
		verrier_update_astronomical_prestige_effect = yes
	}
}

#Verrier 9010: The Sound of Science
### Original Event by Crownsteler ###
verrier.9010 = {
	dlc = dlc_morgenrote

	type = country_event
	placement = scope:sound_of_science_state

	title = verrier.9010.t
	desc = verrier.9010.d
	flavor = verrier.9010.f

	event_image = {
		video = "gfx/event_pictures/unspecific_trains.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		NOT = {
			has_global_variable = verrier_sound_of_science_global_var
		}
		year >= 1842
		any_state = {
			owner = ROOT
			is_incorporated = yes
			has_building = building_railway
			any_scope_building = {
				is_building_type = building_university
				level >= 2
			}
		}
	}

	immediate = {
		set_global_variable = verrier_sound_of_science_global_var
		random_scope_state = {
			limit = {
				is_incorporated = yes
				has_building = building_railway
				any_scope_building = {
					is_building_type = building_university
					level >= 2
				}
			}
			save_scope_as = sound_of_science_state
		}
	}

	option = {
		name = verrier.9010.g
		if = {
			limit = {
				NOT = { has_technology_researched = empiricism }
			}
			add_technology_progress = { progress = 3750 technology = empiricism }
		}
		else_if = {
			limit = {
				can_research = dialectics
			}
			add_technology_progress = { progress = 1500 technology = dialectics }
		}
		else_if = {
			limit = {
				can_research = telephone
			}
			add_technology_progress = { progress = 2250 technology = telephone }
		}
		else_if = {
			limit = {
				can_research = radio
			}
			add_technology_progress = { progress = 3750 technology = radio }
		}
		else = {
			add_modifier = {
				name = verrier_sound_of_science_prestige_modifier
				months = normal_modifier_time	
			}
		}
	}

	option = {
		name = verrier.9010.h
		default_option = yes
		add_modifier = {
			name = verrier_sound_of_science_tech_spread_modifier
			months = short_modifier_time	
		}
	}
}
