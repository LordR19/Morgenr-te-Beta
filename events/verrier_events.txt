#Yearly astronomical report

verrier.1 = {
	dlc = dlc_morgenrote

	type = country_event
		
	hidden = yes

	trigger = {
		NOT = {
			has_variable = verrier_astronomical_objects_cooldown_var
		}
		verrier_astronomical_observation = yes
	}

	immediate = {
		set_variable = {
			name = verrier_astronomical_objects_cooldown_var
			days = 365
		}
		verrier_add_x_astronomical_objects_effect = {
			AMOUNT = 1
		} #base objects

		if = {
			limit = {
				has_technology_researched = camera
			}
			verrier_add_x_astronomical_objects_effect = {
				AMOUNT = 2
			}
		}
		if = {
			limit = {
				has_technology_researched = verrier_electromagnetism_tech
			}
			verrier_add_x_astronomical_objects_effect = {
				AMOUNT = 3
			}
		}
		if = {
			limit = {
				has_technology_researched = verrier_astronomy_tech
			}
			verrier_add_x_astronomical_objects_effect = {
				AMOUNT = 4
			}
		}
		if = {
			limit = {
				has_technology_researched = verrier_astro_physics_tech
			}
			verrier_add_x_astronomical_objects_effect = {
				AMOUNT = 10
			}
		}
		if = {
			limit = {
				has_variable = verrier_astronomer_var
				var:verrier_astronomer_var = {
					has_trait = verrier_astronomer
				}
			}
			verrier_add_x_astronomical_objects_effect = {
				AMOUNT = 5
			}
		}
		if = {
			limit = {
				has_variable = verrier_astronomer_var
				var:verrier_astronomer_var = {
					has_trait = verrier_astronomer_experienced
				}
			}
			verrier_add_x_astronomical_objects_effect = {
				AMOUNT = 6
			}
		}
		if = {
			limit = {
				has_variable = verrier_astronomer_var
				var:verrier_astronomer_var = {
					has_trait = verrier_astronomer_famous
				}
			}
			verrier_add_x_astronomical_objects_effect = {
				AMOUNT = 8
			}
		}
		if = {
			limit = {
				has_variable = verrier_yearly_astronomical_objects_var
			}
			set_variable = {
				name = verrier_astronomical_objects_notification_var
				value = var:verrier_yearly_astronomical_objects_var
			}
			set_variable = {
				name = verrier_yearly_astronomical_objects_var
				value = 0
			}
		}
		if = {
			limit = {
				verrier_astronomical_observation = yes #Notification
			}
			post_notification = verrier_astronomical_observation_notification
		}
	}
}

#Verrier 6: Dissolve Observatory

verrier.6 = {
	dlc = dlc_morgenrote

	type = country_event

	title = verrier.6.t
	desc = verrier.6.d
	flavor = verrier.6.f

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_comet_1881.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_astronomy.dds"

	trigger = {
		any_scope_state = {
			is_capital = no
			has_building = building_verrier_observatory
		}
	}

	immediate = {
	}

	option = {
		name = verrier.6.g
		default_option = yes

		trigger = {
			any_scope_state = {
				is_capital = yes
				has_building = building_verrier_observatory
			}
		}
		verrier_add_x_astronomical_objects_effect = {
			AMOUNT = 3
		}
		verrier_small_add_technology = yes
		every_scope_state = {
    		limit = { 
				has_building = building_verrier_observatory
				is_capital = no
			}
    		remove_building = building_verrier_observatory
  		}
	}

	option = {
		name = verrier.6.h

		trigger = {
			any_scope_state = {
				is_capital = yes
				NOT = {
					has_building = building_verrier_observatory
				}
			}
		}
		every_scope_state = {
    		limit = { 
				is_capital = yes
			}
    		create_building={
				building="building_verrier_observatory"
				level=1
				reserves=1
			}
			activate_building = {
				building = "building_verrier_observatory"
			}
  		}
		if = {
			limit = {
				any_scope_state = {
					is_production_method_active = {
						building_type = building_verrier_observatory
						production_method = pm_gaudi_magnetic_observatory
					}
				}
			}
			capital = {
				activate_production_method = {
					building_type = building_verrier_observatory
					production_method = pm_gaudi_magnetic_observatory
				}
			}
		}
		every_scope_state = {
    		limit = { 
				has_building = building_verrier_observatory
				is_capital = no
			}
    		remove_building = building_verrier_observatory
  		}
	}
}

#Verrier 10: Neptune discovered

verrier.10 = {
	dlc = dlc_morgenrote

	type = country_event

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_neptune.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_astronomy.dds"

	immediate = {
		set_global_variable = verrier_neptune_discovered_var
		set_global_variable = {
			name = verrier_discoverer_neptune_country_global_var
			value = this
		}
		set_global_variable = {
			name = verrier_neptune_year_global_var
			value = year
		}
		country_definition = {
			set_global_variable = {
				name = verrier_discoverer_neptune_global_var
				value = this
			}
		}
		set_variable = verrier_neptun_achievement
		if = {
			limit = {
				has_variable = verrier_astronomer_var
			}
			var:verrier_astronomer_var = {
				save_scope_as = verrier_astronomer_scope
			}
		}
		else = {
			create_new_astronomer_effect = yes
			verrier_set_new_astronomer_as_astronomer_effect = yes
			var:verrier_astronomer_var = {
				save_scope_as = verrier_astronomer_scope
			}
		}
		country_definition = {
			save_scope_as = verrier_discoverer_neptune_country_definition_scope
			set_global_variable = {
				name = verrier_discoverer_neptune_country_definition_global_var
				value = scope:verrier_discoverer_neptune_country_definition_scope
			}
		}
	}

	title = verrier.10.t
	desc = verrier.10.d
	flavor = verrier.10.f

	duration = 3

	option = {
		name = verrier.10.g
		add_modifier = {
			name = verrier_discovered_neptune_modifier
		}
		scope:verrier_astronomer_scope = {
			add_modifier = {
				name = verrier_neptune_discoverer_modifier
			}
			set_variable = verrier_neptune_discoverer_var
		}
		change_global_variable = {
 			name = verrier_planets_number
 			add = 1
		}
		astronomer_experience_gain_effect = {
			AMOUNT = 100
		}
	}

	after = {
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { this = ROOT }
					is_player = yes
				}
				trigger_event = {
					id = verrier.11
					days = 3
				}
			}
		}
	}
}

#Verrier 9000: A teacher from Michigan attempts to go over the Niagara Falls in a barrel

verrier.9000 = {
	dlc = dlc_morgenrote

	type = country_event
	placement = scope:niagara_falls_state_scope

	title = verrier.9000.t
	desc = verrier.9000.d	
	flavor = verrier.9000.f

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_barrel.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/world_fair"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 1

	trigger = {
		OR = {
			p:x24691A = {
				owner = root
			}
			p:xE217B5 = {
				owner = root
			}
		}
		NOT = {
			has_variable = verrier_niagara_falls_barrel_variable
		}
	}

	immediate = {
		random_state  = { #Sets Ontario or New York as scope
			limit = {
				owner = root
				OR = {
					state_region = s:STATE_NEW_YORK
					state_region = s:STATE_ONTARIO
				}
			}
			save_scope_as = niagara_falls_state_scope
		}
		set_variable = verrier_niagara_falls_barrel_variable
	}

	option = { #Stop this madness immediately!
		name = verrier.9000.g
		default_option = yes

	scope:niagara_falls_state_scope = {
		add_radicals_in_state = {
			value = 0.01
		}
	}

		custom_tooltip = verrier_niagara_falls_not_allow_barrel_tt
	}

	option = { #Not my problem! Let her do whatever she wants...
		name = verrier.9000.h

		custom_tooltip = {
			text = verrier_niagara_falls_allow_barrel_tt
			random_list = {
				1 = {
					trigger_event = {
					id = verrier.9001
					days = { 2 14 }
					}
				}
				1 = {
					trigger_event = {
						id = verrier.9002
						days = { 2 14 }
					}
				}
			}
		}
	}
}
