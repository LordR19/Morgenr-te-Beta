#Yearly astronomical report

verrier.1 = {
	type = country_event
		
	hidden = yes

	trigger = {
		NOT = {
			has_variable = verrier_astronomical_objects_cooldown_var
		}
		verrier_astronomical_observation = yes
	}

	immediate = {
		set_variable = {
			name = verrier_astronomical_objects_cooldown_var
			days = 365
		}
		verrier_add_1_astronomical_object = yes #base objects

		if = {
			limit = {
				has_technology_researched = camera
			}
			verrier_add_2_astronomical_object = yes
		}
		if = {
			limit = {
				has_technology_researched = verrier_electromagnetism_tech
			}
			verrier_add_3_astronomical_object = yes
		}
		if = {
			limit = {
				has_technology_researched = verrier_astronomy_tech
			}
			verrier_add_2_astronomical_object = yes
			verrier_add_2_astronomical_object = yes
		}
		if = {
			limit = {
				has_technology_researched = verrier_astro_physics_tech
			}
			verrier_add_10_astronomical_object = yes
		}
		if = {
			limit = {
				any_scope_character = {
					has_trait = verrier_astronomer
				}
			}
			verrier_add_5_astronomical_object = yes
		}
		if = {
			limit = {
				verrier_astronomical_observation = yes #Notification
			}
			post_notification = verrier_astronomical_observation_notification
		}
	}
}

verrier.5 = {
	type = country_event

	title = verrier.5.t
	desc = verrier.5.d
	flavor = verrier.5.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_london_center.bk2"
	}

	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

		random_scope_state = {
			limit = {
				is_capital = yes
			}

			}
		}
	}

		name = verrier.5.g

		random_scope_state = {
			limit = {
				is_capital = yes
			}
			start_building_construction = building_verrier_observatory
		}
	}
}

#Verrier 10: Neptune discovered

verrier.10 = {
	type = country_event

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_neptune.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_astronomy.dds"

	immediate = {
		set_global_variable = verrier_neptune_discovered_var
		set_variable = verrier_neptun_achievement
		random_scope_character = {
			limit = {
				OR = {
					has_trait = verrier_astronomer
					has_trait = innovative
					has_trait = meticulous
					has_trait = ambitious
					has_trait = persistent
				}
			}
			save_scope_as = verrier_neptune_discoverer
		}
	}

	title = verrier.10.t
	desc = verrier.10.d
	flavor = verrier.10.f

	duration = 3

	option = {
		name = verrier.10.g
		add_modifier = {
			name = verrier_discovered_neptune_modifier
		}
		scope:verrier_neptune_discoverer = {
			add_modifier = {
				name = verrier_neptune_discoverer_modifier
			}
			if = {
				limit = {
					NOT = {
						has_trait = verrier_astronomer
					}
				}
				add_trait = verrier_astronomer
			}
		}
		change_global_variable = {
 			name = verrier_planets_number
 			add = 1
		}
	}

	after = {
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { this = ROOT }
					is_player = yes
				}
				trigger_event = {
					id = verrier.11
					days = 3
				}
			}
		}
	}
}

#Verrier 31: Should Ceres remain a planet?

verrier.31 = {
	type = country_event

	title = verrier.31.t
	desc = verrier.31.d
	flavor = verrier.31.f

	duration = 1

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_old_solar_system.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_astronomy.dds"

	immediate = {
		random_country = {
			limit = {
				has_modifier = verrier_congress_host_modifier
			}
			save_scope_as = verrier_congress_host_country
		}
	}

	option = {
		name = verrier.31.g #Ceres is a planet
		change_global_variable = {
			name = verrier_ceres_yes
			add = 1
		}
		ai_chance = {
			base = 40
			modifier = { #Want to defend prestige
				trigger = {
					OR = {
						has_modifier = verrier_discovered_ceres_modifier
						has_modifier = verrier_discovered_pallas_modifier
						has_modifier = verrier_discovered_juno_modifier
						has_modifier = verrier_discovered_vesta_modifier
					}
				}
				add = 100
			}
			modifier = { #No reason to reform if lesser objects
				trigger = {
					global_var:verrier_total_verrier_great_comet <= 2000
				}
				add = 25
			}	
			modifier = { #No reason to reform if lesser objects
				trigger = {
					global_var:verrier_total_verrier_great_comet <= 2500
				}
				add = 20
			}	
			modifier = { #If the player is the host and wants to preserve Ceres, it'll be 50:50 instead 60:40 against Ceres
				trigger = {
					any_country = {
						is_player = yes 
						has_modifier = verrier_congress_host_modifier
					}
				}
				add = 20
			}							
		}
	}

	option = { #Ceres is not a planet
		name = verrier.31.h
		default_option = yes
		change_global_variable = {
			name = verrier_ceres_no
			add = 1
		}
		ai_chance = {
			base = 60
			modifier = { #Want to defend prestige and hurt others
				trigger = {
					OR = {
						has_modifier = verrier_discovered_uranus_modifier
						has_modifier = verrier_discovered_neptune_modifier
					}
				}
				add = 100
			}	
			modifier = { #Reason to reform if more objects
				trigger = {
					global_var:verrier_total_verrier_great_comet >= 3500
				}
				add = 10
			}	
			modifier = { #No reason to reform if lesser objects
				trigger = {
					global_var:verrier_total_verrier_great_comet >= 4000
				}
				add = 20
			}	
			modifier = { #Historically, Prussian academics lead the reform to introduce asteroids and demote Ceres, Vesta, Pallas and Juno
				trigger = {
					ROOT = c:PRU
				}
				add = 100
			}								
		}
	}
}

#Verrier 32: Ceres is not a planet anymore/Ceres remains a planet

verrier.32 = {
	type = country_event

	title = {
		triggered_desc = {
			trigger = {
				has_global_variable = verrier_ceres_demoted_var
			}
			desc = verrier.32.t_demoted
		}
		triggered_desc = {
			trigger = {
				has_global_variable = verrier_ceres_saved_var
			}
			desc = verrier.32.t_saved
		}
	}

	desc = {
		triggered_desc = {
			trigger = {
				has_global_variable = verrier_ceres_demoted_var
			}
			desc = verrier.32.d_demoted
		}
		triggered_desc = {
			trigger = {
				has_global_variable = verrier_ceres_saved_var
			}
			desc = verrier.32.d_saved
		}
	}

	flavor = {
		triggered_desc = {
			trigger = {
				has_global_variable = verrier_ceres_demoted_var
			}
			desc = verrier.32.f_demoted
		}
		triggered_desc = {
			trigger = {
				has_global_variable = verrier_ceres_saved_var
			}
			desc = verrier.32.f_saved
		}
	}

	duration = 1

	event_image = {
		texture = "gfx/event_pictures/verrier_pictures/verrier_old_solar_system.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_astronomy.dds"

	immediate = {
		random_country = {
			limit = {
				has_modifier = verrier_congress_host_modifier
			}
			save_scope_as = verrier_congress_host_country
		}
		if = {
			limit = {
				global_var:verrier_ceres_yes < global_var:verrier_ceres_no
			}
			set_global_variable = verrier_ceres_demoted_var
		}
		if = {
			limit = {
				global_var:verrier_ceres_yes > global_var:verrier_ceres_no
			}
			set_global_variable = verrier_ceres_saved_var
		}
		if = {
			limit = {
				global_var:verrier_ceres_yes = global_var:verrier_ceres_no
			}
			set_global_variable = verrier_ceres_saved_var
		}
	}

	option = {
		trigger = {
			has_global_variable = verrier_ceres_demoted_var
		}
		name = verrier.32.g #Ceres has been only a dwarf of a planet
		change_global_variable = {
 			name = verrier_planets_number
 			add = -1
		}
		every_country = {
			limit = {
				OR = {
					has_modifier = verrier_discovered_ceres_modifier
					has_modifier = verrier_discovered_pallas_modifier
					has_modifier = verrier_discovered_vesta_modifier
					has_modifier = verrier_discovered_juno_modifier
				}
			}
			trigger_event = {
				id = verrier.35
				days = 7
			}
		}
		hidden_effect = {
			random_country = {
				limit = {
					is_player = yes 
				}
				trigger_event = {
					id = verrier.33
					days = 3
				}
			}
		}
	}
	option = {
		trigger = {
			has_global_variable = verrier_ceres_saved_var
		}
		name = verrier.32.h #Ceres is saved!
		default_option = yes
		every_country = {
			limit = {
				OR = {
					has_modifier = verrier_discovered_pallas_modifier
					has_modifier = verrier_discovered_vesta_modifier
					has_modifier = verrier_discovered_juno_modifier
				}
			}
			trigger_event = {
				id = verrier.35
				days = 7
			}
		}
		hidden_effect = {
			random_country = {
				limit = {
					is_player = yes 
				}
				trigger_event = {
					id = verrier.34
					days = 3
				}
			}
		}
	}
}

				trigger_event = {
					id = verrier.61
				}
			}
		}
		else = {
			random_country = {
				limit = {
					is_player = yes 
				}
				set_global_variable = verrier_1843_comet_year_var
				trigger_event = {
					id = verrier.62
				}
			}
		}
	}
}
